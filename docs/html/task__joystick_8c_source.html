53 void task_joystick(taskJoystickConfig_t *param)
54 {
55  //check for param struct
56  if(param == NULL)
57  {
58  ESP_LOGE(LOG_TAG,"param is NULL ");
59  while(1) vTaskDelay(100000/portTICK_PERIOD_MS);
60  return;
61  }
62  //event bits used for pending on debounced buttons
63  EventBits_t uxBits = 0;
64  //calculate array index of EventGroup array (each 4 VB have an own EventGroup)
65  uint8_t evGroupIndex = param->virtualButton / 4;
66  //calculate bitmask offset within the EventGroup
67  uint8_t evGroupShift = param->virtualButton % 4;
68  //save virtualbutton
69  uint vb = param->virtualButton;
70  //final pointer to the EventGroup used by this task
71  EventGroupHandle_t *evGroup = NULL;
72  //ticks between task timeouts
73  const TickType_t xTicksToWait = 2000 / portTICK_PERIOD_MS;
74 
75  //local button mask
76  uint32_t value = param->value;
77  int16_t hat = param->valueS;
78  //action type
79  joystick_action action = param->type;
80  //release mode
81  uint8_t releasemode = param->mode;
82 
83  //do all the eventgroup checking only if this is a persistent task
84  //-> virtualButton is NOT set to VB_SINGLESHOT
85  if(param->virtualButton != VB_SINGLESHOT)
86  {
87  //check for correct offset
88  if(evGroupIndex >= NUMBER_VIRTUALBUTTONS)
89  {
90  ESP_LOGE(LOG_TAG,"virtual button group unsupported: %d ",evGroupIndex);
91  while(1) vTaskDelay(100000/portTICK_PERIOD_MS);
92  return;
93  }
94 
95  //test if event groups are already initialized, otherwise exit immediately
96  while(virtualButtonsOut[evGroupIndex] == 0)
97  {
98  ESP_LOGE(LOG_TAG,"uninitialized event group, retry in 1s");
99  vTaskDelay(1000/portTICK_PERIOD_MS);
100  }
101  //event group intialized, store for later usage
102  evGroup = virtualButtonsOut[evGroupIndex];
103  }
104 
105  if(!((action == BUTTON_PRESS) || (action == BUTTON_RELEASE) || \
106  (action == XAXIS) || (action == YAXIS) || (action == ZAXIS) || \
107  (action == ZROTATE) || (action == SLIDER_LEFT) || \
108  (action == SLIDER_RIGHT) || (action == HAT)))
109  {
110  ESP_LOGE(LOG_TAG,"joystick action unknown: %d ",action);
111  if(vb == VB_SINGLESHOT) return;
112  while(1) vTaskDelay(100000/portTICK_PERIOD_MS);
113  }
114 
115  while(1)
116  {
117  //wait for the flag
118  if(vb != VB_SINGLESHOT)
119  {
120  uxBits = xEventGroupWaitBits(evGroup,(1<<evGroupShift)|(1<<(evGroupShift+4)),pdTRUE,pdFALSE,xTicksToWait);
121  }
122  //test for a valid set flag or trigger if we are using singleshot
123  if((uxBits & (1<<evGroupShift)) || vb == VB_SINGLESHOT)
124  {
125  //press action
126  //set different values of global joystick state accordingly.
127  switch(action)
128  {
129  case BUTTON_PRESS: globalJoystickState.buttonmask |= value; break;
130  case BUTTON_RELEASE: globalJoystickState.buttonmask &= ~value; break;
131  case XAXIS: globalJoystickState.Xaxis = value; break;
132  case YAXIS: globalJoystickState.Yaxis = value; break;
133  case ZAXIS: globalJoystickState.Zaxis = value; break;
134  case ZROTATE: globalJoystickState.Zrotate = value; break;
135  case SLIDER_LEFT: globalJoystickState.sliderLeft = value; break;
136  case SLIDER_RIGHT: globalJoystickState.sliderRight = value; break;
137  case HAT: globalJoystickState.hat = hat; break;
138  }
139 
140  //send either to USB, BLE, both or none
141  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_USB)
142  { xQueueSend(joystick_movement_usb, &globalJoystickState,10); }
143  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_BLE)
144  { xQueueSend(joystick_movement_ble, &globalJoystickState,10); }
145  }
146  //test for vb's release flag or trigger if vb is singleshot
147  if((uxBits & (1<<(evGroupShift+4))) || vb == VB_SINGLESHOT)
148  {
149  //if release flag is used (cannot do for BUTTON_RELEASE)
150  if(releasemode != 0 && action != BUTTON_RELEASE)
151  {
152  //release action
153  //set different values of global joystick state accordingly.
154  switch(action)
155  {
156  case BUTTON_PRESS: globalJoystickState.buttonmask &= ~value; break;
157  case BUTTON_RELEASE: break; //this line is needed, otherwise compile error.
158  case XAXIS: globalJoystickState.Xaxis = 512; break;
159  case YAXIS: globalJoystickState.Yaxis = 512; break;
160  case ZAXIS: globalJoystickState.Zaxis = 512; break;
161  case ZROTATE: globalJoystickState.Zrotate = 512; break;
162  case SLIDER_LEFT: globalJoystickState.sliderLeft = 0; break;
163  case SLIDER_RIGHT: globalJoystickState.sliderRight = 0; break;
164  case HAT: globalJoystickState.hat = -1; break;
165  }
166 
167  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_USB)
168  { xQueueSend(joystick_movement_usb, &globalJoystickState,10); }
169  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_BLE)
170  { xQueueSend(joystick_movement_ble, &globalJoystickState,10); }
171  }
172  }
173 
174  //function tasks in single shot mode MUST return to its caller
175  if(vb == VB_SINGLESHOT) return;
176  }
177 }

209 esp_err_t task_joystick_getAT(char* output, void* cfg)
210 {
211  taskJoystickConfig_t *conf = (taskJoystickConfig_t *)cfg;
212  if(conf == NULL) return ESP_FAIL;
213 
214  //set mode to defined value (if it is messed up)
215  uint8_t releasemode = 0;
216  if(conf->mode != 0) releasemode = 1;
217 
218  //determine type of action and set output buffer accordingly
219  switch(conf->type)
220  {
221  case BUTTON_PRESS:
222  //special for buttons: to be compatible, it is a new command
223  //if button press is set together with releasemode (button click)
224  if(releasemode == 0)
225  {
226  sprintf(output,"AT JP %d",conf->value);
227  } else {
228  sprintf(output,"AT JC %d",conf->value);
229  }
230  break;
231  case BUTTON_RELEASE: sprintf(output,"AT JR %d",conf->value); break;
232 
233  //all following types have the releasmode in the AT command
234  case XAXIS: sprintf(output,"AT JX %d %d",conf->value,releasemode); break;
235  case YAXIS: sprintf(output,"AT JY %d %d",conf->value,releasemode); break;
236  case ZAXIS: sprintf(output,"AT JZ %d %d",conf->value,releasemode); break;
237  case ZROTATE: sprintf(output,"AT JT %d %d",conf->value,releasemode); break;
238  case SLIDER_LEFT: sprintf(output,"AT JS %d %d",conf->value,releasemode); break;
239  case SLIDER_RIGHT: sprintf(output,"AT JU %d %d",conf->value,releasemode); break;
240  case HAT: sprintf(output,"AT JH %d %d",conf->valueS,releasemode); break;
241 
242  default: return ESP_FAIL; //unknown mode, return fail
243  }
244  //fine fine fine, return ESP_OK (AT command found and buffer filled).
245  return ESP_OK;
246 }
