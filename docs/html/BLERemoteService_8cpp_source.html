140 BLERemoteCharacteristic* BLERemoteService::getCharacteristic(BLEUUID uuid) {
141 // Design
142 // ------
143 // We wish to retrieve the characteristic given its UUID. It is possible that we have not yet asked the
144 // device what characteristics it has in which case we have nothing to match against. If we have not
145 // asked the device about its characteristics, then we do that now. Once we get the results we can then
146 // examine the characteristics map to see if it has the characteristic we are looking for.
147  if (!m_haveCharacteristics) {
148  retrieveCharacteristics();
149  }
150  std::string v = uuid.toString();
151  for (auto &myPair : m_characteristicMap) {
152  if (myPair.first == v) {
153  return myPair.second;
154  }
155  }
156  throw new BLEUuidNotFoundException();
157 } // getCharacteristic

165 void BLERemoteService::retrieveCharacteristics() {
166 
167  ESP_LOGD(LOG_TAG, ">> getCharacteristics() for service: %s", getUUID().toString().c_str());
168 
169  removeCharacteristics(); // Forget any previous characteristics.
170 
171  uint16_t offset = 0;
172  esp_gattc_char_elem_t result;
173  while(1) {
174  uint16_t count = 1;
175  esp_gatt_status_t status = ::esp_ble_gattc_get_all_char(
176  getClient()->getGattcIf(),
177  getClient()->getConnId(),
178  m_startHandle,
179  m_endHandle,
180  &result,
181  &count,
182  offset
183  );
184 
185  if (status == ESP_GATT_INVALID_OFFSET) { // We have reached the end of the entries.
186  break;
187  }
188 
189  if (status != ESP_GATT_OK) { // If we got an error, end.
190  ESP_LOGE(LOG_TAG, "esp_ble_gattc_get_all_char: %s", BLEUtils::gattStatusToString(status).c_str());
191  break;
192  }
193 
194  if (count == 0) { // If we failed to get any new records, end.
195  break;
196  }
197 
198  ESP_LOGD(LOG_TAG, "Found a characteristic: Handle: %d, UUID: %s", result.char_handle, BLEUUID(result.uuid).toString().c_str());
199 
200  // We now have a new characteristic ... let us add that to our set of known characteristics
201  BLERemoteCharacteristic *pNewRemoteCharacteristic = new BLERemoteCharacteristic(
202  result.char_handle,
203  BLEUUID(result.uuid),
204  result.properties,
205  this
206  );
207 
208  m_characteristicMap.insert(std::pair<std::string, BLERemoteCharacteristic*>(pNewRemoteCharacteristic->getUUID().toString(), pNewRemoteCharacteristic));
209 
210  offset++; // Increment our count of number of descriptors found.
211  } // Loop forever (until we break inside the loop).
212 
213  m_haveCharacteristics = true; // Remember that we have received the characteristics.
214  ESP_LOGD(LOG_TAG, "<< getCharacteristics()");
215 } // getCharacteristics
