169 void fat_serve(char* resource, struct netconn *conn) {
170  //basepath + 8.3 file + folder + margin
171  char file[sizeof(base_path)+32];
172  sprintf(file,"%s%s",base_path,resource);
173  ESP_LOGI(LOG_TAG,"serving from FAT: %s",file);
174 
175  struct stat st;
176  if (stat(file, &st) == 0) {
177  netconn_write(conn, http_html_hdr, sizeof(http_html_hdr) - 1, NETCONN_NOCOPY);
178 
179  // open the file for reading
180  FILE* f = fopen(file, "r");
181  if(f == NULL) {
182  ESP_LOGE(LOG_TAG,"Unable to open the file %s\n", file);
183  return;
184  }
185 
186  //allocate a buffer in size of one FAT block
187  char* buffer = malloc(512);
188 
189  int i=0, len=0;
190 
191  do {
192  len=fread(buffer, 1, 512, f);
193  i+=len;
194  ESP_LOGV(LOG_TAG,"Writing bytes %d\n", i);
195  netconn_write(conn, buffer, len, NETCONN_NOCOPY);
196  } while (len==512);
197 
198  ESP_LOGI(LOG_TAG,"Sent %u bytes",i);
199  fclose(f);
200  fflush(stdout);
201  free(buffer);
202  }
203  else {
204  ESP_LOGW(LOG_TAG,"Resource not found: %s\n", file);
205  netconn_write(conn, http_404_hdr, sizeof(http_404_hdr) - 1, NETCONN_NOCOPY);
206  }
207 }

216 static void http_server_netconn_serve(struct netconn *conn) {
217 
218  struct netbuf *inbuf;
219  char *buf;
220  u16_t buflen;
221  err_t err;
222 
223  err = netconn_recv(conn, &inbuf);
224 
225  if (err == ERR_OK) {
226 
227  // get the request and terminate the string
228  netbuf_data(inbuf, (void**)&buf, &buflen);
229  buf[buflen] = '\0';
230  char *request_line = strtok(buf, "\n");
231 
232  if(request_line) {
233  // default page -> redirect to index.html
234  if(strstr(request_line, "GET / ")) {
235  fat_serve("/INDEX.HTM", conn);
236  } else {
237  // get the requested resource
238  char* method = strtok(request_line, " ");
239  //get rid of "unused" variable.
240  //strtok although needs this call.
241  (void)(method);
242  char* resource = strtok(NULL, " ");
243  fat_serve(resource, conn);
244  }
245  }
246  }
247 
248  // close the connection and free the buffer
249  netconn_close(conn);
250  netbuf_delete(inbuf);
251 }

296 static esp_err_t wifi_event_handler(void *ctx, system_event_t *event) {
297  //check if event flags are active
298  if(connectionRoutingStatus == NULL)
299  {
300  ESP_LOGE(LOG_TAG,"Status flags for wifi are NULL! Should not receive an event!");
301  return ESP_FAIL;
302  }
303 
304  //determine event: print information and set flags accordingly
305  switch(event->event_id) {
306 
307  case SYSTEM_EVENT_STA_START:
308  esp_wifi_connect();
309  break;
310 
311  case SYSTEM_EVENT_AP_START:
312  ESP_LOGD(LOG_TAG,"Access point started");
313  xEventGroupSetBits(connectionRoutingStatus, WIFI_ACTIVE);
314  //start off timer (switch off wifi, if no client connects in timeout)
315  wifiStartStopOffTimer(1);
316  break;
317 
318  case SYSTEM_EVENT_AP_STOP:
319  ESP_LOGD(LOG_TAG,"Access point stopped");
320  xEventGroupClearBits(connectionRoutingStatus, WIFI_ACTIVE);
321  break;
322 
323  case SYSTEM_EVENT_AP_STACONNECTED:
324  //a new client connected
325  ESP_LOGI(LOG_TAG,"Client connected, currently connected: %d",getNumberOfWifiStations());
326  //set client connected flag
327  xEventGroupSetBits(connectionRoutingStatus, WIFI_CLIENT_CONNECTED);
328 
329  //add the websocket sending functions to hal_serial for getting output data
330  halSerialAddOutputStream(WS_write_data);
331 
332  //if the wifi timer is active, disable the timer (client is connected)
333  wifiStartStopOffTimer(0);
334  break;
335 
336  case SYSTEM_EVENT_AP_STADISCONNECTED:
337  //a client got disconnected
338  ESP_LOGI(LOG_TAG,"Client disconnected, currently connected: %d",getNumberOfWifiStations());
339  if(getNumberOfWifiStations() == 0)
340  {
341  //remove the callback, not necessary if no client is connected
342  halSerialRemoveOutputStream();
343  //clear client connected flag
344  xEventGroupClearBits(connectionRoutingStatus, WIFI_CLIENT_CONNECTED);
345  //start the auto-disable timer
346  wifiStartStopOffTimer(1);
347  }
348  break;
349  default:
350  break;
351  }
352  return ESP_OK;
353 }
