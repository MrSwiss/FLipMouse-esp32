156  }
157

159  {
160  ESP_LOGE(LOG_TAG,"Cannot use singleshot task as VB FUNCTIONAL task!");
161  return ESP_FAIL;
162  }
163

165  currentTaskParametersUpdate[vb] = param;
166  //set new command type
167  currentConfigLoaded.virtualButtonCommand[vb] = type;
168  //trigger config update

187 void configSwitcherTask(void * params)
188 {
189  char command[SLOTNAME_LENGTH];
190  char tasksignature[8];
191  generalConfig_t currentConfig;
192 
193  uint32_t tid = 0;
194  size_t vbparametersize = 0;
195  TaskFunction_t newTaskCode = NULL;
196  short stacksize = 0;
197  esp_err_t ret;
198 
199  if(config_switcher == 0)
200  {
201  ESP_LOGE(LOG_TAG,"config_switcher queue uninitialized, exiting.");
202  vTaskDelete(NULL);
203  }
204 
205  //reset all allocated task parameter pointers
206  for(uint8_t i = 0;i<NUMBER_VIRTUALBUTTONS*4;i++)
207  {
208  currentTaskParameters[i] = NULL;
209  }
210 
211  while(1)
212  {
213  //wait for a command.
214  vTaskDelay(1000/portTICK_PERIOD_MS);
215  if(xQueueReceive(config_switcher,command,1000/portTICK_PERIOD_MS) == pdTRUE)
216  {
217  //request storage access
218  if(halStorageStartTransaction(&tid,20) != ESP_OK)
219  {
220  ESP_LOGE(LOG_TAG,"Cannot start storage transaction");
221  continue;
222  }
223 
224  //command received, load new slot:
225  //__NEXT, __PREV, __DEFAULT, __UPDATE, __RESTOREFACTORY
226  if(strcmp(command,"__NEXT") == 0)
227  {
228  ESP_LOGD(LOG_TAG,"Load next slot");
229  ret = halStorageLoad(NEXT,&currentConfig,tid);
230  } else if(strcmp(command,"__PREV") == 0) {
231  ESP_LOGD(LOG_TAG,"Load prev slot");
232  ret = halStorageLoad(PREV,&currentConfig,tid);
233  } else if(strcmp(command,"__DEFAULT") == 0) {
234  ret = halStorageLoad(DEFAULT,&currentConfig,tid);
235  //for default, we try here again (in the case we had to build a new
236  //fresh default config
237  if(ret != ESP_OK) ret = halStorageLoad(DEFAULT,&currentConfig,tid);
238  ESP_LOGD(LOG_TAG,"loading default");
239  } else if(strcmp(command,"__RESTOREFACTORY") == 0) {
240  ret = halStorageDeleteSlot(0,tid);
241 
242  halStorageCreateDefault(tid);
243  halStorageFinishTransaction(tid);
244  if(ret != ESP_OK)
245  {
246  ESP_LOGE(LOG_TAG,"Error deleting all slots");
247  } else {
248  ESP_LOGW(LOG_TAG,"Deleted all slots");
249  }
250  continue;
251  } else if(strcmp(command,"__UPDATE") == 0) {
252  ESP_LOGD(LOG_TAG,"config update");
253  //on config updates, use the currentConfigLoaded,
254  //it is possible the currentConfigLoaded was changed by external parts
255  memcpy(&currentConfig,&currentConfigLoaded,sizeof(generalConfig_t));
256  ret = ESP_OK;
257  } else {
258  ret = halStorageLoadName(command,&currentConfig,tid);
259  ESP_LOGD(LOG_TAG,"Load by name: %s",command);
260  }
261 
262  if(ret != ESP_OK)
263  {
264  halStorageFinishTransaction(tid);
265  ESP_LOGE(LOG_TAG,"Error loading general slot config!");
266  continue;
267  }
268 
269  //reload ADC
270  if(halAdcUpdateConfig(&currentConfig.adc) != ESP_OK)
271  {
272  ESP_LOGE(LOG_TAG,"error reloading adc config");
273  }
274 
275  //set other config infos
276  ESP_LOGD(LOG_TAG,"setting connection bits (USB: %d, BLE: %d)",currentConfig.usb_active,currentConfig.ble_active);
277  if(currentConfig.ble_active != 0) xEventGroupSetBits(connectionRoutingStatus,DATATO_BLE);
278  else xEventGroupClearBits(connectionRoutingStatus,DATATO_BLE);
279  if(currentConfig.usb_active != 0) xEventGroupSetBits(connectionRoutingStatus,DATATO_USB);
280  else xEventGroupClearBits(connectionRoutingStatus,DATATO_USB);
281 
282  //reset HID channels (USB&BLE)
283  halBLEReset(0);
284 
285  //perform task switching...
286  for(uint8_t i = 0; i<(NUMBER_VIRTUALBUTTONS*4); i++)
287  {
288  //unload
289  if(currentTasks[i] != NULL) vTaskDelete(currentTasks[i]);
290  //if memory was previously allocated, free now
291  if(currentTaskParameters[i] != NULL) free(currentTaskParameters[i]);
292 
293  //determine which tasks should be loaded
294  switch(currentConfig.virtualButtonCommand[i])
295  {
296  case T_MOUSE:
297  newTaskCode = (TaskFunction_t)task_mouse;
298  currentTaskParameters[i] = malloc(sizeof(taskMouseConfig_t));
299  vbparametersize = sizeof(taskMouseConfig_t);
300  stacksize = TASK_MOUSE_STACKSIZE;
301  ESP_LOGD(LOG_TAG,"creating new mouse task on VB %d",i);
302  break;
303  case T_KEYBOARD:
304  newTaskCode = (TaskFunction_t)task_keyboard;
305  currentTaskParameters[i] = malloc(sizeof(taskKeyboardConfig_t));
306  vbparametersize = sizeof(taskKeyboardConfig_t);
307  stacksize = TASK_KEYBOARD_STACKSIZE;
308  ESP_LOGD(LOG_TAG,"creating new keyboard task on VB %d",i);
309  break;
310  case T_CONFIGCHANGE:
311  newTaskCode = (TaskFunction_t)task_configswitcher;
312  currentTaskParameters[i] = malloc(sizeof(taskConfigSwitcherConfig_t));
313  vbparametersize = sizeof(taskConfigSwitcherConfig_t);
314  stacksize = TASK_CONFIGSWITCHER_STACKSIZE;
315  ESP_LOGD(LOG_TAG,"creating new configchange task on VB %d",i);
316  break;
317  case T_CALIBRATE:
318  newTaskCode = (TaskFunction_t)task_calibration;
319  currentTaskParameters[i] = malloc(sizeof(taskNoParameterConfig_t));
320  vbparametersize = sizeof(taskNoParameterConfig_t);
321  stacksize = TASK_CALIB_STACKSIZE;
322  ESP_LOGD(LOG_TAG,"creating new calibrate task on VB %d",i);
323  break;
324  case T_SENDIR:
325  newTaskCode = NULL;
326  //allocate just a little bit to have a pointer
327  currentTaskParameters[i] = malloc(sizeof(uint8_t));
328  vbparametersize = 0;
329  ESP_LOGD(LOG_TAG,"creating new IR task on VB %d",i);
330  break;
331  case T_NOFUNCTION:
332  newTaskCode = NULL;
333  //allocate just a little bit to have a pointer
334  currentTaskParameters[i] = malloc(sizeof(uint8_t));
335  vbparametersize = 0;
336  break;
337  default:
338  newTaskCode = NULL;
339  ESP_LOGE(LOG_TAG,"unkown button command, cannot load task");
340  break;
341  }
342 
343  //check if we allocated memory:
344  if(currentTaskParameters[i] != NULL)
345  {
346  currentConfig.virtualButtonConfig[i] = currentTaskParameters[i];
347  //load VB config to memory
348  if(vbparametersize > 0)
349  {
350  //is there a pointer available for update?
351  if(currentTaskParametersUpdate[i] == NULL)
352  {
353  //if no, load VB config from storage
354  halStorageLoadGetVBConfigs(i,currentConfig.virtualButtonConfig[i], vbparametersize, tid);
355  } else {
356  //if yes, use this pointer for task parameter
357  currentTaskParameters[i] = currentTaskParametersUpdate[i];
358  //and reset the update pointer
359  currentTaskParametersUpdate[i] = NULL;
360  }
361  }
362  } else {
363  ESP_LOGE(LOG_TAG,"error allocating memory for VB%u config",i);
364  }
365 
366  //create task name
367  sprintf(tasksignature,"vb%d",i);
368 
369  //finally load new task
370  if(newTaskCode != NULL)
371  {
372  if(xTaskCreate(newTaskCode,tasksignature,stacksize,
373  currentConfig.virtualButtonConfig[i],HAL_VB_TASK_PRIORITY,
374  &currentTasks[i]) != pdPASS)
375  {
376  ESP_LOGE(LOG_TAG,"error creating new virtualbutton task");
377  }
378  } else {
379  currentTasks[i] = NULL;
380  ESP_LOGD(LOG_TAG,"unused virtual button %d",i);
381  }
382  }
383 
384  //clean up
385  halStorageFinishTransaction(tid);
386  //save newly loaded cfg
387  memcpy(&currentConfigLoaded,&currentConfig,sizeof(generalConfig_t));
388  tid = 0;
389  ESP_LOGD(LOG_TAG,"----------Config Switch Complete----------");
390  }
391  }
392 }

401 void task_configswitcher(taskConfigSwitcherConfig_t *param)
402 {
403  EventBits_t uxBits;
404  //check for param struct
405  if(param == NULL)
406  {
407  ESP_LOGE(LOG_TAG,"param is NULL ");
408  vTaskDelete(NULL);
409  return;
410  }
411  //calculate array index of EventGroup array (each 4 VB have an own EventGroup)
412  uint8_t evGroupIndex = param->virtualButton / 4;
413  //calculate bitmask offset within the EventGroup
414  uint8_t evGroupShift = param->virtualButton % 4;
415  //final pointer to the EventGroup used by this task
416  EventGroupHandle_t *evGroup;
417  //ticks between task timeouts
418  const TickType_t xTicksToWait = 2000 / portTICK_PERIOD_MS;
419 
420  //check for valid parameter
421  if(evGroupIndex >= NUMBER_VIRTUALBUTTONS)
422  {
423  ESP_LOGE(LOG_TAG,"virtual button group unsupported: %d ",evGroupIndex);
424  vTaskDelete(NULL);
425  return;
426  }
427 
428  //test if event groups are already initialized, otherwise exit immediately
429  if(virtualButtonsOut[evGroupIndex] == 0)
430  {
431  ESP_LOGE(LOG_TAG,"uninitialized event group for virtual buttons, quitting this task");
432  vTaskDelete(NULL);
433  return;
434  } else {
435  evGroup = virtualButtonsOut[evGroupIndex];
436  }
437 
438  while(1)
439  {
440  //wait for the flag
441  uxBits = xEventGroupWaitBits(evGroup,(1<<evGroupShift),pdTRUE,pdFALSE,xTicksToWait);
442  if(uxBits & (1<<evGroupShift))
443  {
444  xQueueSend(config_switcher,&param->slotName,0);
445  ESP_LOGD(LOG_TAG,"requesting slot switch by name: %s",param->slotName);
446 
447  }
448  }
449 }

456 esp_err_t configSwitcherInit(void)
457 {
458  //test if the config_switcher queue is initialized...
459  if(config_switcher == NULL)
460  {
461  ESP_LOGE(LOG_TAG,"Error init config switcher, please create \
462  config_switcher queue before calling configSwitcherInit()!");
463  return ESP_FAIL;
464  }
465 
466  //reset parameters memory
467  for(uint8_t i = 0; i < NUMBER_VIRTUALBUTTONS*4; i++)
468  {
469  currentTaskParameters[i] = NULL;
470  currentTaskParametersUpdate[i] = NULL;
471  }
472 
473  //start configSwitcherTask
474  if(xTaskCreate(configSwitcherTask,"configswitcher",CONFIGSWITCHERTASK_PERMANENT_STACKSIZE,(void *)NULL,
475  HAL_CONFIG_TASK_PRIORITY,configswitcher_handle) != pdPASS)
476  {
477  ESP_LOGE(LOG_TAG,"error creating config switcher task, cannot proceed.");
478  } else {
479  ESP_LOGD(LOG_TAG,"configSwitcherTask created");
480  }
481 
482  //load the default slot by sending "__DEFAULT" to
483  // the config_switcher queue
484  char commandname[SLOTNAME_LENGTH];
485  strcpy(commandname,"__DEFAULT");
486  if(xQueueSend(config_switcher,commandname,10) == pdPASS)
487  {
488  ESP_LOGD(LOG_TAG,"requesting default slot switch on startup");
489  } else {
490  ESP_LOGE(LOG_TAG,"Error requesting default slot switch");
491  }
492 
493  return ESP_OK;
494 }
