73 void app_main()
74 {
75  //enter critical section & suspend all tasks for initialising
76  vTaskSuspendAll();
77  //init all remaining rtos stuff
78  //eventgroups
79  connectionRoutingStatus = xEventGroupCreate();
80  //init flags if not created
81  for(uint8_t i = 0; i<NUMBER_VIRTUALBUTTONS; i++)
82  {
83  virtualButtonsIn[i] = xEventGroupCreate();
84  virtualButtonsOut[i] = xEventGroupCreate();
85  }
86  //queues
87  keyboard_usb_press = xQueueCreate(10,sizeof(uint16_t));
88  keyboard_usb_release = xQueueCreate(10,sizeof(uint16_t));
89  keyboard_ble_press = xQueueCreate(10,sizeof(uint16_t));
90  keyboard_ble_release = xQueueCreate(10,sizeof(uint16_t));
91  mouse_movement_usb = xQueueCreate(10,sizeof(mouse_command_t));
92  mouse_movement_ble = xQueueCreate(10,sizeof(mouse_command_t));
93  joystick_movement_usb = xQueueCreate(10,sizeof(joystick_command_t)); //TBD: right size?
94  joystick_movement_ble = xQueueCreate(10,sizeof(joystick_command_t)); //TBD: right size?
95  config_switcher = xQueueCreate(5,sizeof(char)*SLOTNAME_LENGTH);
96  //exit critical section & resume all tasks for initialising
97  xTaskResumeAll();
98 
99  //start debouncer
100  if(xTaskCreate(task_debouncer,"debouncer",TASK_DEBOUNCER_STACKSIZE,
101  (void*)NULL,DEBOUNCER_TASK_PRIORITY, NULL) == pdPASS)
102  {
103  ESP_LOGD(LOG_TAG,"created new debouncer task");
104  } else {
105  ESP_LOGE(LOG_TAG,"error creatin new debouncer task");
106  }
107 
108  //start adc continous task
109  if(halAdcInit(NULL) == ESP_OK)
110  {
111  ESP_LOGD(LOG_TAG,"initialized halAdcInit");
112  } else {
113  ESP_LOGE(LOG_TAG,"error initializing halAdcInit");
114  }
115  //start IO continous task
116  if(halIOInit() == ESP_OK)
117  {
118  ESP_LOGD(LOG_TAG,"initialized halIOInit");
119  } else {
120  ESP_LOGE(LOG_TAG,"error initializing halIOInit");
121  }
122  //start config switcher
123  if(configSwitcherInit() == ESP_OK)
124  {
125  ESP_LOGD(LOG_TAG,"initialized configSwitcherInit");
126  } else {
127  ESP_LOGE(LOG_TAG,"error initializing configSwitcherInit");
128  }
129  //start config switcher
130  //TODO: load locale...
131  if(halBLEInit(0) == ESP_OK)
132  {
133  ESP_LOGD(LOG_TAG,"initialized halBle");
134  } else {
135  ESP_LOGE(LOG_TAG,"error initializing halBle");
136  }
137 
138  //initialize serial communication task
139  //(USB-HID & USB Serial for commands)
140  if(halSerialInit() == ESP_OK)
141  {
142  ESP_LOGD(LOG_TAG,"initialized halSerial");
143  } else {
144  ESP_LOGE(LOG_TAG,"error initializing halSerial");
145  }
146 
147  //command parser
148  if(taskCommandsInit() == ESP_OK)
149  {
150  ESP_LOGD(LOG_TAG,"initialized taskCommands");
151  } else {
152  ESP_LOGE(LOG_TAG,"error initializing taskCommands");
153  }
154 
155  //calibrate directly after start-up
156  halAdcCalibrate();
157 
158  //delete this task after initializing.
159  ESP_LOGI(LOG_TAG,"Finished intializing, deleting app_main");
160  vTaskDelete(NULL);
161 }
