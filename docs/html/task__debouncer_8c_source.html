180 void debouncerCallback(TimerHandle_t xTimer) {
181  //get own ID (virtual button)
182  uint8_t virtualButton = (uint32_t) pvTimerGetTimerID(xTimer);
183  int8_t timerindex = isDebouncerActive(virtualButton);
184  generalConfig_t *cfg = configGetCurrent();
185  uint16_t deadtime = 0;
186  int8_t timerId;
187 
188  if(cfg == NULL)
189  {
190  ESP_LOGE(LOG_TAG,"Cannot do deadtime for VB %d, global config is NULL!",virtualButton);
191  deadtime = 0;
192  } else {
193  //check if an individual deadtime is set
194  if(cfg->debounce_idle_vb[virtualButton] != 0) deadtime = cfg->debounce_idle_vb[virtualButton];
195  //check if we have to use global deadtime
196  if(deadtime == 0 && cfg->debounce_idle != 0) deadtime = cfg->debounce_idle;
197  }
198 
199  if(timerindex == -1)
200  {
201  ESP_LOGE(LOG_TAG,"Major error: debounceCallback but no timer for this button %d",virtualButton);
202  return;
203  }
204 
205  switch(xTimerDirection[timerindex])
206  {
207  case TIMER_IDLE:
208  ESP_LOGE(LOG_TAG,"Major error: debounceCallback but timer should be idle %d",virtualButton);
209  return;
210  case TIMER_PRESS:
211  ESP_LOGD(LOG_TAG,"Debounce finished, map in to out for press VB%d",virtualButton);
212  //map in to out and clear from in...
213  xEventGroupSetBits(virtualButtonsOut[virtualButton/4],(1<<(virtualButton%4)));
214  xEventGroupClearBits(virtualButtonsIn[virtualButton/4],(1<<(virtualButton%4)));
215  //stop timer
216  if(cancelTimer(virtualButton) == -1) ESP_LOGE(LOG_TAG,"Cannot cancel timer!");
217  //send feedback to host, if enabled
218  sendButtonLearn(virtualButton,0,cfg);
219  break;
220  case TIMER_RELEASE:
221  ESP_LOGD(LOG_TAG,"Debounce finished, map in to out for release VB%d",virtualButton);
222  //map in to out and clear from in...
223  xEventGroupSetBits(virtualButtonsOut[virtualButton/4],(1<<((virtualButton%4)+4)));
224  xEventGroupClearBits(virtualButtonsIn[virtualButton/4],(1<<((virtualButton%4)+4)));
225  //stop timer
226  if(cancelTimer(virtualButton) == -1) ESP_LOGE(LOG_TAG,"Cannot cancel timer!");
227  //send feedback to host, if enabled
228  sendButtonLearn(virtualButton,1,cfg);
229  return;
230  case TIMER_DEADTIME:
231  ESP_LOGE(LOG_TAG,"Deadtime finished, ready");
232  if(cancelTimer(virtualButton) == -1) ESP_LOGE(LOG_TAG,"Cannot cancel timer!");
233  return;
234  }
235  //if necessary, start timer again for deadtime.
236  if(deadtime != 0)
237  {
238  timerId = startTimer(virtualButton,deadtime);
239  if(timerId == -1)
240  {
241  ESP_LOGE(LOG_TAG,"Cannot start timer...");
242  } else {
243  ESP_LOGD(LOG_TAG,"Deadtime start for VB %d",virtualButton);
244  xTimerDirection[timerId] = TIMER_DEADTIME;
245  }
246  }
247 }

261 int8_t startTimer(uint32_t virtualButton, uint16_t debounceTime)
262 {
263  //check all debouncerchannels for a free slot.
264  for(uint8_t i = 0; i<DEBOUNCERCHANNELS; i++)
265  {
266  //if slot is found
267  if(xTimers[i] == NULL)
268  {
269  //create timer
270  xTimers[i] = xTimerCreate("debouncer",debounceTime / portTICK_PERIOD_MS, \
271  pdFALSE, (void *) virtualButton, debouncerCallback);
272  if(xTimerStart(xTimers[i], 0) != pdPASS)
273  {
274  ESP_LOGE(LOG_TAG,"Cannot start debouncing timer!");
275  }
276  //return array index for further use
277  return i;
278  }
279  }
280  //no slot is found...
281  return -1;
282 }

305 void task_debouncer(void *param)
306 {
307  generalConfig_t *cfg = configGetCurrent();
308 
309  for(uint8_t i =0; i<NUMBER_VIRTUALBUTTONS;i++)
310  {
311  //test if eventgroup is created
312  while(virtualButtonsIn[i] == 0 || virtualButtonsOut[i] == 0)
313  {
314  ESP_LOGE(LOG_TAG,"Eventgroup uninitialized, retry in 1s");
315  vTaskDelay(1000/portTICK_PERIOD_MS);
316  }
317  //set timer direction to idle on startup
318  xTimerDirection[i] = TIMER_IDLE;
319  }
320 
321  //wait until config is valid
322  while(cfg == NULL)
323  {
324  ESP_LOGE(LOG_TAG,"Global config uninitialized, retry in 1s");
325  vTaskDelay(1000/portTICK_PERIOD_MS);
326  cfg = configGetCurrent();
327  }
328 
329  ESP_LOGD(LOG_TAG,"Debouncer started, resolution: %d ms",DEBOUNCE_RESOLUTION_MS);
330 
331  while(1)
332  {
333  //check each eventgroup and start timer accordingly
334  for(uint8_t i = 0; i<NUMBER_VIRTUALBUTTONS; i++)
335  {
336  //check each virtual button, press (0-3) & release (4-7) flag
337  //iterate over all 4 VBs each group
338  for(uint8_t j = 0; j<4;j++)
339  {
340  //TODO TODO: avoid forwarding short edges (short release should not trigger press again).
341 
342 
343  //get in&out event bits and mask for this virtualbutton
344  EventBits_t in = xEventGroupGetBits(virtualButtonsIn[i]) & VB_FLAG_BOTH(j);
345  EventBits_t out = xEventGroupGetBits(virtualButtonsOut[i]) & VB_FLAG_BOTH(j);
346  int8_t timerId = isDebouncerActive(VB_ITER(i,j));
347 
348  //no in flag set, nothing to do.
349  if(in == 0) continue;
350 
351  //if both flags are set, we cannot know which one was first
352  //so clear both flags & don't start/stop any timer
353  if(in == VB_FLAG_BOTH(j))
354  {
355  ESP_LOGW(LOG_TAG,"press&release set for VB %d, discarding",(i*4)+j);
356  xEventGroupClearBits(virtualButtonsIn[i],VB_FLAG_BOTH(j));
357  continue;
358  }
359 
360  //if timer is not running, start one with the corresponding
361  //edge and set xTimerDirection.
362  if(timerId == -1)
363  {
364  //check if this is a press & out flag is unset
365  if((in == VB_FLAG_PRESS(j)) && !(out == VB_FLAG_PRESS(j)))
366  {
367  //check which time to use (either VB, global value or default)
368  uint16_t time = 0;
369  //is a VB value set in config?
370  if(cfg->debounce_press_vb[VB_ITER(i,j)] != 0) time = cfg->debounce_press_vb[VB_ITER(i,j)];
371  //is a global value set in config?
372  if(time == 0 && cfg->debounce_press != 0) time = cfg->debounce_press;
373  //no? just use the default value
374  if(time == 0) time = DEBOUNCETIME_MS;
375 
376  if(time > DEBOUNCE_RESOLUTION_MS)
377  {
378  timerId = startTimer(VB_ITER(i,j),time);
379  if(timerId == -1)
380  {
381  ESP_LOGE(LOG_TAG,"Cannot start timer...");
382  } else {
383  ESP_LOGD(LOG_TAG,"Press debounce started for VB%d",VB_ITER(i,j));
384  xTimerDirection[timerId] = TIMER_PRESS;
385  }
386  continue;
387  //note: currently, following branch is unused, but maybe we need a directly mapped VB.
388  } else {
389  //if no debounce time is used
390  ESP_LOGD(LOG_TAG,"Min. debounce time, just mapping press");
391  //set flag in output eventgroup
392  xEventGroupSetBits(virtualButtonsOut[i],VB_FLAG_PRESS(j));
393  //delete flag in input eventgroup
394  xEventGroupClearBits(virtualButtonsIn[i],VB_FLAG_PRESS(j));
395  continue;
396  }
397  }
398  //check if this is a release & out flag is unset
399  if((in == VB_FLAG_RELEASE(j)) && !(out == VB_FLAG_RELEASE(j)))
400  {
401  //check which time to use (either VB, global value or default)
402  uint16_t time = 0;
403  //is a VB value set in config?
404  if(cfg->debounce_release_vb[VB_ITER(i,j)] != 0) time = cfg->debounce_release_vb[VB_ITER(i,j)];
405  //is a global value set in config?
406  if(time == 0 && cfg->debounce_release != 0) time = cfg->debounce_release;
407  //no? just use the default value
408  if(time == 0) time = DEBOUNCETIME_MS;
409 
410  if(time > DEBOUNCE_RESOLUTION_MS)
411  {
412  timerId = startTimer(VB_ITER(i,j),time);
413  if(timerId == -1)
414  {
415  ESP_LOGE(LOG_TAG,"Cannot start timer...");
416  } else {
417  ESP_LOGD(LOG_TAG,"Release debounce started for VB%d",VB_ITER(i,j));
418  xTimerDirection[timerId] = TIMER_RELEASE;
419  }
420  continue;
421  //note: currently, following branch is unused, but maybe we need a directly mapped VB.
422  } else {
423  //if no debounce time is used
424  ESP_LOGD(LOG_TAG,"Min. debounce time, just mapping release");
425  //set flag in output eventgroup
426  xEventGroupSetBits(virtualButtonsOut[i],VB_FLAG_RELEASE(j));
427  //delete flag in input eventgroup
428  xEventGroupClearBits(virtualButtonsIn[i],VB_FLAG_RELEASE(j));
429  continue;
430  }
431  }
432  } else {
433  //if timer is running, check if this flag requests the
434  //opposite direction OR the same direction is cleared
435  //if yes -> stop & delete this timer
436  switch(xTimerDirection[timerId])
437  {
438  case TIMER_PRESS:
439  //press is cleared OR release is set
440  if( ((in & VB_FLAG_PRESS(j)) == 0) || ((in & VB_FLAG_RELEASE(j)) != 0) )
441  {
442  ESP_LOGD(LOG_TAG,"Press canceled for VB%d",VB_ITER(i,j));
443  if(cancelTimer(VB_ITER(i,j)) == -1) //stop current press debouncer
444  { ESP_LOGE(LOG_TAG,"Cannot cancel press timer!"); }
445  }
446  break;
447  case TIMER_RELEASE:
448  //press is set OR release is cleared
449  if( ((in & VB_FLAG_PRESS(j)) != 0) || ((in & VB_FLAG_RELEASE(j)) == 0) )
450  {
451  ESP_LOGD(LOG_TAG,"Release canceled for VB%d",VB_ITER(i,j));
452  if(cancelTimer(VB_ITER(i,j)) == -1) //stop current release debouncer
453  { ESP_LOGE(LOG_TAG,"Cannot cancel release timer!"); }
454  }
455  break;
456  case TIMER_IDLE:
457  ESP_LOGE(LOG_TAG,"Timer is idle but a valid ID?");
458  break;
459  case TIMER_DEADTIME:
460  ESP_LOGD(LOG_TAG,"Deadtime active, waiting.");
461  break;
462  default:
463  ESP_LOGE(LOG_TAG,"Unknown status in xTimerDirection[%d]",timerId);
464  break;
465  }
466  }
467  }
468  }
469 
470  //wait for defined resolution
471  vTaskDelay(DEBOUNCE_RESOLUTION_MS / portTICK_PERIOD_MS);
472  }
