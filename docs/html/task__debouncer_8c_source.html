65 int8_t cancelTimer(uint8_t virtualButton)
66 {
67  //check all debouncerchannels for a running timer for this
68  //virtual button
69  for(uint8_t i = 0; i<DEBOUNCERCHANNELS; i++)
70  {
71  //if timer is found
72  if(xTimers[i] != NULL && (uint32_t)pvTimerGetTimerID(xTimers[i]) == virtualButton)
73  {
74  //delete & return
75  xTimerDelete(xTimers[i],0);

85 void debouncerCallback(TimerHandle_t xTimer) {
86  //get own ID (virtual button)
87  uint8_t virtualButton = (uint32_t) pvTimerGetTimerID(xTimer);
88  int8_t timerindex = isDebouncerActive(virtualButton);
89 
90  if(timerindex == -1)
91  {
92  ESP_LOGE(LOG_TAG,"Major error: debounceCallback but no timer for this button %d",virtualButton);
93  return;
94  }
95 
96  switch(xTimerDirection[timerindex])
97  {
98  case TIMER_IDLE:
99  ESP_LOGE(LOG_TAG,"Major error: debounceCallback but timer should be idle %d",virtualButton);
100  break;
101  case TIMER_PRESS:
102  ESP_LOGD(LOG_TAG,"Debounce finished, map in to out for press VB%d",virtualButton);
103  xEventGroupSetBits(virtualButtonsOut[virtualButton/4],(1<<(virtualButton%4)));
104  xEventGroupClearBits(virtualButtonsIn[virtualButton/4],(1<<(virtualButton%4)));
105  if(cancelTimer(virtualButton) == -1) ESP_LOGE(LOG_TAG,"Cannot cancel timer!");
106  break;
107  case TIMER_RELEASE:
108  ESP_LOGD(LOG_TAG,"Debounce finished, map in to out for release VB%d",virtualButton);
109  xEventGroupSetBits(virtualButtonsOut[virtualButton/4],(1<<((virtualButton%4)+4)));
110  xEventGroupClearBits(virtualButtonsIn[virtualButton/4],(1<<((virtualButton%4)+4)));
111  if(cancelTimer(virtualButton) == -1) ESP_LOGE(LOG_TAG,"Cannot cancel timer!");
112  break;
113  case TIMER_DEADTIME:
114  //TODO: add deadtime
115  ESP_LOGE(LOG_TAG,"TODO: add deadtime to debouncerCallback");
116  break;
117  }
118 }

121 int8_t startTimer(uint32_t virtualButton, uint16_t debounceTime)
122 {
123  //check all debouncerchannels for a free slot.
124  for(uint8_t i = 0; i<DEBOUNCERCHANNELS; i++)
125  {
126  //if slot is found
127  if(xTimers[i] == NULL)
128  {
129  //create timer
130  xTimers[i] = xTimerCreate("debouncer",debounceTime / portTICK_PERIOD_MS, \
131  pdFALSE, (void *) virtualButton, debouncerCallback);
132  if(xTimerStart(xTimers[i], 0) != pdPASS)
133  {
134  ESP_LOGE(LOG_TAG,"Cannot start debouncing timer!");
135  }
136  return i;
137  }
138  }
139  //no slot is found...
140  return -1;
141 }

144 void task_debouncer(void *param)
145 {
146  for(uint8_t i =0; i<NUMBER_VIRTUALBUTTONS;i++)
147  {
148  //test if eventgroup is created
149  while(virtualButtonsIn[i] == 0 || virtualButtonsOut[i] == 0)
150  {
151  ESP_LOGE(LOG_TAG,"Eventgroup uninitialized, retry in 1s");
152  vTaskDelay(1000/portTICK_PERIOD_MS);
153  }
154  //set timer direction to idle on startup
155  xTimerDirection[i] = TIMER_IDLE;
156 
157  //set default debounce time for press, release and deadtime
158  for(uint8_t j = 0; j<4;j++)
159  {
160  xTimeDebounceDeadtime[VB_ITER(i,j)] = 0;
161  xTimeDebouncePress[VB_ITER(i,j)] = DEBOUNCETIME_MS;
162  xTimeDebounceRelease[VB_ITER(i,j)] = DEBOUNCETIME_MS;
163  }
164  }
165 
166  ESP_LOGD(LOG_TAG,"Debouncer started, resolution: %d ms",DEBOUNCE_RESOLUTION_MS);
167 
168  while(1)
169  {
170  //check each eventgroup and start timer accordingly
171  for(uint8_t i = 0; i<NUMBER_VIRTUALBUTTONS; i++)
172  {
173  //check each virtual button, press (0-3) & release (4-7) flag
174  //iterate over all 4 VBs each group
175  for(uint8_t j = 0; j<4;j++)
176  {
177  //TODO TODO: avoid forwarding short edges (short release should not trigger press again).
178 
179 
180  //get in&out event bits and mask for this virtualbutton
181  EventBits_t in = xEventGroupGetBits(virtualButtonsIn[i]) & VB_FLAG_BOTH(j);
182  EventBits_t out = xEventGroupGetBits(virtualButtonsOut[i]) & VB_FLAG_BOTH(j);
183  int8_t timerId = isDebouncerActive(VB_ITER(i,j));
184 
185  //no in flag set, nothing to do.
186  if(in == 0) continue;
187 
188  //if both flags are set, we cannot know which one was first
189  //so clear both flags & don't start/stop any timer
190  if(in == VB_FLAG_BOTH(j))
191  {
192  ESP_LOGW(LOG_TAG,"press&release set for VB %d, discarding",(i*4)+j);
193  xEventGroupClearBits(virtualButtonsIn[i],VB_FLAG_BOTH(j));
194  continue;
195  }
196 
197  //if timer is not running, start one with the corresponding
198  //edge and set xTimerDirection.
199  if(timerId == -1)
200  {
201  //check if this is a press & out flag is unset
202  if(in == VB_FLAG_PRESS(j) && (out & VB_FLAG_PRESS(j)) == 0)
203  {
204  if(xTimeDebouncePress[VB_ITER(i,j)] > DEBOUNCE_RESOLUTION_MS)
205  {
206  timerId = startTimer(VB_ITER(i,j),xTimeDebouncePress[VB_ITER(i,j)]);
207  if(timerId == -1)
208  {
209  ESP_LOGE(LOG_TAG,"Cannot start timer...");
210  } else {
211  ESP_LOGD(LOG_TAG,"Press debounce started for VB%d",VB_ITER(i,j));
212  xTimerDirection[timerId] = TIMER_PRESS;
213  }
214  continue;
215  } else {
216  //if no debounce time is used
217  ESP_LOGD(LOG_TAG,"Min. debounce time, just mapping press");
218  //set flag in output eventgroup
219  xEventGroupSetBits(virtualButtonsOut[i],VB_FLAG_PRESS(j));
220  //delete flag in input eventgroup
221  xEventGroupClearBits(virtualButtonsIn[i],VB_FLAG_PRESS(j));
222  continue;
223  }
224  }
225  //check if this is a release & out flag is unset
226  if(in == VB_FLAG_RELEASE(j) && (out & VB_FLAG_RELEASE(j)) == 0)
227  {
228  if(xTimeDebounceRelease[VB_ITER(i,j)] > DEBOUNCE_RESOLUTION_MS)
229  {
230  timerId = startTimer(VB_ITER(i,j),xTimeDebounceRelease[VB_ITER(i,j)]);
231  if(timerId == -1)
232  {
233  ESP_LOGE(LOG_TAG,"Cannot start timer...");
234  } else {
235  ESP_LOGD(LOG_TAG,"Release debounce started for VB%d",VB_ITER(i,j));
236  xTimerDirection[timerId] = TIMER_RELEASE;
237  }
238  continue;
239  } else {
240  //if no debounce time is used
241  ESP_LOGD(LOG_TAG,"Min. debounce time, just mapping release");
242  //set flag in output eventgroup
243  xEventGroupSetBits(virtualButtonsOut[i],VB_FLAG_RELEASE(j));
244  //delete flag in input eventgroup
245  xEventGroupClearBits(virtualButtonsIn[i],VB_FLAG_RELEASE(j));
246  continue;
247  }
248  }
249  } else {
250  //if timer is running, check if this flag requests the
251  //opposite direction OR the same direction is cleared
252  //if yes -> stop & delete this timer
253  switch(xTimerDirection[timerId])
254  {
255  case TIMER_PRESS:
256  //press is cleared OR release is set
257  if( ((in & VB_FLAG_PRESS(j)) == 0) || ((in & VB_FLAG_RELEASE(j)) != 0) )
258  {
259  ESP_LOGD(LOG_TAG,"Press canceled for VB%d",VB_ITER(i,j));
260  if(cancelTimer(VB_ITER(i,j)) == -1) //stop current press debouncer
261  { ESP_LOGE(LOG_TAG,"Cannot cancel press timer!"); }
262  }
263  break;
264  case TIMER_RELEASE:
265  //press is set OR release is cleared
266  if( ((in & VB_FLAG_PRESS(j)) != 0) || ((in & VB_FLAG_RELEASE(j)) == 0) )
267  {
268  ESP_LOGD(LOG_TAG,"Release canceled for VB%d",VB_ITER(i,j));
269  if(cancelTimer(VB_ITER(i,j)) == -1) //stop current release debouncer
270  { ESP_LOGE(LOG_TAG,"Cannot cancel release timer!"); }
271  }
272  break;
273  case TIMER_IDLE:
274  ESP_LOGE(LOG_TAG,"Timer is idle but a valid ID?");
275  break;
276  case TIMER_DEADTIME:
277  //TODO: add deadtime
278  ESP_LOGE(LOG_TAG,"TODO: add deadtime capability");
279  break;
280  default:
281  ESP_LOGE(LOG_TAG,"Unknown status in xTimerDirection[%d]",timerId);
282  break;
283  }
284  }
285  }
286  }
287 
288  //wait for defined resolution
289  vTaskDelay(DEBOUNCE_RESOLUTION_MS / portTICK_PERIOD_MS);
290  }
291 }
