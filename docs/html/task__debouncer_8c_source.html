181 void debouncerCallback(TimerHandle_t xTimer) {
182  //get own ID (virtual button)
183  uint8_t virtualButton = (uint32_t) pvTimerGetTimerID(xTimer);
184  int8_t timerindex = isDebouncerActive(virtualButton);
185 
186  if(timerindex == -1)
187  {
188  ESP_LOGE(LOG_TAG,"Major error: debounceCallback but no timer for this button %d",virtualButton);
189  return;
190  }
191 
192  switch(xTimerDirection[timerindex])
193  {
194  case TIMER_IDLE:
195  ESP_LOGE(LOG_TAG,"Major error: debounceCallback but timer should be idle %d",virtualButton);
196  break;
197  case TIMER_PRESS:
198  ESP_LOGD(LOG_TAG,"Debounce finished, map in to out for press VB%d",virtualButton);
199  xEventGroupSetBits(virtualButtonsOut[virtualButton/4],(1<<(virtualButton%4)));
200  xEventGroupClearBits(virtualButtonsIn[virtualButton/4],(1<<(virtualButton%4)));
201  if(cancelTimer(virtualButton) == -1) ESP_LOGE(LOG_TAG,"Cannot cancel timer!");
202  break;
203  case TIMER_RELEASE:
204  ESP_LOGD(LOG_TAG,"Debounce finished, map in to out for release VB%d",virtualButton);
205  xEventGroupSetBits(virtualButtonsOut[virtualButton/4],(1<<((virtualButton%4)+4)));
206  xEventGroupClearBits(virtualButtonsIn[virtualButton/4],(1<<((virtualButton%4)+4)));
207  if(cancelTimer(virtualButton) == -1) ESP_LOGE(LOG_TAG,"Cannot cancel timer!");
208  break;
209  case TIMER_DEADTIME:
210  //TODO: add deadtime
211  ESP_LOGE(LOG_TAG,"TODO: add deadtime to debouncerCallback");
212  break;
213  }
214 }

228 int8_t startTimer(uint32_t virtualButton, uint16_t debounceTime)
229 {
230  //check all debouncerchannels for a free slot.
231  for(uint8_t i = 0; i<DEBOUNCERCHANNELS; i++)
232  {
233  //if slot is found
234  if(xTimers[i] == NULL)
235  {
236  //create timer
237  xTimers[i] = xTimerCreate("debouncer",debounceTime / portTICK_PERIOD_MS, \
238  pdFALSE, (void *) virtualButton, debouncerCallback);
239  if(xTimerStart(xTimers[i], 0) != pdPASS)
240  {
241  ESP_LOGE(LOG_TAG,"Cannot start debouncing timer!");
242  }
243  //return array index for further use
244  return i;
245  }
246  }
247  //no slot is found...
248  return -1;
249 }

272 void task_debouncer(void *param)
273 {
274  for(uint8_t i =0; i<NUMBER_VIRTUALBUTTONS;i++)
275  {
276  //test if eventgroup is created
277  while(virtualButtonsIn[i] == 0 || virtualButtonsOut[i] == 0)
278  {
279  ESP_LOGE(LOG_TAG,"Eventgroup uninitialized, retry in 1s");
280  vTaskDelay(1000/portTICK_PERIOD_MS);
281  }
282  //set timer direction to idle on startup
283  xTimerDirection[i] = TIMER_IDLE;
284 
285  //set default debounce time for press, release and deadtime
286  for(uint8_t j = 0; j<4;j++)
287  {
288  xTimeDebounceDeadtime[VB_ITER(i,j)] = 0;
289  xTimeDebouncePress[VB_ITER(i,j)] = DEBOUNCETIME_MS;
290  xTimeDebounceRelease[VB_ITER(i,j)] = DEBOUNCETIME_MS;
291  }
292  }
293 
294  ESP_LOGD(LOG_TAG,"Debouncer started, resolution: %d ms",DEBOUNCE_RESOLUTION_MS);
295 
296  while(1)
297  {
298  //check each eventgroup and start timer accordingly
299  for(uint8_t i = 0; i<NUMBER_VIRTUALBUTTONS; i++)
300  {
301  //check each virtual button, press (0-3) & release (4-7) flag
302  //iterate over all 4 VBs each group
303  for(uint8_t j = 0; j<4;j++)
304  {
305  //TODO TODO: avoid forwarding short edges (short release should not trigger press again).
306 
307 
308  //get in&out event bits and mask for this virtualbutton
309  EventBits_t in = xEventGroupGetBits(virtualButtonsIn[i]) & VB_FLAG_BOTH(j);
310  EventBits_t out = xEventGroupGetBits(virtualButtonsOut[i]) & VB_FLAG_BOTH(j);
311  int8_t timerId = isDebouncerActive(VB_ITER(i,j));
312 
313  //no in flag set, nothing to do.
314  if(in == 0) continue;
315 
316  //if both flags are set, we cannot know which one was first
317  //so clear both flags & don't start/stop any timer
318  if(in == VB_FLAG_BOTH(j))
319  {
320  ESP_LOGW(LOG_TAG,"press&release set for VB %d, discarding",(i*4)+j);
321  xEventGroupClearBits(virtualButtonsIn[i],VB_FLAG_BOTH(j));
322  continue;
323  }
324 
325  //if timer is not running, start one with the corresponding
326  //edge and set xTimerDirection.
327  if(timerId == -1)
328  {
329  //check if this is a press & out flag is unset
330  if(in == VB_FLAG_PRESS(j) && (out & VB_FLAG_PRESS(j)) == 0)
331  {
332  if(xTimeDebouncePress[VB_ITER(i,j)] > DEBOUNCE_RESOLUTION_MS)
333  {
334  timerId = startTimer(VB_ITER(i,j),xTimeDebouncePress[VB_ITER(i,j)]);
335  if(timerId == -1)
336  {
337  ESP_LOGE(LOG_TAG,"Cannot start timer...");
338  } else {
339  ESP_LOGD(LOG_TAG,"Press debounce started for VB%d",VB_ITER(i,j));
340  xTimerDirection[timerId] = TIMER_PRESS;
341  }
342  continue;
343  } else {
344  //if no debounce time is used
345  ESP_LOGD(LOG_TAG,"Min. debounce time, just mapping press");
346  //set flag in output eventgroup
347  xEventGroupSetBits(virtualButtonsOut[i],VB_FLAG_PRESS(j));
348  //delete flag in input eventgroup
349  xEventGroupClearBits(virtualButtonsIn[i],VB_FLAG_PRESS(j));
350  continue;
351  }
352  }
353  //check if this is a release & out flag is unset
354  if(in == VB_FLAG_RELEASE(j) && (out & VB_FLAG_RELEASE(j)) == 0)
355  {
356  if(xTimeDebounceRelease[VB_ITER(i,j)] > DEBOUNCE_RESOLUTION_MS)
357  {
358  timerId = startTimer(VB_ITER(i,j),xTimeDebounceRelease[VB_ITER(i,j)]);
359  if(timerId == -1)
360  {
361  ESP_LOGE(LOG_TAG,"Cannot start timer...");
362  } else {
363  ESP_LOGD(LOG_TAG,"Release debounce started for VB%d",VB_ITER(i,j));
364  xTimerDirection[timerId] = TIMER_RELEASE;
365  }
366  continue;
367  } else {
368  //if no debounce time is used
369  ESP_LOGD(LOG_TAG,"Min. debounce time, just mapping release");
370  //set flag in output eventgroup
371  xEventGroupSetBits(virtualButtonsOut[i],VB_FLAG_RELEASE(j));
372  //delete flag in input eventgroup
373  xEventGroupClearBits(virtualButtonsIn[i],VB_FLAG_RELEASE(j));
374  continue;
375  }
376  }
377  } else {
378  //if timer is running, check if this flag requests the
379  //opposite direction OR the same direction is cleared
380  //if yes -> stop & delete this timer
381  switch(xTimerDirection[timerId])
382  {
383  case TIMER_PRESS:
384  //press is cleared OR release is set
385  if( ((in & VB_FLAG_PRESS(j)) == 0) || ((in & VB_FLAG_RELEASE(j)) != 0) )
386  {
387  ESP_LOGD(LOG_TAG,"Press canceled for VB%d",VB_ITER(i,j));
388  if(cancelTimer(VB_ITER(i,j)) == -1) //stop current press debouncer
389  { ESP_LOGE(LOG_TAG,"Cannot cancel press timer!"); }
390  }
391  break;
392  case TIMER_RELEASE:
393  //press is set OR release is cleared
394  if( ((in & VB_FLAG_PRESS(j)) != 0) || ((in & VB_FLAG_RELEASE(j)) == 0) )
395  {
396  ESP_LOGD(LOG_TAG,"Release canceled for VB%d",VB_ITER(i,j));
397  if(cancelTimer(VB_ITER(i,j)) == -1) //stop current release debouncer
398  { ESP_LOGE(LOG_TAG,"Cannot cancel release timer!"); }
399  }
400  break;
401  case TIMER_IDLE:
402  ESP_LOGE(LOG_TAG,"Timer is idle but a valid ID?");
403  break;
404  case TIMER_DEADTIME:
405  //TODO: add deadtime
406  ESP_LOGE(LOG_TAG,"TODO: add deadtime capability");
407  break;
408  default:
409  ESP_LOGE(LOG_TAG,"Unknown status in xTimerDirection[%d]",timerId);
410  break;
411  }
412  }
413  }
414  }
415 
416  //wait for defined resolution
417  vTaskDelay(DEBOUNCE_RESOLUTION_MS / portTICK_PERIOD_MS);
418  }
