3  *
4  * This class encapsulates advertising a BLE Server.
5  * Created on: Jun 21, 2017
6  * Author: kolban
7  *
8  * The ESP-IDF provides a framework for BLE advertising. It has determined that there are a common set
9  * of properties that are advertised and has built a data structure that can be populated by the programmer.
10  * This means that the programmer doesn't have to "mess with" the low level construction of a low level
11  * BLE advertising frame. Many of the fields are determined for us while others we can set before starting
12  * to advertise.
13  *
14  * Should we wish to construct our own payload, we can use the BLEAdvertisementData class and call the setters
15  * upon it. Once it is populated, we can then associate it with the advertising and what ever the programmer
16  * set in the data will be advertised.
17  *
18  */

173 void BLEAdvertising::start() {
174  ESP_LOGD(LOG_TAG, ">> start: customAdvData: %d, customScanResponseData: %d", m_customAdvData, m_customScanResponseData);
175 
176 
177  // We have a vector of service UUIDs that we wish to advertise. In order to use the
178  // ESP-IDF framework, these must be supplied in a contiguous array of their 128bit (16 byte)
179  // representations. If we have 1 or more services to advertise then we allocate enough
180  // storage to host them and then copy them in one at a time into the contiguous storage.
181  int numServices = m_serviceUUIDs.size();
182  if (numServices > 0) {
183  m_advData.service_uuid_len = 16*numServices;
184  m_advData.p_service_uuid = new uint8_t[m_advData.service_uuid_len];
185  uint8_t* p = m_advData.p_service_uuid;
186  for (int i=0; i<numServices; i++) {
187  ESP_LOGD(LOG_TAG, "- advertising service: %s", m_serviceUUIDs[i].toString().c_str());
188  BLEUUID serviceUUID128 = m_serviceUUIDs[i].to128();
189  memcpy(p, serviceUUID128.getNative()->uuid.uuid128, 16);
190  p+=16;
191  }
192  } else {
193  m_advData.service_uuid_len = 0;
194  ESP_LOGD(LOG_TAG, "- no services advertised");
195  }
196 
197  esp_err_t errRc;
198 
199  if (m_customAdvData == false) {
200  // Set the configuration for advertising.
201  m_advData.set_scan_rsp = false;
202  errRc = ::esp_ble_gap_config_adv_data(&m_advData);
203  if (errRc != ESP_OK) {
204  ESP_LOGE(LOG_TAG, "<< esp_ble_gap_config_adv_data: rc=%d %s", errRc, GeneralUtils::errorToString(errRc));
205  return;
206  }
207  }
208 
209  if (m_customScanResponseData == false) {
210  m_advData.set_scan_rsp = true;
211  errRc = ::esp_ble_gap_config_adv_data(&m_advData);
212  if (errRc != ESP_OK) {
213  ESP_LOGE(LOG_TAG, "<< esp_ble_gap_config_adv_data (Scan response): rc=%d %s", errRc, GeneralUtils::errorToString(errRc));
214  return;
215  }
216  }
217 
218  // If we had services to advertise then we previously allocated some storage for them.
219  // Here we release that storage.
220  if (m_advData.service_uuid_len > 0) {
221  delete[] m_advData.p_service_uuid;
222  m_advData.p_service_uuid = nullptr;
223  }
224 
225  // Start advertising.
226  errRc = ::esp_ble_gap_start_advertising(&m_advParams);
227  if (errRc != ESP_OK) {
228  ESP_LOGE(LOG_TAG, "<< esp_ble_gap_start_advertising: rc=%d %s", errRc, GeneralUtils::errorToString(errRc));
229  return;
230  }
231  ESP_LOGD(LOG_TAG, "<< start");
232 } // start
