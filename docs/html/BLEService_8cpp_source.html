121 void BLEService::start() {
122 // We ask the BLE runtime to start the service and then create each of the characteristics.
123 // We start the service through its local handle which was returned in the ESP_GATTS_CREATE_EVT event
124 // obtained as a result of calling esp_ble_gatts_create_service().
125 //
126  ESP_LOGD(LOG_TAG, ">> start(): Starting service (esp_ble_gatts_start_service): %s", toString().c_str());
127  if (m_handle == NULL_HANDLE) {
128  ESP_LOGE(LOG_TAG, "<< !!! We attempted to start a service but don't know its handle!");
129  return;
130  }
131 
132 
133  BLECharacteristic *pCharacteristic = m_characteristicMap.getFirst();
134 
135  while(pCharacteristic != nullptr) {
136  m_lastCreatedCharacteristic = pCharacteristic;
137  pCharacteristic->executeCreate(this);
138 
139  pCharacteristic = m_characteristicMap.getNext();
140  }
141  // Start each of the characteristics ... these are found in the m_characteristicMap.
142 
143  m_semaphoreStartEvt.take("start");
144  esp_err_t errRc = ::esp_ble_gatts_start_service(m_handle);
145 
146  if (errRc != ESP_OK) {
147  ESP_LOGE(LOG_TAG, "<< esp_ble_gatts_start_service: rc=%d %s", errRc, GeneralUtils::errorToString(errRc));
148  return;
149  }
150  m_semaphoreStartEvt.wait("start");
151 
152  ESP_LOGD(LOG_TAG, "<< start()");
153 } // start

184 void BLEService::addCharacteristic(BLECharacteristic* pCharacteristic) {
185 // We maintain a mapping of characteristics owned by this service. These are managed by the
186 // BLECharacteristicMap class instance found in m_characteristicMap. We add the characteristic
187 // to the map and then ask the service to add the characteristic at the BLE level (ESP-IDF).
188 //
189  ESP_LOGD(LOG_TAG, ">> addCharacteristic()");
190  ESP_LOGD(LOG_TAG, "Adding characteristic: uuid=%s to service: %s",
191  pCharacteristic->getUUID().toString().c_str(),
192  toString().c_str());
193 
194  // Check that we don't add the same characteristic twice.
195  if (m_characteristicMap.getByUUID(pCharacteristic->getUUID()) != nullptr) {
196  ESP_LOGW(LOG_TAG, "<< Adding a new characteristic with the same UUID as a previous one");
197  //return;
198  }
199 
200  // Remember this characteristic in our map of characteristics. At this point, we can lookup by UUID
201  // but not by handle. The handle is allocated to us on the ESP_GATTS_ADD_CHAR_EVT.
202  m_characteristicMap.setByUUID(pCharacteristic, pCharacteristic->getUUID());
203 
204  ESP_LOGD(LOG_TAG, "<< addCharacteristic()");
205 } // addCharacteristic

235 void BLEService::handleGATTServerEvent(
236  esp_gatts_cb_event_t event,
237  esp_gatt_if_t gatts_if,
238  esp_ble_gatts_cb_param_t *param) {
239 
240 
241  switch(event) {
242  // ESP_GATTS_ADD_CHAR_EVT - Indicate that a characteristic was added to the service.
243  // add_char:
244  // - esp_gatt_status_t status
245  // - uint16_t attr_handle
246  // - uint16_t service_handle
247  // - esp_bt_uuid_t char_uuid
248 
249  // If we have reached the correct service, then locate the characteristic and remember the handle
250  // for that characteristic.
251  case ESP_GATTS_ADD_CHAR_EVT: {
252  if (m_handle == param->add_char.service_handle) {
253  BLECharacteristic *pCharacteristic = getLastCreatedCharacteristic();
254  if (pCharacteristic == nullptr) {
255  ESP_LOGE(LOG_TAG, "Expected to find characteristic with UUID: %s, but didnt!",
256  BLEUUID(param->add_char.char_uuid).toString().c_str());
257  dump();
258  break;
259  }
260  pCharacteristic->setHandle(param->add_char.attr_handle);
261  m_characteristicMap.setByHandle(param->add_char.attr_handle, pCharacteristic);
262  //ESP_LOGD(tag, "Characteristic map: %s", m_characteristicMap.toString().c_str());
263  break;
264  } // Reached the correct service.
265  break;
266  } // ESP_GATTS_ADD_CHAR_EVT
267 
268 
269  // ESP_GATTS_START_EVT
270  //
271  // start:
272  // esp_gatt_status_t status
273  // uint16_t service_handle
274  case ESP_GATTS_START_EVT: {
275  if (param->start.service_handle == getHandle()) {
276  m_semaphoreStartEvt.give();
277  }
278  break;
279  } // ESP_GATTS_START_EVT
280 
281 
282  // ESP_GATTS_CREATE_EVT
283  // Called when a new service is registered as having been created.
284  //
285  // create:
286  // * esp_gatt_status_t status
287  // * uint16_t service_handle
288  // * esp_gatt_srvc_id_t service_id
289  // * - esp_gatt_id id
290  // * - esp_bt_uuid uuid
291  // * - uint8_t inst_id
292  // * - bool is_primary
293  //
294  case ESP_GATTS_CREATE_EVT: {
295  if (getUUID().equals(BLEUUID(param->create.service_id.id.uuid))) {
296  setHandle(param->create.service_handle);
297  m_semaphoreCreateEvt.give();
298  }
299  break;
300  } // ESP_GATTS_CREATE_EVT
301 
302  default: {
303  break;
304  } // Default
305  } // Switch
306 
307  // Invoke the GATTS handler in each of the associated characteristics.
308  m_characteristicMap.handleGATTServerEvent(event, gatts_if, param);
309 } // handleGATTServerEvent
