133 void BLEDescriptor::handleGATTServerEvent(
134  esp_gatts_cb_event_t event,
135  esp_gatt_if_t gatts_if,
136  esp_ble_gatts_cb_param_t *param) {
137  switch(event) {
138  // ESP_GATTS_ADD_CHAR_DESCR_EVT
139  //
140  // add_char_descr:
141  // - esp_gatt_status_t status
142  // - uint16_t attr_handle
143  // - uint16_t service_handle
144  // - esp_bt_uuid_t char_uuid
145  case ESP_GATTS_ADD_CHAR_DESCR_EVT: {
146  /*
147  ESP_LOGD(LOG_TAG, "DEBUG: m_pCharacteristic: %x", (uint32_t)m_pCharacteristic);
148  ESP_LOGD(LOG_TAG, "DEBUG: m_bleUUID: %s, add_char_descr.char_uuid: %s, equals: %d",
149  m_bleUUID.toString().c_str(),
150  BLEUUID(param->add_char_descr.char_uuid).toString().c_str(),
151  m_bleUUID.equals(BLEUUID(param->add_char_descr.char_uuid)));
152  ESP_LOGD(LOG_TAG, "DEBUG: service->getHandle: %x, add_char_descr.service_handle: %x",
153  m_pCharacteristic->getService()->getHandle(), param->add_char_descr.service_handle);
154  ESP_LOGD(LOG_TAG, "DEBUG: service->lastCharacteristic: %x",
155  (uint32_t)m_pCharacteristic->getService()->getLastCreatedCharacteristic());
156  */
157  if (m_pCharacteristic != nullptr &&
158  m_bleUUID.equals(BLEUUID(param->add_char_descr.descr_uuid)) &&
159  m_pCharacteristic->getService()->getHandle() == param->add_char_descr.service_handle &&
160  m_pCharacteristic == m_pCharacteristic->getService()->getLastCreatedCharacteristic()) {
161  setHandle(param->add_char_descr.attr_handle);
162  m_semaphoreCreateEvt.give();
163  }
164  break;
165  } // ESP_GATTS_ADD_CHAR_DESCR_EVT
166 
167  // ESP_GATTS_WRITE_EVT - A request to write the value of a descriptor has arrived.
168  //
169  // write:
170  // - uint16_t conn_id
171  // - uint16_t trans_id
172  // - esp_bd_addr_t bda
173  // - uint16_t handle
174  // - uint16_t offset
175  // - bool need_rsp
176  // - bool is_prep
177  // - uint16_t len
178  // - uint8_t *value
179  case ESP_GATTS_WRITE_EVT: {
180  if (param->write.handle == m_handle) {
181  setValue(param->write.value, param->write.len); // Set the value of the descriptor.
182 
183  esp_gatt_rsp_t rsp; // Build a response.
184  rsp.attr_value.len = getLength();
185  rsp.attr_value.handle = m_handle;
186  rsp.attr_value.offset = 0;
187  rsp.attr_value.auth_req = ESP_GATT_AUTH_REQ_NONE;
188  memcpy(rsp.attr_value.value, getValue(), rsp.attr_value.len);
189  esp_err_t errRc = ::esp_ble_gatts_send_response(
190  gatts_if,
191  param->write.conn_id,
192  param->write.trans_id,
193  ESP_GATT_OK,
194  &rsp);
195 
196  if (errRc != ESP_OK) { // Check the return code from the send of the response.
197  ESP_LOGE(LOG_TAG, "esp_ble_gatts_send_response: rc=%d %s", errRc, GeneralUtils::errorToString(errRc));
198  }
199 
200  if (m_pCallback != nullptr) { // We have completed the write, if there is a user supplied callback handler, invoke it now.
201  m_pCallback->onWrite(this); // Invoke the onWrite callback handler.
202  }
203  } // End of ... this is our handle.
204 
205  break;
206  } // ESP_GATTS_WRITE_EVT
207 
208  // ESP_GATTS_READ_EVT - A request to read the value of a descriptor has arrived.
209  //
210  // read:
211  // - uint16_t conn_id
212  // - uint32_t trans_id
213  // - esp_bd_addr_t bda
214  // - uint16_t handle
215  // - uint16_t offset
216  // - bool is_long
217  // - bool need_rsp
218  //
219  case ESP_GATTS_READ_EVT: {
220  if (param->read.handle == m_handle) { // If this event is for this descriptor ... process it
221 
222  if (m_pCallback != nullptr) { // If we have a user supplied callback, invoke it now.
223  m_pCallback->onRead(this); // Invoke the onRead callback method in the callback handler.
224  }
225 
226  if (param->read.need_rsp) { // Do we need a response
227  ESP_LOGD(LOG_TAG, "Sending a response (esp_ble_gatts_send_response)");
228  esp_gatt_rsp_t rsp;
229  rsp.attr_value.len = getLength();
230  rsp.attr_value.handle = param->read.handle;
231  rsp.attr_value.offset = 0;
232  rsp.attr_value.auth_req = ESP_GATT_AUTH_REQ_NONE;
233  memcpy(rsp.attr_value.value, getValue(), rsp.attr_value.len);
234 
235  esp_err_t errRc = ::esp_ble_gatts_send_response(
236  gatts_if,
237  param->read.conn_id,
238  param->read.trans_id,
239  ESP_GATT_OK,
240  &rsp);
241 
242  if (errRc != ESP_OK) { // Check the return code from the send of the response.
243  ESP_LOGE(LOG_TAG, "esp_ble_gatts_send_response: rc=%d %s", errRc, GeneralUtils::errorToString(errRc));
244  }
245  } // End of need a response.
246  } // End of this is our handle
247  break;
248  } // ESP_GATTS_READ_EVT
249 
250  default: {
251  break;
252  }
253  }// switch event
254 } // handleGATTServerEvent
