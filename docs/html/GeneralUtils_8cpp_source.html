39 static void a3_to_a4(unsigned char * a4, unsigned char * a3) {
40  a4[0] = (a3[0] & 0xfc) >> 2;
41  a4[1] = ((a3[0] & 0x03) << 4) + ((a3[1] & 0xf0) >> 4);
42  a4[2] = ((a3[1] & 0x0f) << 2) + ((a3[2] & 0xc0) >> 6);
43  a4[3] = (a3[2] & 0x3f);
44 } // a3_to_a4

59 bool GeneralUtils::base64Encode(const std::string &in, std::string *out) {
60  int i = 0, j = 0;
61  size_t enc_len = 0;
62  unsigned char a3[3];
63  unsigned char a4[4];
64 
65  out->resize(base64EncodedLength(in));
66 
67  int input_len = in.size();
68  std::string::const_iterator input = in.begin();
69 
70  while (input_len--) {
71  a3[i++] = *(input++);
72  if (i == 3) {
73  a3_to_a4(a4, a3);
74 
75  for (i = 0; i < 4; i++) {
76  (*out)[enc_len++] = kBase64Alphabet[a4[i]];
77  }
78 
79  i = 0;
80  }
81  }
82 
83  if (i) {
84  for (j = i; j < 3; j++) {
85  a3[j] = '\0';

97  }

149  if(c >='A' && c <='Z') return c - 'A';
150  if(c >='a' && c <='z') return c - 71;
151  if(c >='0' && c <='9') return c + 4;
