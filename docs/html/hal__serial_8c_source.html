146 void halSerialRXTask(void *pvParameters)
147 {
148  uint16_t cmdoffset;
149  uint8_t *buf;
150  uint8_t data;
151  uint8_t parserstate = 0;
152  atcmd_t currentcmd;
153 
154  while(1)
155  {
156  int len = uart_read_bytes(HAL_SERIAL_UART, &data, 1, portMAX_DELAY);
157 
158  //only parse if a valid byte is received
159  if(len == 1)
160  {
161  switch(parserstate)
162  {
163  case 0:
164  //check for leading "A"
165  if(data == 'A' || data == 'a')
166  {
168  parserstate++;
169  }
170  //if starting with '@', go to CIM mode.
171  if(data == '@')
172  {
174  ESP_LOGW(LOG_TAG,"CIM mdoe currently unsupported!");
175  }
176  break;
177 
178 
179  case 1:
180  //check for second 'T'/'t'
181  if(data == 'T' || data == 't')
182  {
183  parserstate++;
184  //reset command offset to beginning (but after "AT")
185  cmdoffset = 2;
186  //allocate new buffer for command & save "AT"
187  //free is either done in error case or in halSerialReceiveUSBSerial.
188  buf = malloc(ATCMD_LENGTH*sizeof(uint8_t));
189  if(buf != NULL)
190  {
191  buf[0] = 'A';
192  buf[1] = 'T';
193  } else {
194  ESP_LOGE(LOG_TAG,"Cannot allocate buffer for command");
195  parserstate = 0;
196  }
197  } else {
198  //reset parser of no leading "AT" is detected
199  parserstate = 0;
200  }
201  break;
202 
203 
204  case 2:
205  //now read data until we reach \r or \n
206  if(data == '\r' || data == '\n')
207  {
208  //terminate string
209  buf[cmdoffset] = 0;
210  //send buffer to queue
211  currentcmd.buf = buf;
212  currentcmd.len = cmdoffset+1;
213  if(halSerialATCmds != NULL)
214  {
215  if(xQueueSend(halSerialATCmds,(void*)&currentcmd,10) != pdTRUE)
216  {
217  ESP_LOGE(LOG_TAG,"AT cmd queue is full, cannot send cmd");
218  free(buf);
219  } else {
220  ESP_LOGI(LOG_TAG,"Sent AT cmd with len %d to queue",cmdoffset);
221  }
222  } else {
223  ESP_LOGE(LOG_TAG,"AT cmd queue is NULL, cannot send cmd");
224  free(buf);
225  }
226  parserstate = 0;
227  break;
228  }
229 
230  //if everything is fine, just save this byte.
231  buf[cmdoffset] = data;
232  cmdoffset++;
233 
234  //check for memory length
235  if(cmdoffset == ATCMD_LENGTH)
236  {
237  ESP_LOGW(LOG_TAG,"AT cmd too long, discarding");
238  free(buf);
239  parserstate = 0;
240  }
241  break;
242 
243  default:
244  ESP_LOGE(LOG_TAG,"Unknown parser state");
245  parserstate = 0;
246  break;
247  }
248  }
249  }
250 
251  //we should never be here...
252  vTaskDelete(NULL);
253 }

443 
445 
446  /*++++ build report ++++*/
447 
448  //13 bytes:
449  //'J' 0
450  //buttonmask 1
451  //buttonmask 2
452  //buttonmask 3
453  //buttonmask 4
454 
455  //hat [b0-b3]. Xaxis [b4-b7] 5
456  //Xaxis [b0-b5], Yaxis[b6-b7] 6
457  //Yaxis [b0-b7] 7
458  //Zaxis [b0-b7] 8
459 
460  //Zaxis [b0-b1], zrotate [b2-b7] 9
461  //zrotate[b0-b3],sliderleft[b4-b7] 10
462  //sliderLeft[b0-b5],sliderRight[b6-b7] 11

482  else if (rxJ.hat < 338) report[5] = 7;

485  report[6] = (rxJ.Xaxis & 0x03F0) >> 4;
486  //add Yaxis

495  //add sliderLeft

497  report[11] = (rxJ.sliderLeft & 0x03F0) >> 4;
498  //add sliderRight

500  report[12] = (rxJ.sliderRight & 0x03FC) >> 2;
501 
502  //log buffer

504  esp_log_buffer_hex(LOG_TAG,report,13);
505 
506  //send it to serial port
507  uint8_t sent = 0;

578 int halSerialSendUSBSerial(uint8_t channel, char *data, uint32_t length, TickType_t ticks_to_wait)
579 {
580  //check if sem is already initialized
581  if(serialsendingsem == NULL)
582  {
583  ESP_LOGW(LOG_TAG,"Sem not ready, waiting for 1s");
584  vTaskDelay(1000/portTICK_PERIOD_MS);
585  }
586 
587  //acquire mutex to have TX permission on UART
588  if(xSemaphoreTake(serialsendingsem, ticks_to_wait) == pdTRUE)
589  {
590  //try to switch to USB-Serial
591  if(halSerialSwitchTXChannel(channel,ticks_to_wait)!=ESP_OK)
592  {
593  ESP_LOGE(LOG_TAG,"switch GPIO channel didn't finish");
594  xSemaphoreGive(serialsendingsem);
595  return -1;
596  }
597  //send the data
598  int txBytes = uart_write_bytes(HAL_SERIAL_UART, data, length);
599 
600  //release mutex
601  xSemaphoreGive(serialsendingsem);
602 
603  return txBytes;
604  } else return -1;
605 }
