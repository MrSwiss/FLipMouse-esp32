123 void halSerialRXTask(void *pvParameters)
124 {
125  uint16_t cmdoffset;
126  uint8_t *buf;
127  uint8_t *bufstatic = malloc(ATCMD_LENGTH);
128  uint8_t data;
129  uint8_t parserstate = 0;
130  atcmd_t currentcmd;
131 
132  if(bufstatic == NULL)
133  {
134  ESP_LOGE(LOG_TAG,"Cannot allocate mem");
135  vTaskDelete(NULL);
136  }
137 
138  while(1)
139  {
140  int len = uart_read_bytes(HAL_SERIAL_UART, &data, 1, portMAX_DELAY);
141 
142  //only parse if a valid byte is received
143  if(len == 1)
144  {
145  switch(parserstate)
146  {
147  case 0:
148  //check for leading "A"
149  if(data == 'A' || data == 'a')
150  {
152  parserstate++;
153  }
154  //if starting with '@', go to CIM mode.
155  if(data == '@')
156  {
158  ESP_LOGW(LOG_TAG,"CIM mode currently unsupported!");
159  }
160  break;
161 
162 
163  case 1:
164  //check for second 'T'/'t'
165  if(data == 'T' || data == 't')
166  {
167  parserstate++;
168  //reset command offset to beginning (but after "AT")
169  cmdoffset = 2;
170  bufstatic[0] = 'A';
171  bufstatic[1] = 'T';
172 
173  } else {
174  //reset parser of no leading "AT" is detected
175  parserstate = 0;
176  }
177  break;
178 
179 
180  case 2:
181  //now read data until we reach \r or \n
182  if(data == '\r' || data == '\n')
183  {
184  //terminate string
185  bufstatic[cmdoffset] = 0;
186  //allocate a new buffer for this command to be processed later
187  //free is either done in halSerialReceiveUSBSerial.
188  buf = malloc(cmdoffset+1);
189  if(buf == NULL)
190  {
191  ESP_LOGE(LOG_TAG,"Cannot allocate %d B buffer for new AT cmd",cmdoffset+1);
192  parserstate = 0;
193  break;
194  }
195  memcpy(buf,bufstatic,cmdoffset+1);
196 
197  //send buffer to queue
198  currentcmd.buf = buf;
199  currentcmd.len = cmdoffset+1;
200  if(halSerialATCmds != NULL)
201  {
202  if(xQueueSend(halSerialATCmds,(void*)&currentcmd,10) != pdTRUE)
203  {
204  ESP_LOGE(LOG_TAG,"AT cmd queue is full, cannot send cmd");
205  free(buf);
206  } else {
207  ESP_LOGI(LOG_TAG,"Sent AT cmd with len %d to queue: %s",cmdoffset,bufstatic);
208  }
209  } else {
210  ESP_LOGE(LOG_TAG,"AT cmd queue is NULL, cannot send cmd");
211  free(buf);
212  }
213  parserstate = 0;
214  break;
215  }
216 
217  //if everything is fine, just save this byte.
218  bufstatic[cmdoffset] = data;
219  cmdoffset++;
220 
221  //check for memory length
222  if(cmdoffset == ATCMD_LENGTH)
223  {
224  ESP_LOGW(LOG_TAG,"AT cmd too long, discarding");
225  parserstate = 0;
226  }
227  break;
228 
229  default:
230  ESP_LOGE(LOG_TAG,"Unknown parser state");
231  parserstate = 0;
232  break;
233  }
234  }
235  }
236 
237  //we should never be here...
238  vTaskDelete(NULL);
239 }

273 void halSerialHIDTask(void *param)
274 {
275  hid_command_t rx;
276  //RMT RAM has 64x32bit memory each block
277  rmt_item32_t *rmtBuf = malloc(sizeof(rmt_item32_t)*64);
278 
279  if(rmtBuf == NULL)
280  {
281  ESP_LOGE(LOG_TAG,"Cannot allocate memory for RMT");
282  vTaskDelete(NULL);
283  }
284 
285  uint8_t rmtCount = 0;
286 
287  //levels are always same, set here.
288  rmt_item32_t item;
289  item.level0 = 1;
290  item.level1 = 0;
291 
292  while(1)
293  {
294  //check if queue is initialized
295  if(hid_usb != NULL)
296  {
297  //pend on MQ, if timeout triggers, just wait again.
298  if(xQueueReceive(hid_usb,&rx,portMAX_DELAY))
299  {
300  //reset RMT offset
301  rmtCount = 0;
302 
303  //debug output
304  #if LOG_LEVEL_SERIAL >= ESP_LOG_DEBUG
305  switch(rx.data[0])
306  {
307  case 'M': ESP_LOGD(LOG_TAG,"USB Mouse: B: %d, X/Y: %d/%d, wheel: %d", \
308  rx.data[1],(int8_t)rx.data[2],(int8_t)rx.data[3],(int8_t)rx.data[4]);
309  break;
310  case 'K': ESP_LOGD(LOG_TAG,"USB Kbd: Mod: %d, keys: %d/%d/%d/%d/%d/%d", \
311  rx.data[1],rx.data[2],rx.data[3],rx.data[4],rx.data[5],rx.data[6],rx.data[7]);
312  break;
313  case 'J': ESP_LOGD(LOG_TAG,"USB joystick:");
314  ESP_LOG_BUFFER_HEXDUMP(LOG_TAG,&rx.data[1], 12 ,ESP_LOG_DEBUG);
315  break;
316  default: break;
317  }
318  #endif
319 
320  if(rx.data[0] != 'M' && rx.data[0] != 'K' && rx.data[0] != 'J') {
321  ESP_LOGE(LOG_TAG,"Unknown USB report");
322  }
323 
324  //build RMT buffer
325  for(uint8_t i = 0; i<16; i++)
326  {
327  //if all bytes are processed, quit this loop
328  if(i==rx.len) break;
329 
330  //process 2 bits at once (one rmt item)
331  for(uint8_t j = 0; j<4; j++)
332  {
333  //process even bit
334  if((rx.data[i] & (1<<(j*2))) != 0)
335  {
336  item.duration0 = HAL_SERIAL_HID_DURATION_1; //long timing
337  } else {
338  item.duration0 = HAL_SERIAL_HID_DURATION_0; //short timing
339  }
340  //process odd bit
341  if((rx.data[i] & (1<<(j*2+1))) != 0)
342  {
343  item.duration1 = HAL_SERIAL_HID_DURATION_1;
344  } else {
345  item.duration1 = HAL_SERIAL_HID_DURATION_0;
346  }
347  //fill buffer
348  //rmtCount = ((i*8+j*2)/2) + 1;
349  rmtBuf[rmtCount] = item;
350  rmtCount++;
351  }
352  }
353 
354  //signal the RMT the end of this transmission
355  item.duration0 = HAL_SERIAL_HID_DURATION_S; //stop bit timing
356  item.duration1 = 0;
357  rmtBuf[rmtCount] = item;
358  rmtCount++;
359 
360  #if LOG_LEVEL_SERIAL>=ESP_LOG_VERBOSE
361  ESP_LOGV(LOG_TAG,"RMT dump:");
362  ESP_LOG_BUFFER_HEXDUMP(LOG_TAG,rmtBuf, sizeof(rmt_item32_t)*(rmtCount),ESP_LOG_VERBOSE);
363  #endif
364 
365  //put data into RMT buffer
366  if(rmt_fill_tx_items(HAL_SERIAL_HID_CHANNEL, rmtBuf, \
367  rmtCount, 0) != ESP_OK)
368  {
369  ESP_LOGE(LOG_TAG,"Cannot send data to RMT");
370  }
371  //start TX, reset memory index (always send from beginning)
372  if(rmt_tx_start(HAL_SERIAL_HID_CHANNEL, 1) != ESP_OK)
373  {
374  ESP_LOGE(LOG_TAG,"Cannot start RMT TX");
375  }
376  rmt_wait_tx_done(HAL_SERIAL_HID_CHANNEL,portMAX_DELAY);
377  }
378  } else {
379  ESP_LOGW(LOG_TAG,"usb hid queue not initialized, retry in 1s");
380  vTaskDelay(1000/portTICK_PERIOD_MS);
381  }
382  }
383 }

437 int halSerialSendUSBSerial(char *data, uint32_t length, TickType_t ticks_to_wait)
438 {
439  //check if sem is already initialized
440  if(serialsendingsem == NULL)
441  {
442  ESP_LOGW(LOG_TAG,"Sem not ready, waiting for 1s");
443  vTaskDelay(1000/portTICK_PERIOD_MS);
444  }
445 
446  //if an additional stream is registered, send data there as well
447  //no HID commands will be sent there.
448  if(outputcb != NULL)
449  {
450  if(outputcb(data,length) != ESP_OK)
451  {
452  ESP_LOGE(LOG_TAG,"Additional stream cannot be sent");
453  }
454  }
455 
456  //acquire mutex to have TX permission on UART
457  if(xSemaphoreTake(serialsendingsem, ticks_to_wait) == pdTRUE)
458  {
459  //send the data
460  int txBytes = uart_write_bytes(HAL_SERIAL_UART, data, length);
461  uart_write_bytes(HAL_SERIAL_UART, HAL_SERIAL_LINE_ENDING, strnlen(HAL_SERIAL_LINE_ENDING,4));
462  //release mutex
463  xSemaphoreGive(serialsendingsem);
464 
465  return txBytes;
466  } else return -1;
467 }

482 void halSerialReset(uint8_t exceptDevice)
483 {
484  ESP_LOGD(LOG_TAG,"USB-HID reset reports, except mask: %d",exceptDevice);
485  //reset mouse
486  if(!(exceptDevice & (1<<2)))
487  {
488  hid_command_t m;
489  m.len = 5;
490  m.data[0] = 'M';
491  m.data[1] = 0; //buttons
492  m.data[2] = 0; //X
493  m.data[3] = 0; //Y
494  m.data[4] = 0; //wheel
495 
496  //send to queue
497  xQueueSend(hid_usb, &m, 0);
498  }
499  //reset keyboard
500  if(!(exceptDevice & (1<<0)))
501  {
502  hid_command_t k;
503  for(uint8_t i=1;i<=8;i++) k.data[i] = 0;
504  k.data[0] = 'K';
505  k.len = 9;
506  //send to queue
507  xQueueSend(hid_usb, &k, 0);
508  }
509  //reset joystick
510  if(!(exceptDevice & (1<<1)))
511  {
512  hid_command_t j;
513  for(uint8_t i=1;i<=12;i++) j.data[i] = 0;
514  j.data[0] = 'J';
515  j.len = 13;
516 
517  //send to queue
518  xQueueSend(hid_usb, &j, 0);
519  }
520 }

557 esp_err_t halSerialInit(void)
558 {
559  //set log level to given log level
560  esp_log_level_set(LOG_TAG,LOG_LEVEL_SERIAL);
561 
562  /*++++ UART config (UART to CDC) ++++*/
563 
564  esp_err_t ret = ESP_OK;
565  const uart_config_t uart_config = {
566  .baud_rate = 230400,
567  .data_bits = UART_DATA_8_BITS,
568  .parity = UART_PARITY_DISABLE,
569  .stop_bits = UART_STOP_BITS_1,
570  .flow_ctrl = UART_HW_FLOWCTRL_DISABLE
571  };
572 
573  //update UART config
574  ret = uart_param_config(HAL_SERIAL_UART, &uart_config);
575  if(ret != ESP_OK)
576  {
577  ESP_LOGE(LOG_TAG,"UART param config failed");
578  return ret;
579  }
580 
581  //set IO pins
582  ret = uart_set_pin(HAL_SERIAL_UART, HAL_SERIAL_TXPIN, HAL_SERIAL_RXPIN, UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE);
583  if(ret != ESP_OK)
584  {
585  ESP_LOGE(LOG_TAG,"UART set pin failed");
586  return ret;
587  }
588 
589  //Install UART driver with RX and TX buffers
590  ret = uart_driver_install(HAL_SERIAL_UART, BUF_SIZE_RX, BUF_SIZE_TX, 0, NULL, 0);
591  if(ret != ESP_OK)
592  {
593  ESP_LOGE(LOG_TAG,"UART driver install failed");
594  return ret;
595  }
596 
597  //create mutex for all tasks sending to serial TX queue.
598  //avoids splitting of different packets due to preemption
599  serialsendingsem = xSemaphoreCreateMutex();
600  if(serialsendingsem == NULL)
601  {
602  ESP_LOGE(LOG_TAG,"Cannot create semaphore for TX");
603  return ESP_FAIL;
604  }
605  //mutex for sending HID commands
606  //acquired before sending, released in RMT finish callback
607  hidsendingsem = xSemaphoreCreateMutex();
608  if(hidsendingsem == NULL)
609  {
610  ESP_LOGE(LOG_TAG,"Cannot create semaphore for HID");
611  return ESP_FAIL;
612  }
613 
614  //create the AT command queue
615  halSerialATCmds = xQueueCreate(CMDQUEUE_SIZE,sizeof(atcmd_t));
616 
617  /*++++ RMT config (sending HID commands) ++++*/
618  rmt_config_t hidoutput_rmt;
619 
620  hidoutput_rmt.rmt_mode = RMT_MODE_TX; //TX mode
621  hidoutput_rmt.channel = HAL_SERIAL_HID_CHANNEL; //use define RMT channel
622  hidoutput_rmt.clk_div = 10; //8MHz clock; we need to be fast...
623  hidoutput_rmt.gpio_num = HAL_SERIAL_HIDPIN; //output pin
624  hidoutput_rmt.mem_block_num = 1;
625  hidoutput_rmt.tx_config.loop_en = 0; //do not loop
626  hidoutput_rmt.tx_config.carrier_en = 0; //we don't need a carrier
627  hidoutput_rmt.tx_config.idle_output_en = 1;
628  hidoutput_rmt.tx_config.idle_level = 0;
629 
630  if(rmt_config(&hidoutput_rmt) != ESP_OK)
631  {
632  ESP_LOGE(LOG_TAG,"Cannot init RMT for HID output");
633  return ESP_FAIL;
634  }
635 
636  if(rmt_driver_install(HAL_SERIAL_HID_CHANNEL,0,0) != ESP_OK)
637  {
638  ESP_LOGE(LOG_TAG,"Cannot install driver for HID output");
639  return ESP_FAIL;
640  }
641 
642  /*++++ task setup ++++*/
643  //task for sending HID commands via RMT
644  xTaskCreate(halSerialHIDTask, "serialHID", HAL_SERIAL_TASK_STACKSIZE+256, NULL, configMAX_PRIORITIES-3, NULL);
645 
646  //Create a task to handler UART event from ISR
647  xTaskCreate(halSerialRXTask, "serialRX", HAL_SERIAL_TASK_STACKSIZE, NULL, 5, NULL);
648 
649  //everything went fine
650  ESP_LOGI(LOG_TAG,"Driver installation complete");
651  return ESP_OK;
652 }
