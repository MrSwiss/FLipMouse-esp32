75 esp_err_t halSerialSwitchTXChannel(uint8_t target_level, TickType_t ticks_to_wait)
76 {
77  //check if we are currently on target channel/level
78  if(gpio_get_level(HAL_SERIAL_SWITCHPIN) == target_level)
79  {
80  //wait for any previously sent transmission to finish
81  if(uart_wait_tx_done(HAL_SERIAL_UART, ticks_to_wait) != ESP_OK)
82  {
83  ESP_LOGE(LOG_TAG,"tx_done not finished on time");
84  return ESP_FAIL;
85  }
86  //set GPIO to route data to USB-Serial
87  gpio_set_level(HAL_SERIAL_SWITCHPIN, target_level);
88  }
89  return ESP_OK;
90 }

92 void halSerialTaskKeyboardPress(void *param)
93 {
94  uint16_t rxK = 0;
95  uint8_t keycode;
96  uint8_t keycodesLocal[8] = {'K',0,0,0,0,0,0,0};
97 
98  while(1)
99  {
100  //check if queue is created
101  if(keyboard_usb_press != 0)
102  {
103  //pend on MQ, if timeout triggers, just wait again.
104  if(xQueueReceive(keyboard_usb_press,&rxK,300))
105  {
106  //data here, parse keyboard press command
107  //check if this is a unicode/ascii byte or a keycode
108  if((rxK & 0xFF00) == 0)
109  {
110  //single bytes are ascii or unicode bytes, sent to keycode parser
111  keycode = parse_for_keycode((uint8_t) rxK, currentConfig.locale, &keycode_modifier, &keycode_deadkey_first);
112  if(keycode != 0)
113  {
114  //if a deadkey is issued (no release necessary), we send the deadkey before:
115  if(keycode_deadkey_first != 0)
116  {
117  ESP_LOGD(LOG_TAG,"Sending deadkey first: %d",keycode_deadkey_first);
118  memset(&keycodesLocal[1],0,7);
119  keycodesLocal[2] = keycode_deadkey_first;
120  //send to device (wait maximum of 30 ticks)
121  if(halSerialSendUSBSerial(HAL_SERIAL_TX_TO_HID,(char *)keycodesLocal, sizeof(keycodesLocal), 30) != ESP_OK)
122  {
123  ESP_LOGE(LOG_TAG,"Error sending keyboard deadkey press report");
124  }
125  //wait 1 tick to release key
126  vTaskDelay(1); //suspend for 1 tick
127  keycodesLocal[2] = 0;
128  //send to device
129  if(halSerialSendUSBSerial(HAL_SERIAL_TX_TO_HID,(char *)keycodesLocal, sizeof(keycodesLocal), 30) != ESP_OK)
130  {
131  ESP_LOGE(LOG_TAG,"Error sending keyboard deadkey release report");
132  }
133  }
134  } //we need another byte to calculate keycode...
135 
136  } else {
137  keycode = rxK & 0x00FF;
138  }
139 
140  //keycodes, add directly to the keycode array
141  switch(add_keycode(keycode,&(keycode_arr[2])))
142  {
143  case 0:
144  keycode_arr[0] = 'K';
145  keycode_arr[1] = keycode_modifier;
146  if(halSerialSendUSBSerial(HAL_SERIAL_TX_TO_HID,(char *)keycode_arr, sizeof(keycode_arr), 30) != ESP_OK)
147  {
148  ESP_LOGE(LOG_TAG,"Error sending keyboard HID report");
149  }
150  ESP_LOGD(LOG_TAG,"HID report (on press):");
151  esp_log_buffer_hex(LOG_TAG,keycode_arr,8);
152  break;
153  case 1: ESP_LOGW(LOG_TAG,"keycode already in queue"); break;
154  case 2: ESP_LOGE(LOG_TAG,"no space in keycode arr!"); break;
155  default: ESP_LOGE(LOG_TAG,"add_keycode return unknown code..."); break;
156  }
157 
158  keycode = 0;
159  }
160  } else {
161  ESP_LOGE(LOG_TAG,"keyboard_usb_press queue not initialized, retry in 1s");
162  vTaskDelay(1000/portTICK_PERIOD_MS);
163  }
164  }
165 }

215 void halSerialTaskMouse(void *param)
216 {
217  mouse_command_t rxM;
218  static uint8_t emptySent = 0;
219  mouse_command_t emptyReport = {0,0,0,0};
220  uint8_t hidreport[5] = {'M',0,0,0,0};
221 
222  while(1)
223  {
224  //check if queue is created
225  if(mouse_movement_usb != 0)
226  {
227  //pend on MQ, if timeout triggers, just wait again.
228  if(xQueueReceive(mouse_movement_usb,&rxM,300))
229  {
230  //send report always if not empty
231  if(memcmp(&rxM,&emptyReport,sizeof(mouse_command_t)) != 0)
232  {
233  //copy mouse command to hid report
234  memcpy(&hidreport[1],&rxM,4);
235  if(halSerialSendUSBSerial(HAL_SERIAL_TX_TO_HID,(char *)hidreport,sizeof(hidreport),30) != ESP_OK)
236  {
237  ESP_LOGE(LOG_TAG,"Error sending mouse report");
238  }
239  ESP_LOGD(LOG_TAG,"Mouse: %d,%d,%d,%d",rxM.buttons, \
240  rxM.x, rxM.y, rxM.wheel);
241  //after one non-empty report, we can resend an empty one
242  emptySent = 0;
243  } else {
244  //if empty, check if it was sent once, if not: send empty
245  if(emptySent == 0)
246  {
247  memcpy(&hidreport[1],&rxM,4);
248  if(halSerialSendUSBSerial(HAL_SERIAL_TX_TO_HID,(char *)hidreport,sizeof(hidreport),30) != ESP_OK)
249  {
250  ESP_LOGE(LOG_TAG,"Error sending mouse report");
251  }
252  ESP_LOGD(LOG_TAG,"Mouse: %d,%d,%d,%d",rxM.buttons, \
253  rxM.x, rxM.y, rxM.wheel);
254  //remember we sent an empty report
255  emptySent = 1;
256  }
257  }
258  }
259  } else {
260  ESP_LOGE(LOG_TAG,"mouse_movement_usb queue not initialized, retry in 1s");
261  vTaskDelay(1000/portTICK_PERIOD_MS);
262  }
263  }

319 int halSerialSendUSBSerial(uint8_t channel, char *data, uint32_t length, TickType_t ticks_to_wait)
320 {
321  //acquire mutex to have TX permission on UART
322  if(xSemaphoreTake(serialsendingsem, ticks_to_wait) == pdTRUE)
323  {
324  //try to switch to USB-Serial
325  if(halSerialSwitchTXChannel(channel,ticks_to_wait)!=ESP_OK)
326  {
327  ESP_LOGE(LOG_TAG,"switch GPIO channel didn't finish");
328  xSemaphoreGive(serialsendingsem);
329  return -1;
330  }
331  //send the data
332  int txBytes = uart_write_bytes(HAL_SERIAL_UART, data, length);
333 
334  //release mutex
335  xSemaphoreGive(serialsendingsem);
336 
337  return txBytes;
338  } else return -1;
339 }

346 esp_err_t halSerialInit(void)
347 {
348  esp_err_t ret = ESP_OK;
349  const uart_config_t uart_config = {
350  .baud_rate = 115200,
351  .data_bits = UART_DATA_8_BITS,
352  .parity = UART_PARITY_DISABLE,
353  .stop_bits = UART_STOP_BITS_1,
354  .flow_ctrl = UART_HW_FLOWCTRL_DISABLE
355  };
356  //update UART config
357  ret = uart_param_config(HAL_SERIAL_UART, &uart_config);
358  if(ret != ESP_OK) ESP_LOGE(LOG_TAG,"UART param config failed"); return ret;
359  //set IO pins
360  ret = uart_set_pin(HAL_SERIAL_UART, HAL_SERIAL_TXPIN, HAL_SERIAL_RXPIN, UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE);
361  if(ret != ESP_OK) ESP_LOGE(LOG_TAG,"UART set pin failed"); return ret;
362  //Install UART driver with RX and TX buffers
363  ret = uart_driver_install(HAL_SERIAL_UART, BUF_SIZE_RX, BUF_SIZE_TX, 0, NULL, 0);
364  if(ret != ESP_OK) ESP_LOGE(LOG_TAG,"UART driver install failed"); return ret;
365  //Setup GPIO for UART sending direction (either use UART data to drive
366  //USB HID or send UART data USB-serial)
367  ret = gpio_set_direction(HAL_SERIAL_SWITCHPIN,GPIO_MODE_OUTPUT);
368  if(ret != ESP_OK) ESP_LOGE(LOG_TAG,"UART GPIO signal set direction failed"); return ret;
369 
370  //create mutex for all tasks sending to serial TX queue.
371  //avoids splitting of different packets due to preemption
372  serialsendingsem = xSemaphoreCreateMutex();
373  if(serialsendingsem == NULL) ESP_LOGE(LOG_TAG,"Cannot create semaphore"); return ESP_FAIL;
374 
375  //install serial tasks (4x -> keyboard press/release; mouse; joystick)
376  xTaskCreate(halSerialTaskKeyboardPress, "serialKbdPress", HAL_SERIAL_TASK_STACKSIZE, NULL, configMAX_PRIORITIES-1, NULL);
377  xTaskCreate(halSerialTaskKeyboardRelease, "serialKbdRelease", HAL_SERIAL_TASK_STACKSIZE, NULL, configMAX_PRIORITIES-1, NULL);
378  xTaskCreate(halSerialTaskMouse, "serialMouse", HAL_SERIAL_TASK_STACKSIZE, NULL, configMAX_PRIORITIES-1, NULL);
379  xTaskCreate(halSerialTaskJoystick, "serialJoystick", HAL_SERIAL_TASK_STACKSIZE, NULL, configMAX_PRIORITIES-1, NULL);
380 
381  //set GPIO to route data to USB-HID
382  gpio_set_level(HAL_SERIAL_SWITCHPIN, 0);
383 
384  //everything went fine
385  ESP_LOGI(LOG_TAG,"Driver installation complete");
386  return ESP_OK;
387 }
