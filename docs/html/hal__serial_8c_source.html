114 esp_err_t halSerialSwitchTXChannel(uint8_t target_level, TickType_t ticks_to_wait)
115 {
116  static uint8_t current_level = 0;
117  //check if we are currently on target channel/level
118  if(current_level != target_level)
119  {

155 static void halSerialEventTask(void *pvParameters)
156 {
157  uart_event_t event;
158  for(;;) {
159  //Waiting for UART event.
160  if(xQueueReceive(uarteventsqueue, (void * )&event, (portTickType)portMAX_DELAY)) {
161  switch(event.type) {
162  //Event of UART receving data. Not used here, data is read
163  //by halSerialReceiveUSBSerial
164  case UART_DATA:
165  break;
166 
167  //Event of UART ring buffer full
168  // & FIFO overflow: reset queue & flush uart
169  case UART_FIFO_OVF:
170  case UART_BUFFER_FULL:
171  ESP_LOGW(LOG_TAG,"FIFO/buffer full, maybe missing UART consumer?");
172  uart_flush_input(HAL_SERIAL_UART);
173  xQueueReset(uarteventsqueue);
174  break;
175  //UART_PATTERN_DET, give semaphore to unblock pending
176  //UART readers...
177  case UART_PATTERN_DET:
178  xSemaphoreGive(serialpatternsem);
179  break;
180  //Others, do nothing....
181  default: break;
182  }
183  }
184  }
185  vTaskDelete(NULL);
186 }

188 void halSerialTaskKeyboardPress(void *param)
189 {
190  uint16_t rxK = 0;
191  uint8_t keycode;
192  uint8_t modifier;
193 
194  while(1)
195  {
196  //check if queue is created
197  if(keyboard_usb_press != 0)
198  {
199  //pend on MQ, if timeout triggers, just wait again.
200  if(xQueueReceive(keyboard_usb_press,&rxK,300))
201  {
202  //data here, split into modifier & keycode
203  keycode = rxK & 0x00FF;
204  modifier = (rxK & 0xFF00)>>8;
205 
206  //add modifier to global mask
207  keycode_modifier |= modifier;
208 
209  //keycodes, add directly to the keycode array
210  switch(add_keycode(keycode,&(keycode_arr[2])))
211  {
212  case 0:
213  keycode_arr[0] = 'K';
214  keycode_arr[1] = keycode_modifier;
215  uint8_t sent = 0;
216  for(uint8_t i = 0; i<8; i++) { sent+= halSerialSendUSBSerial(HAL_SERIAL_TX_TO_HID,(char *)&keycode_arr[i], 1, 30); }
217  if(sent != 8) ESP_LOGE(LOG_TAG,"Error sending keyboard HID report");
218 
219  ESP_LOGD(LOG_TAG,"HID report (on press):");
220  esp_log_buffer_hex(LOG_TAG,keycode_arr,8);
221  break;
222  case 1: ESP_LOGW(LOG_TAG,"keycode already in queue"); break;
223  case 2: ESP_LOGE(LOG_TAG,"no space in keycode arr!"); break;
224  default: ESP_LOGE(LOG_TAG,"add_keycode return unknown code..."); break;
225  }
226 
227  modifier = 0;
228  keycode = 0;
229  rxK = 0;
230  }
231  } else {
232  ESP_LOGE(LOG_TAG,"keyboard_usb_press queue not initialized, retry in 1s");
233  vTaskDelay(1000/portTICK_PERIOD_MS);
234  }
235  }

314  {
315  ESP_LOGE(LOG_TAG,"Error sending report:");
316  }
317  ESP_LOGD(LOG_TAG,"Mouse: %d,%d,%d,%d",rxM.buttons, \

389 int halSerialReceiveUSBSerial(uint8_t *data, uint32_t length)
390 {
391  if(xSemaphoreTake(serialpatternsem,HAL_SERIAL_UART_TIMEOUT_MS / portTICK_PERIOD_MS) == pdTRUE)
392  {
393  //got this semaphore on time, return data & length
394  size_t buffered_size;
395  uart_get_buffered_data_len(HAL_SERIAL_UART, &buffered_size);
396  if(buffered_size < length)
397  {
398  int pos = uart_pattern_pop_pos(HAL_SERIAL_UART);
399  ESP_LOGI(LOG_TAG, "pos: %d, buffered size: %d", pos, buffered_size);
400  return uart_read_bytes(HAL_SERIAL_UART, data, buffered_size, HAL_SERIAL_UART_TIMEOUT_MS / portTICK_PERIOD_MS);
401  } else {
402  ESP_LOGE(LOG_TAG,"Provide bigger buffer for UART receive (buffer size %d, buffered %d)",length,buffered_size);
403  return -1;
404  }
405  } else {
406  //no pattern detected on time...
407  return -1;
408  }
409 }

427 int halSerialSendUSBSerial(uint8_t channel, char *data, uint32_t length, TickType_t ticks_to_wait)
428 {
429  //acquire mutex to have TX permission on UART
430  if(xSemaphoreTake(serialsendingsem, ticks_to_wait) == pdTRUE)
431  {
432  //try to switch to USB-Serial
433  if(halSerialSwitchTXChannel(channel,ticks_to_wait)!=ESP_OK)
434  {
435  ESP_LOGE(LOG_TAG,"switch GPIO channel didn't finish");
436  xSemaphoreGive(serialsendingsem);
437  return -1;
438  }
439  //send the data
440  int txBytes = uart_write_bytes(HAL_SERIAL_UART, data, length);
441 
442  //release mutex
443  xSemaphoreGive(serialsendingsem);
444 
445  return txBytes;
446  } else return -1;
447 }

501 esp_err_t halSerialInit(void)
502 {
503  esp_err_t ret = ESP_OK;
504  const uart_config_t uart_config = {
505  .baud_rate = 115200,
506  .data_bits = UART_DATA_8_BITS,
507  .parity = UART_PARITY_DISABLE,
508  .stop_bits = UART_STOP_BITS_1,
509  .flow_ctrl = UART_HW_FLOWCTRL_DISABLE
510  };
511 
512  //update UART config
513  ret = uart_param_config(HAL_SERIAL_UART, &uart_config);
514  if(ret != ESP_OK)
515  {
516  ESP_LOGE(LOG_TAG,"UART param config failed");
517  return ret;
518  }
519 
520  //set IO pins
521  ret = uart_set_pin(HAL_SERIAL_UART, HAL_SERIAL_TXPIN, HAL_SERIAL_RXPIN, UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE);
522  if(ret != ESP_OK)
523  {
524  ESP_LOGE(LOG_TAG,"UART set pin failed");
525  return ret;
526  }
527 
528  //Install UART driver with RX and TX buffers
529  ret = uart_driver_install(HAL_SERIAL_UART, BUF_SIZE_RX, BUF_SIZE_TX, 10, &uarteventsqueue, 0);
530  if(ret != ESP_OK)
531  {
532  ESP_LOGE(LOG_TAG,"UART driver install failed");
533  return ret;
534  }
535 
536  if(uarteventsqueue == NULL)
537  {
538  ESP_LOGE(LOG_TAG,"Cannot create queue for UART events");
539  return ESP_FAIL;
540  }
541 
542  //Setup GPIO for UART sending direction (either use UART data to drive
543  //USB HID or send UART data USB-serial)
544  ret = gpio_set_direction(HAL_SERIAL_SWITCHPIN,GPIO_MODE_OUTPUT);
545  if(ret != ESP_OK)
546  {
547  ESP_LOGE(LOG_TAG,"UART GPIO signal set direction failed");
548  return ret;
549  }
550 
551  //create mutex for all tasks sending to serial TX queue.
552  //avoids splitting of different packets due to preemption
553  serialsendingsem = xSemaphoreCreateMutex();
554  if(serialsendingsem == NULL)
555  {
556  ESP_LOGE(LOG_TAG,"Cannot create semaphore for TX");
557  return ESP_FAIL;
558  }
559 
560  //create mutex for all tasks sending to serial TX queue.
561  //avoids splitting of different packets due to preemption
562  serialpatternsem = xSemaphoreCreateMutex();
563  if(serialpatternsem == NULL)
564  {
565  ESP_LOGE(LOG_TAG,"Cannot create semaphore for pattern detection");
566  return ESP_FAIL;
567  }
568 
569  //Set uart pattern detect function.
570  uart_enable_pattern_det_intr(HAL_SERIAL_UART, '\n', 1, 10000, 10, 10);
571  //Reset the pattern queue length to record at most 10 pattern positions.
572  uart_pattern_queue_reset(HAL_SERIAL_UART, 10);
573 
574  //install serial tasks (4x -> keyboard press/release; mouse; joystick)
575  xTaskCreate(halSerialTaskKeyboardPress, "serialKbdPress", HAL_SERIAL_TASK_STACKSIZE, NULL, configMAX_PRIORITIES-1, NULL);
576  xTaskCreate(halSerialTaskKeyboardRelease, "serialKbdRelease", HAL_SERIAL_TASK_STACKSIZE, NULL, configMAX_PRIORITIES-1, NULL);
577  xTaskCreate(halSerialTaskMouse, "serialMouse", HAL_SERIAL_TASK_STACKSIZE, NULL, configMAX_PRIORITIES-1, NULL);
578  xTaskCreate(halSerialTaskJoystick, "serialJoystick", HAL_SERIAL_TASK_STACKSIZE, NULL, configMAX_PRIORITIES-1, NULL);
579 
580  //Create a task to handler UART event from ISR
581  xTaskCreate(halSerialEventTask, "serialeventT", HAL_SERIAL_TASK_STACKSIZE, NULL, configMAX_PRIORITIES-1, NULL);
582 
583  //set GPIO to route data to USB-HID
584  gpio_set_level(HAL_SERIAL_SWITCHPIN, 0);
585 
586  //everything went fine
587  ESP_LOGI(LOG_TAG,"Driver installation complete");
588  return ESP_OK;
589 }
