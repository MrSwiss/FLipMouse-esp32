155 static void halSerialRXTask(void *pvParameters)
156 {
157  uint16_t cmdoffset;
158  uint8_t *buf;
159  uint8_t data;
160  uint8_t parserstate = 0;
161  atcmd_t currentcmd;
162 
163  while(1)
164  {
165  int len = uart_read_bytes(HAL_SERIAL_UART, &data, 1, portMAX_DELAY);
166 
167  //only parse if a valid byte is received
168  if(len == 1)
169  {
170  switch(parserstate)
171  {
172  case 0:
173  //check for leading "A"
174  if(data == 'A' || data == 'a')
175  {
177  parserstate++;
178  }
179  //if starting with '@', go to CIM mode.
180  if(data == '@')
181  {
183  ESP_LOGW(LOG_TAG,"CIM mdoe currently unsupported!");
184  }
185  break;
186 
187 
188  case 1:
189  //check for second 'T'/'t'
190  if(data == 'T' || data == 't')
191  {
192  parserstate++;
193  //reset command offset to beginning (but after "AT")
194  cmdoffset = 2;
195  //allocate new buffer for command & save "AT"
196  //free is either done in error case or in halSerialReceiveUSBSerial.
197  buf = malloc(ATCMD_LENGTH*sizeof(uint8_t));
198  if(buf != NULL)
199  {
200  buf[0] = 'A';
201  buf[1] = 'T';
202  } else {
203  ESP_LOGE(LOG_TAG,"Cannot allocate buffer for command");
204  parserstate = 0;
205  }
206  } else {
207  //reset parser of no leading "AT" is detected
208  parserstate = 0;
209  }
210  break;
211 
212 
213  case 2:
214  //now read data until we reach \r or \n
215  if(data == '\r' || data == '\n')
216  {
217  //terminate string
218  buf[cmdoffset] = 0;
219  //send buffer to queue
220  currentcmd.buf = buf;
221  currentcmd.len = cmdoffset+1;
222  if(cmds != NULL)
223  {
224  if(xQueueSend(cmds,(void*)&currentcmd,10) != pdTRUE)
225  {
226  ESP_LOGE(LOG_TAG,"AT cmd queue is full, cannot send cmd");
227  free(buf);
228  } else {
229  ESP_LOGI(LOG_TAG,"Sent AT cmd with len %d to queue",cmdoffset);
230  }
231  } else {
232  ESP_LOGE(LOG_TAG,"AT cmd queue is NULL, cannot send cmd");
233  free(buf);
234  }
235  parserstate = 0;
236  break;
237  }
238 
239  //if everything is fine, just save this byte.
240  buf[cmdoffset] = data;
241  cmdoffset++;
242 
243  //check for memory length
244  if(cmdoffset == ATCMD_LENGTH)
245  {
246  ESP_LOGW(LOG_TAG,"AT cmd too long, discarding");
247  free(buf);
248  parserstate = 0;
249  }
250  break;
251 
252  default:
253  ESP_LOGE(LOG_TAG,"Unknown parser state");
254  parserstate = 0;
255  break;
256  }
257  }
258  }
259 
260  //we should never be here...
261  vTaskDelete(NULL);
262 }

513 int halSerialSendUSBSerial(uint8_t channel, char *data, uint32_t length, TickType_t ticks_to_wait)
514 {
515  //check if sem is already initialized
516  if(serialsendingsem == NULL)
517  {
518  ESP_LOGW(LOG_TAG,"Sem not ready, waiting for 1s");
519  vTaskDelay(1000/portTICK_PERIOD_MS);
520  }
521 
522  //acquire mutex to have TX permission on UART
523  if(xSemaphoreTake(serialsendingsem, ticks_to_wait) == pdTRUE)
524  {
525  //try to switch to USB-Serial
526  if(halSerialSwitchTXChannel(channel,ticks_to_wait)!=ESP_OK)
527  {
528  ESP_LOGE(LOG_TAG,"switch GPIO channel didn't finish");
529  xSemaphoreGive(serialsendingsem);
530  return -1;
531  }
532  //send the data
533  int txBytes = uart_write_bytes(HAL_SERIAL_UART, data, length);
534 
535  //release mutex
536  xSemaphoreGive(serialsendingsem);
537 
538  return txBytes;
539  } else return -1;
540 }

594 esp_err_t halSerialInit(void)
595 {
596  esp_err_t ret = ESP_OK;
597  const uart_config_t uart_config = {
598  .baud_rate = 115200,
599  .data_bits = UART_DATA_8_BITS,
600  .parity = UART_PARITY_DISABLE,
601  .stop_bits = UART_STOP_BITS_1,
602  .flow_ctrl = UART_HW_FLOWCTRL_DISABLE
603  };
604 
605  //update UART config
606  ret = uart_param_config(HAL_SERIAL_UART, &uart_config);
607  if(ret != ESP_OK)
608  {
609  ESP_LOGE(LOG_TAG,"UART param config failed");
610  return ret;
611  }
612 
613  //set IO pins
614  ret = uart_set_pin(HAL_SERIAL_UART, HAL_SERIAL_TXPIN, HAL_SERIAL_RXPIN, UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE);
615  if(ret != ESP_OK)
616  {
617  ESP_LOGE(LOG_TAG,"UART set pin failed");
618  return ret;
619  }
620 
621  //Install UART driver with RX and TX buffers
622  ret = uart_driver_install(HAL_SERIAL_UART, BUF_SIZE_RX, BUF_SIZE_TX, 0, NULL, 0);
623  if(ret != ESP_OK)
624  {
625  ESP_LOGE(LOG_TAG,"UART driver install failed");
626  return ret;
627  }
628 
629  //Setup GPIO for UART sending direction (either use UART data to drive
630  //USB HID or send UART data USB-serial)
631  ret = gpio_set_direction(HAL_SERIAL_SWITCHPIN,GPIO_MODE_OUTPUT);
632  if(ret != ESP_OK)
633  {
634  ESP_LOGE(LOG_TAG,"UART GPIO signal set direction failed");
635  return ret;
636  }
637 
638  //create mutex for all tasks sending to serial TX queue.
639  //avoids splitting of different packets due to preemption
640  serialsendingsem = xSemaphoreCreateMutex();
641  if(serialsendingsem == NULL)
642  {
643  ESP_LOGE(LOG_TAG,"Cannot create semaphore for TX");
644  return ESP_FAIL;
645  }
646 
647  //create the AT command queue
648  cmds = xQueueCreate(CMDQUEUE_SIZE,sizeof(atcmd_t));
649 
650  //Set uart pattern detect function.
651  //uart_enable_pattern_det_intr(HAL_SERIAL_UART, '\r', 1, 10000, 10, 10);
652  //Reset the pattern queue length to record at most 10 pattern positions.
653  //uart_pattern_queue_reset(HAL_SERIAL_UART, EVENTQUEUE_SIZE);
654 
655  //install serial tasks (4x -> keyboard press/release; mouse; joystick)
656  xTaskCreate(halSerialTaskKeyboardPress, "serialKbdPress", HAL_SERIAL_TASK_STACKSIZE, NULL, configMAX_PRIORITIES-1, NULL);
657  xTaskCreate(halSerialTaskKeyboardRelease, "serialKbdRelease", HAL_SERIAL_TASK_STACKSIZE, NULL, configMAX_PRIORITIES-1, NULL);
658  xTaskCreate(halSerialTaskMouse, "serialMouse", HAL_SERIAL_TASK_STACKSIZE, NULL, configMAX_PRIORITIES-1, NULL);
659  xTaskCreate(halSerialTaskJoystick, "serialJoystick", HAL_SERIAL_TASK_STACKSIZE, NULL, configMAX_PRIORITIES-1, NULL);
660 
661  //Create a task to handler UART event from ISR
662  xTaskCreate(halSerialRXTask, "serialRX", HAL_SERIAL_TASK_STACKSIZE, NULL, configMAX_PRIORITIES-1, NULL);
663 
664  //set GPIO to route data to USB-HID
665  gpio_set_level(HAL_SERIAL_SWITCHPIN, 0);
666 
667  //everything went fine
668  ESP_LOGI(LOG_TAG,"Driver installation complete");
669  return ESP_OK;
670 }
