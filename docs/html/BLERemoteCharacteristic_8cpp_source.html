149 void BLERemoteCharacteristic::gattClientEventHandler(
150  esp_gattc_cb_event_t event,
151  esp_gatt_if_t gattc_if,
152  esp_ble_gattc_cb_param_t* evtParam) {
153  switch(event) {
154  //
155  // ESP_GATTC_NOTIFY_EVT
156  //
157  // notify
158  // - uint16_t conn_id - The connection identifier of the server.
159  // - esp_bd_addr_t remote_bda - The device address of the BLE server.
160  // - uint16_t handle - The handle of the characteristic for which the event is being received.
161  // - uint16_t value_len - The length of the received data.
162  // - uint8_t* value - The received data.
163  // - bool is_notify - True if this is a notify, false if it is an indicate.
164  //
165  // We have received a notification event which means that the server wishes us to know about a notification
166  // piece of data. What we must now do is find the characteristic with the associated handle and then
167  // invoke its notification callback (if it has one).
168  //
169  case ESP_GATTC_NOTIFY_EVT: {
170  if (evtParam->notify.handle != getHandle()) {
171  break;
172  }
173  if (m_notifyCallback != nullptr) {
174  ESP_LOGD(LOG_TAG, "Invoking callback for notification on characteristic %s", toString().c_str());
175  m_notifyCallback(
176  this,
177  evtParam->notify.value,
178  evtParam->notify.value_len,
179  evtParam->notify.is_notify
180  );
181  } // End we have a callback function ...
182  break;
183  } // ESP_GATTC_NOTIFY_EVT
184 
185 
186  //
187  // ESP_GATTC_READ_CHAR_EVT
188  // This event indicates that the server has responded to the read request.
189  //
190  // read:
191  // - esp_gatt_status_t status
192  // - uint16_t conn_id
193  // - uint16_t handle
194  // - uint8_t* value
195  // - uint16_t value_len
196  //
197  case ESP_GATTC_READ_CHAR_EVT: {
198  // If this event is not for us, then nothing further to do.
199  if (evtParam->read.handle != getHandle()) {
200  break;
201  }
202 
203  // At this point, we have determined that the event is for us, so now we save the value
204  // and unlock the semaphore to ensure that the requestor of the data can continue.
205  if (evtParam->read.status == ESP_GATT_OK) {
206  m_value = std::string((char*)evtParam->read.value, evtParam->read.value_len);
207  } else {
208  m_value = "";
209  }
210 
211  m_semaphoreReadCharEvt.give();
212  break;
213  } // ESP_GATTC_READ_CHAR_EVT
214 
215 
216  //
217  // ESP_GATTC_REG_FOR_NOTIFY_EVT
218  //
219  // reg_for_notify:
220  // - esp_gatt_status_t status
221  // - uint16_t handle
222  //
223  case ESP_GATTC_REG_FOR_NOTIFY_EVT: {
224  // If the request is not for this BLERemoteCharacteristic then move on to the next.
225  if (evtParam->reg_for_notify.handle != getHandle()) {
226  break;
227  }
228 
229  // We have processed the notify registration and can unlock the semaphore.
230  m_semaphoreRegForNotifyEvt.give();
231  break;
232  } // ESP_GATTC_REG_FOR_NOTIFY_EVT
233 
234 
235  //
236  // ESP_GATTC_UNREG_FOR_NOTIFY_EVT
237  //
238  // unreg_for_notify:
239  // - esp_gatt_status_t status
240  // - uint16_t handle
241  //
242  case ESP_GATTC_UNREG_FOR_NOTIFY_EVT: {
243  if (evtParam->unreg_for_notify.handle != getHandle()) {
244  break;
245  }
246  // We have processed the notify un-registration and can unlock the semaphore.
247  m_semaphoreRegForNotifyEvt.give();
248  break;
249  } // ESP_GATTC_UNREG_FOR_NOTIFY_EVT:
250 
251 
252  //
253  // ESP_GATTC_WRITE_CHAR_EVT
254  //
255  // write:
256  // - esp_gatt_status_t status
257  // - uint16_t conn_id
258  // - uint16_t handle
259  //
260  case ESP_GATTC_WRITE_CHAR_EVT: {
261  // Determine if this event is for us and, if not, pass onwards.
262  if (evtParam->write.handle != getHandle()) {
263  break;
264  }
265 
266  // There is nothing further we need to do here. This is merely an indication
267  // that the write has completed and we can unlock the caller.
268  m_semaphoreWriteCharEvt.give();
269  break;
270  } // ESP_GATTC_WRITE_CHAR_EVT
271 
272 
273  default: {
274  break;
275  }
276  } // End switch
277 }; // gattClientEventHandler

283 void BLERemoteCharacteristic::retrieveDescriptors() {
284  ESP_LOGD(LOG_TAG, ">> retrieveDescriptors() for characteristic: %s", getUUID().toString().c_str());
285 
286  removeDescriptors(); // Remove any existing descriptors.
287 
288  // Loop over each of the descriptors within the service associated with this characteristic.
289  // For each descriptor we find, create a BLERemoteDescriptor instance.
290  uint16_t offset = 0;
291  esp_gattc_descr_elem_t result;
292  while(1) {
293  uint16_t count = 1;
294  esp_gatt_status_t status = ::esp_ble_gattc_get_all_descr(
295  getRemoteService()->getClient()->getGattcIf(),
296  getRemoteService()->getClient()->getConnId(),
297  getHandle(),
298  &result,
299  &count,
300  offset
301  );
302 
303  if (status == ESP_GATT_INVALID_OFFSET) { // We have reached the end of the entries.
304  break;
305  }
306 
307  if (status != ESP_GATT_OK) {
308  ESP_LOGE(LOG_TAG, "esp_ble_gattc_get_all_descr: %s", BLEUtils::gattStatusToString(status).c_str());
309  break;
310  }
311 
312  if (count == 0) {
313  break;
314  }
315  ESP_LOGE(LOG_TAG, "");
316  ESP_LOGD(LOG_TAG, "Found a descriptor: Handle: %d, UUID: %s", result.handle, BLEUUID(result.uuid).toString().c_str());
317 
318  // We now have a new characteristic ... let us add that to our set of known characteristics
319  BLERemoteDescriptor *pNewRemoteDescriptor = new BLERemoteDescriptor(
320  result.handle,
321  BLEUUID(result.uuid),
322  this
323  );
324 
325  m_descriptorMap.insert(std::pair<std::string, BLERemoteDescriptor*>(pNewRemoteDescriptor->getUUID().toString(), pNewRemoteDescriptor));
326 
327  offset++;
328  } // while true
329  //m_haveCharacteristics = true; // Remember that we have received the characteristics.
330  ESP_LOGD(LOG_TAG, "<< retrieveDescriptors(): Found %d descriptors.", offset);
331 } // getDescriptors

433 std::string BLERemoteCharacteristic::readValue() {
434  ESP_LOGD(LOG_TAG, ">> readValue(): uuid: %s, handle: %d 0x%.2x", getUUID().toString().c_str(), getHandle(), getHandle());
435 
436  // Check to see that we are connected.
437  if (!getRemoteService()->getClient()->isConnected()) {
438  ESP_LOGE(LOG_TAG, "Disconnected");
439  throw BLEDisconnectedException();
440  }
441 
442  m_semaphoreReadCharEvt.take("readValue");
443 
444  // Ask the BLE subsystem to retrieve the value for the remote hosted characteristic.
445  // This is an asynchronous request which means that we must block waiting for the response
446  // to become available.
447  esp_err_t errRc = ::esp_ble_gattc_read_char(
448  m_pRemoteService->getClient()->getGattcIf(),
449  m_pRemoteService->getClient()->getConnId(), // The connection ID to the BLE server
450  getHandle(), // The handle of this characteristic
451  ESP_GATT_AUTH_REQ_NONE); // Security
452 
453  if (errRc != ESP_OK) {
454  ESP_LOGE(LOG_TAG, "esp_ble_gattc_read_char: rc=%d %s", errRc, GeneralUtils::errorToString(errRc));
455  return "";
456  }
457 
458  // Block waiting for the event that indicates that the read has completed. When it has, the std::string found
459  // in m_value will contain our data.
460  m_semaphoreReadCharEvt.wait("readValue");
461 
462  ESP_LOGD(LOG_TAG, "<< readValue(): length: %d", m_value.length());
463  return m_value;
464 } // readValue
