255 class BLETask : public Task {
256  void run(void*){
257  hid_cmd_t rx;
258 
259  //Empty queue if initialized (there might be something left from last connection)
260  if(hid_ble != nullptr) xQueueReset(hid_ble);
261 
262  //check if queue is initialized
263  if(hid_ble != NULL)
264  {
265  while(1)
266  {
267  //pend on MQ, if timeout triggers, just wait again.
268  if(xQueueReceive(hid_ble,&rx,portMAX_DELAY))
269  {
270  //parse command (similar to usb_bridge controller)
271  switch(rx.cmd[0] & 0xF0)
272  {
273  //reset all reports
274  case 0x00:
275  halBLEReset(0);
276  break;
277  //mouse handling
278  case 0x10:
279  switch(rx.cmd[0] & 0x0F)
280  {
281  case 0: //move X
282  mouse_report[1] = rx.cmd[1];
283  break;
284  case 1: //move Y
285  mouse_report[2] = rx.cmd[1];
286  break;
287  case 2: //move wheel
288  mouse_report[3] = rx.cmd[1];
289  break;
290  /* Press & release */
291  case 3: //left
292  mouse_report[0] |= (1<<0);
293  //send press report, wait for free EP (sending release is done after switch)
294  inputMouse->setValue(mouse_report,sizeof(mouse_report));
295  inputMouse->notify();
296  mouse_report[0] &= ~(1<<0);
297  break;
298  case 4: //right
299  mouse_report[0] |= (1<<1);
300  //send press report, wait for free EP (sending release is done after switch)
301  inputMouse->setValue(mouse_report,sizeof(mouse_report));
302  inputMouse->notify();
303  mouse_report[0] &= ~(1<<1);
304  break;
305  case 5: //middle
306  mouse_report[0] |= (1<<2);
307  //send press report, wait for free EP (sending release is done after switch)
308  inputMouse->setValue(mouse_report,sizeof(mouse_report));
309  inputMouse->notify();
310  mouse_report[0] &= ~(1<<2);
311  break;
312  /* Press */
313  case 6: //left
314  mouse_report[0] |= (1<<0);
315  break;
316  case 7: //right
317  mouse_report[0] |= (1<<1);
318  break;
319  case 8: //middle
320  mouse_report[0] |= (1<<2);
321  break;
322  /* Release */
323  case 9: //left
324  mouse_report[0] &= ~(1<<0);
325  break;
326  case 10: //right
327  mouse_report[0] &= ~(1<<1);
328  break;
329  case 11: //middle
330  mouse_report[0] &= ~(1<<2);
331  break;
332  /* Toggle */
333  case 12: //left
334  mouse_report[0] ^= (1<<0);
335  break;
336  case 13: //right
337  mouse_report[0] ^= (1<<1);
338  break;
339  case 14: //middle
340  mouse_report[0] ^= (1<<2);
341  break;
342  case 15: //reset mouse (excepting keyboard & joystick)
343  halBLEReset((1<<0)|(1<<1));
344  break;
345  }
346  inputMouse->setValue(mouse_report,sizeof(mouse_report));
347  inputMouse->notify();
348  break;
349  //Keyboard handling
350  case 0x20:
351  switch(rx.cmd[0] & 0x0F)
352  {
353  case 0: //Press & release a key
354  //press key & send
355  add_keycode(rx.cmd[1], &keyboard_report[2]);
356  inputKbd->setValue(keyboard_report,sizeof(keyboard_report));
357  inputKbd->notify();
358  //remove keycode
359  //sending the second report is done after this switch
360  remove_keycode(rx.cmd[1], &keyboard_report[2]);
361  break;
362  case 1: //Press a key
363  add_keycode(rx.cmd[1], &keyboard_report[2]);
364  break;
365  case 2: //Release a key
366  remove_keycode(rx.cmd[1], &keyboard_report[2]);
367  break;
368  case 3:
369  if(is_in_keycode_arr(rx.cmd[1],&keyboard_report[2])) remove_keycode(rx.cmd[1], &keyboard_report[2]);
370  else add_keycode(rx.cmd[1], &keyboard_report[2]);
371  break;
372  case 4: //Press & release a modifier (mask!)
373  keyboard_report[0] |= rx.cmd[1];
374  inputKbd->setValue(keyboard_report,sizeof(keyboard_report));
375  inputKbd->notify();
376  //remove modifier
377  //sending the second report is done after this switch
378  keyboard_report[0] &= ~rx.cmd[1];
379  break;
380  case 5: //Press a modifier (mask!)
381  keyboard_report[0] |= rx.cmd[1];
382  break;
383  case 6: //Release a modifier (mask!)
384  keyboard_report[0] &= ~rx.cmd[1];
385  break;
386  case 7: //Toggle a modifier (mask!)
387  keyboard_report[0] ^= rx.cmd[1];
388  break;
389  case 15: //reset mouse (excepting mouse & joystick)
390  halBLEReset((1<<1)|(1<<2));
391  break;
392  }
393  inputKbd->setValue(keyboard_report,sizeof(keyboard_report));
394  inputKbd->notify();
395  break;
396  case 0x30:
397  switch(rx.cmd[0] & 0x0F)
398  {
399  case 0: //Press & release button/hat
400  //test if it is buttons or hat?
401  if((rx.cmd[1] & (1<<7)) == 0)
402  {
403  //buttons, map to corresponding bits in 4 bytes
404  if(rx.cmd[1] <= 7) joystick_report[0] |= (1<<rx.cmd[1]);
405  else if(rx.cmd[1] <= 15) joystick_report[1] |= (1<<(rx.cmd[1]-8));
406  else if(rx.cmd[1] <= 24) joystick_report[2] |= (1<<(rx.cmd[1]-16));
407  else if(rx.cmd[1] <= 31) joystick_report[3] |= (1<<(rx.cmd[1]-24));
408  } else {
409  //hat, remove bit 7 and set to report (don't touch 4 bits of X)
410  joystick_report[4] = (joystick_report[4] & 0xF0) | (rx.cmd[1] & 0x0F);
411  }
412  //send press action
413  inputJoystick->setValue(joystick_report,sizeof(joystick_report));
414  inputJoystick->notify();
415  //release button/hat
416  //test if it is buttons or hat?
417  if((rx.cmd[1] & (1<<7)) == 0)
418  {
419  //buttons, map to corresponding bits in 4 bytes
420  if(rx.cmd[1] <= 7) joystick_report[0] &= ~(1<<rx.cmd[1]);
421  else if(rx.cmd[1] <= 15) joystick_report[1] &= ~(1<<(rx.cmd[1]-8));
422  else if(rx.cmd[1] <= 24) joystick_report[2] &= ~(1<<(rx.cmd[1]-16));
423  else if(rx.cmd[1] <= 31) joystick_report[3] &= ~(1<<(rx.cmd[1]-24));
424  } else {
425  //hat release means always 15.
426  joystick_report[4] = (joystick_report[4] & 0xF0) | 0x0F;
427  }
428  break;
429  case 1: //Press button/hat
430  //test if it is buttons or hat?
431  if((rx.cmd[1] & (1<<7)) == 0)
432  {
433  //buttons, map to corresponding bits in 4 bytes
434  if(rx.cmd[1] <= 7) joystick_report[0] |= (1<<rx.cmd[1]);
435  else if(rx.cmd[1] <= 15) joystick_report[1] |= (1<<(rx.cmd[1]-8));
436  else if(rx.cmd[1] <= 24) joystick_report[2] |= (1<<(rx.cmd[1]-16));
437  else if(rx.cmd[1] <= 31) joystick_report[3] |= (1<<(rx.cmd[1]-24));
438  } else {
439  //hat, remove bit 7 and set to report (don't touch 4 bits of X)
440  joystick_report[4] = (joystick_report[4] & 0xF0) | (rx.cmd[1] & 0x0F);
441  }
442  break;
443  case 2: //Release button/hat
444  //test if it is buttons or hat?
445  if((rx.cmd[1] & (1<<7)) == 0)
446  {
447  //buttons, map to corresponding bits in 4 bytes
448  if(rx.cmd[1] <= 7) joystick_report[0] &= ~(1<<rx.cmd[1]);
449  else if(rx.cmd[1] <= 15) joystick_report[1] &= ~(1<<(rx.cmd[1]-8));
450  else if(rx.cmd[1] <= 24) joystick_report[2] &= ~(1<<(rx.cmd[1]-16));
451  else if(rx.cmd[1] <= 31) joystick_report[3] &= ~(1<<(rx.cmd[1]-24));
452  } else {
453  //hat release means always 15.
454  joystick_report[4] = (joystick_report[4] & 0xF0) | 0x0F;
455  }
456  break;
457  case 4: //X Axis
458  //preserve 4 bits of hat
459  joystick_report[4] = (joystick_report[4] & 0x0F) | ((rx.cmd[1] & 0x0F) << 4);
460  //preserve 2 bits of Y
461  joystick_report[5] = (joystick_report[5] & 0xC0) | ((rx.cmd[1] & 0xF0) >> 4) | ((rx.cmd[2] & 0x03) << 4);
462  break;
463  case 5: //Y Axis
464  //preserve 6 bits of X
465  joystick_report[5] = (joystick_report[5] & 0x3F) | ((rx.cmd[1] & 0x03) << 6);
466  //save remaining Y
467  joystick_report[6] = ((rx.cmd[1] & 0xFC) >> 2) | ((rx.cmd[2] & 0x03) << 6);
468  break;
469  case 6: //Z Axis
470  joystick_report[7] = rx.cmd[1];
471  joystick_report[8] = (joystick_report[8] & 0xFC) | (rx.cmd[2] & 0x03);
472  break;
473  case 7: //Z-rotate
474  //preserve 2 bits of Z-axis
475  joystick_report[8] = (joystick_report[8] & 0x03) | ((rx.cmd[1] & 0x3F) << 2);
476  //preserve slider left & combine 2 bits of LSB & MSB to one nibble
477  joystick_report[9] = (joystick_report[9] & 0xF0) | ((rx.cmd[1] & 0xC0) >> 6) | ((rx.cmd[2] & 0x03) << 2);
478  break;
479  case 8: //slider left
480  //preserve 4 bits of Z-rotate, add low nibble of first byte
481  joystick_report[9] = (joystick_report[9] & 0x0F) | ((rx.cmd[1] & 0x0F) << 4);
482  //preserve 2 bits of slider right, add high nibble of first byte and second byte
483  joystick_report[10] = (joystick_report[10] & 0xC0) | ((rx.cmd[1] & 0xF0) >> 4) | ((rx.cmd[2] & 0x03) << 4);
484  break;
485  case 9: //slider right
486  //preserve 6 bits of slider left, add 2 bits for slider right
487  joystick_report[10] = (joystick_report[10] & 0x3F) | ((rx.cmd[1] & 0x03) << 6);
488  //save remaining slider right
489  joystick_report[11] = ((rx.cmd[1] & 0xFC) >> 2) | ((rx.cmd[2] & 0x03) << 6);
490  break;
491  case 15: //reset joystick (excepting mouse & keyboard)
492  halBLEReset((1<<0)|(1<<2));
493  break;
494  }
495  inputJoystick->setValue(joystick_report,sizeof(joystick_report));
496  inputJoystick->notify();
497  break;
498  }
499  }
500  }
501  } else {
502  ESP_LOGE(LOG_TAG,"ble hid queue not initialized, retry in 1s");
503  vTaskDelay(1000/portTICK_PERIOD_MS);
504  }
505  }

509 class CBs: public BLEServerCallbacks {

512

514

522

531  desc = (BLE2902*) inputJoystick->getDescriptorByUUID(BLEUUID((uint16_t)0x2902));

538  BLE2902* desc;
539 
540  isConnected = 0;
541 
542  //disable notifications for input service, suggested by chegewara to
543  //reduce power & memory usage
544  //https://github.com/asterics/esp32_mouse_keyboard/commit/a1796ce91155ec7db62af4a53dbdef32bc4adf08#commitcomment-28888676
545  if(activateKeyboard)
546  {
547  desc = (BLE2902*) inputKbd->getDescriptorByUUID(BLEUUID((uint16_t)0x2902));
548  desc->setNotifications(false);
549  }
550 
551  if(activateMouse)

556 
557  if(activateJoystick)
558  {
559  desc = (BLE2902*) inputJoystick->getDescriptorByUUID(BLEUUID((uint16_t)0x2902));
560  desc->setNotifications(false);
561  }
562  //stop HID receiving task
563  bletask->stop();
564 
565  //restart advertising
566  BLEAdvertising *pAdvertising = pServer->getAdvertising();
567  pAdvertising->start();
568  ESP_LOGI(LOG_TAG,"Client disconnected, restarting advertising");
569  }
