128 void halAdcProcessStrongMode(adcData_t *D)
129 {
130  //on a FABI device, we cannot do this combination with analog values.
131  //maybe it will be done on another firmware part.
132  #ifdef DEVICE_FABI
133  D->strongmode = STRONG_NORMAL;
134  return;
135  #endif
136 
137  #ifdef DEVICE_FLIPMOUSE
138  if(adcStrongTimerHandle == NULL)
139  {
140  ESP_LOGE(LOG_TAG,"Strong mode timer uninitialized!!!");
141  return;
142  }
143 
144  //timer is not started, and we have a mode != NORMAL
145  if((xTimerIsTimerActive(adcStrongTimerHandle) == pdFALSE) && (D->strongmode != STRONG_NORMAL))
146  {
147  //than we need to start a timer for the timeout moving back to NORMAL
148  xTimerStart(adcStrongTimerHandle,0);
149  //return in this case & wait for next iteration
150  return;
151  }
152 
153  //if we have a mode != NORMAL
154  if(D->strongmode != STRONG_NORMAL)
155  {
156  //check if we have a movement in any direction
157  if(D->x != 0 || D->y != 0)
158  {
159  //if yes, trigger action (depending on SIP/PUFF mode)
160  if(D->strongmode == STRONG_PUFF)
161  {
162  if(abs(D->x) > abs(D->y))
163  {
164  //x has higher values -> use LEFT/RIGHT
165  if(D->x > 0) xEventGroupSetBits(virtualButtonsIn[VB_STRONGPUFF_RIGHT/4],(1<<(VB_STRONGPUFF_RIGHT%4)));
166  else xEventGroupSetBits(virtualButtonsIn[VB_STRONGPUFF_LEFT/4],(1<<(VB_STRONGPUFF_LEFT%4)));
167  } else {
168  //y has higher values -> use UP/DOWN
169  if(D->y > 0) xEventGroupSetBits(virtualButtonsIn[VB_STRONGPUFF_DOWN/4],(1<<(VB_STRONGPUFF_DOWN%4)));
170  else xEventGroupSetBits(virtualButtonsIn[VB_STRONGPUFF_UP/4],(1<<(VB_STRONGPUFF_UP%4)));
171  }
172  }
173  if(D->strongmode == STRONG_SIP)
174  {
175  if(abs(D->x) > abs(D->y))
176  {
177  //x has higher values -> use LEFT/RIGHT
178  if(D->x > 0) xEventGroupSetBits(virtualButtonsIn[VB_STRONGSIP_RIGHT/4],(1<<(VB_STRONGSIP_RIGHT%4)));
179  else xEventGroupSetBits(virtualButtonsIn[VB_STRONGSIP_LEFT/4],(1<<(VB_STRONGSIP_LEFT%4)));
180  } else {
181  //y has higher values -> use UP/DOWN
182  if(D->y > 0) xEventGroupSetBits(virtualButtonsIn[VB_STRONGSIP_DOWN/4],(1<<(VB_STRONGSIP_DOWN%4)));
183  else xEventGroupSetBits(virtualButtonsIn[VB_STRONGSIP_UP/4],(1<<(VB_STRONGSIP_UP%4)));
184  }
185  }
186  //cancel timer
187  xTimerStop(adcStrongTimerHandle,0);
188  xTimerReset(adcStrongTimerHandle,0);
189 
190  //and set back to normal
191  D->strongmode = STRONG_NORMAL;
192  }
193  }
194  #endif
195 }

239 void halAdcReadData(adcData_t *values)
240 {
241  //read all sensors
242  int32_t tmp = 0;
243  int32_t x,y;
244  int32_t up,down,left,right,pressure;
245  up = down = left = right = pressure = 0;
246 
247  //read sensor data & apply characteristics
248  #ifdef HAL_IO_ADC_CHANNEL_UP
249  up = adc1_get_raw(HAL_IO_ADC_CHANNEL_UP);
250  if(up == -1) { ESP_LOGE(LOG_TAG,"Cannot read channel up"); return; }
251  #endif
252  #ifdef HAL_IO_ADC_CHANNEL_DOWN
253  down = adc1_get_raw(HAL_IO_ADC_CHANNEL_DOWN);
254  if(down == -1) { ESP_LOGE(LOG_TAG,"Cannot read channel down"); return; }
255  #endif
256  #ifdef HAL_IO_ADC_CHANNEL_LEFT
257  left = adc1_get_raw(HAL_IO_ADC_CHANNEL_LEFT);
258  if(left == -1) { ESP_LOGE(LOG_TAG,"Cannot read channel left"); return; }
259  #endif
260  #ifdef HAL_IO_ADC_CHANNEL_RIGHT
261  right = adc1_get_raw(HAL_IO_ADC_CHANNEL_RIGHT);
262  if(right == -1) { ESP_LOGE(LOG_TAG,"Cannot read channel right"); return; }
263  #endif
264  #ifdef HAL_IO_ADC_CHANNEL_PRESSURE
265  pressure = adc1_get_raw(HAL_IO_ADC_CHANNEL_PRESSURE);
266  pressure += 52; //todo: just for one sensor -> do it in calibrate...
267  #endif
268  if(pressure == -1)
269  {
270  ESP_LOGE(LOG_TAG,"Cannot read channel pressure"); return;
271  } else {
272  //pressure = esp_adc_cal_raw_to_voltage(pressure,&characteristics);
273  values->pressure = pressure;
274  }
275 
276  //do the mouse rotation
277  switch (adc_conf.orientation) {
278  case 90: tmp=up; up=left; left=down; down=right; right=tmp; break;
279  case 180: tmp=up; up=down; down=tmp; tmp=right; right=left; left=tmp; break;
280  case 270: tmp=up; up=right; right=down; down=left; left=tmp;break;
281  }
282 
283  //apply individual sensor gain
284  values->up = up * adc_conf.gain[0] / 50;
285  values->down = down * adc_conf.gain[1] / 50;
286  values->left = left * adc_conf.gain[2] / 50;
287  values->right = right * adc_conf.gain[3] / 50;
288 
289  //calculate X and Y values
290  x = (values->left - values->right) - offsetx;
291  y = (values->up - values->down) - offsety;
292 
293  //apply elliptic deadzone
294  //formula:
295  //https://sebadorn.de/2012/01/02/herausfinden-ob-ein-punkt-in-einer-ellipse-liegt
296  //A point in an elliptic curve:
297  //(px - cx)² / rx² + (py - cy)² / ry² <= 1
298  //cx/cy is 0 (we start at 0/0)
299  //py/py is our x/y
300  //rx/ry are deadzone values
301  //to shorten the formulas
302  uint8_t a = adc_conf.deadzone_x;
303  uint8_t b = adc_conf.deadzone_y;
304  float status = pow(x,2) / pow(a,2) + pow(y,2) / pow(b,2);
305 
306  //check if point is outside deadzone
307  if(status > 1.0)
308  {
309  //if outside deadzone, subtract ellipse itself to start with
310  //0 for a movement
311 
312  //formula for ellipse point:
313  //https://math.stackexchange.com/questions/22064/calculating-a-point-that-lies-on-an-ellipse-given-an-angle
314  float deadzoneX = 0;
315  float deadzoneY = 0;
316 
317  //angle of the given mouthpiece
318  //calculate only if x&y is != 0
319  if((x < 0 || x > 0) && (y < 0 || y > 0))
320  {
321  float angle = atan(y/x);
322  deadzoneX = abs((a*b)/sqrt(pow(b,2)+pow(a,2)*pow(tan(angle),2)));
323  deadzoneY = abs((a*b)/sqrt(pow(a,2)+ pow(b,2)/pow(tan(angle),2)));
324  }
325 
326  //subtract calculated ellipse coordinates from output X/Y values
327  if(x > 0)
328  {
329  values->x = x - (int)deadzoneX;
330  } else {
331  values->x = x + (int)deadzoneX;
332  }
333  if(y > 0)
334  {
335  values->y = y - (int)deadzoneY;
336  } else {
337  values->y = y + (int)deadzoneY;
338  }
339  } else {
340  //otherwise, x&y is 0
341  values->x = 0;
342  values->y = 0;
343  }
344 }

351 void halAdcProcessPressure(adcData_t *D)
352 {
353  //track changes and fire event only if not issued already.
354  //this is done individually for SIP, PUFF, STRONG_SIP & STRONG_PUFF
355  //0 means nothing is fired, 1 is a fired press action, 2 is a fired release action
356  static uint8_t fired[4] = {0,0,0,0};
357  //currently measured pressure value
358  uint32_t pressurevalue = D->pressure;
359  //currently active general config
360  generalConfig_t *cfg = configGetCurrent();
361  //cannot proceed if no global config is available
362  if(cfg == NULL) return;
363 
364  //if we are in a special mode don't process further (only on FLipMouse)
365  #ifdef DEVICE_FLIPMOUSE
366  if(D->strongmode != STRONG_NORMAL) return;
367  #endif
368 
369  //SIP triggered
370  if(pressurevalue < cfg->adc.threshold_sip && \
371  pressurevalue > cfg->adc.threshold_strongsip)
372  {
373  if(fired[0] != 1)
374  {
375  //create a tone
376  TONE(TONE_SIP_FREQ,TONE_SIP_DURATION);
377  //set/clear VBs
378  CLEARVB_RELEASE(VB_SIP);
379  SETVB_PRESS(VB_SIP);
380  //save fired state
381  fired[0] = 1;
382  }
383  } else {
384  if(fired[0] != 2)
385  {
386  //set/clear VBs
387  CLEARVB_PRESS(VB_SIP);
388  SETVB_RELEASE(VB_SIP);
389  //track fired state
390  fired[0] = 2;
391  }
392  }
393 
394  //STRONGSIP triggered
395  if(pressurevalue < cfg->adc.threshold_strongsip)
396  {
397  //check if strong sip + up/down/left/right is set and
398  //strong sip alone is unused
399  #ifdef DEVICE_FLIPMOUSE
400  if(cfg->virtualButtonCommand[VB_STRONGSIP] == T_NOFUNCTION && \
401  (cfg->virtualButtonCommand[VB_STRONGSIP_UP] != T_NOFUNCTION || \
402  cfg->virtualButtonCommand[VB_STRONGSIP_DOWN] != T_NOFUNCTION || \
403  cfg->virtualButtonCommand[VB_STRONGSIP_LEFT] != T_NOFUNCTION || \
404  cfg->virtualButtonCommand[VB_STRONGSIP_RIGHT] != T_NOFUNCTION))
405  {
406  //if at least one strong action is defined and strong sip
407  //is unused, enter strong sip mode
408  D->strongmode = STRONG_SIP;
409  ESP_LOGI(LOG_TAG,"Enter STRONG SIP");
410  TONE(TONE_STRONGSIP_ENTER_FREQ,TONE_STRONGSIP_ENTER_DURATION);
411  } else {
412  #endif
413  if(fired[1] != 1)
414  {
415  //make a tone
416  TONE(TONE_STRONGSIP_ENTER_FREQ,TONE_STRONGSIP_ENTER_DURATION);
417  //either no strong sip + <yy> action is defined or strong
418  // is used, trigger strong sip VB.
419  CLEARVB_RELEASE(VB_STRONGSIP);
420  SETVB_PRESS(VB_STRONGSIP);
421  //save fired stated
422  fired[1] = 1;
423  }
424  #ifdef DEVICE_FLIPMOUSE
425  }
426  #endif
427  } else {
428  if(fired[1] != 2)
429  {
430  //set/clear VBs
431  CLEARVB_PRESS(VB_STRONGSIP);
432  SETVB_RELEASE(VB_STRONGSIP);
433  //track fired state
434  fired[1] = 2;
435  }
436  }
437 
438  //PUFF triggered
439  if(pressurevalue > cfg->adc.threshold_puff && \
440  pressurevalue < cfg->adc.threshold_strongpuff)
441  {
442 
443  if(fired[2] != 1)
444  {
445  //create a tone
446  TONE(TONE_PUFF_FREQ,TONE_PUFF_DURATION);
447  //set/clear VBs
448  CLEARVB_RELEASE(VB_PUFF);
449  SETVB_PRESS(VB_PUFF);
450  //save fired state
451  fired[2] = 1;
452  }
453  } else {
454  if(fired[2] != 2)
455  {
456  //set/clear VBs
457  CLEARVB_PRESS(VB_PUFF);
458  SETVB_RELEASE(VB_PUFF);
459  //track fired state
460  fired[2] = 2;
461  }
462  }
463 
464  //STRONGPUFF triggered
465  if(pressurevalue > cfg->adc.threshold_strongpuff)
466  {
467  //check if strong puff + up/down/left/right is set and
468  //strong puff alone is unused
469  #ifdef DEVICE_FLIPMOUSE
470  if(cfg->virtualButtonCommand[VB_STRONGPUFF] == T_NOFUNCTION && \
471  (cfg->virtualButtonCommand[VB_STRONGPUFF_UP] != T_NOFUNCTION || \
472  cfg->virtualButtonCommand[VB_STRONGPUFF_DOWN] != T_NOFUNCTION || \
473  cfg->virtualButtonCommand[VB_STRONGPUFF_LEFT] != T_NOFUNCTION || \
474  cfg->virtualButtonCommand[VB_STRONGPUFF_RIGHT] != T_NOFUNCTION))
475  {
476  //if at least one strong action is defined and strong puff
477  //is unused, enter strong puff mode
478  D->strongmode = STRONG_PUFF;
479  ESP_LOGI(LOG_TAG,"Enter STRONG PUFF");
480  TONE(TONE_STRONGPUFF_ENTER_FREQ,TONE_STRONGPUFF_ENTER_DURATION);
481  } else {
482  #endif
483  if(fired[3] != 1)
484  {
485  //make a tone
486  TONE(TONE_STRONGPUFF_ENTER_FREQ,TONE_STRONGPUFF_ENTER_DURATION);
487  //either no strong puff + <yy> action is defined or strong
488  // is used, trigger strong puff VB.
489  CLEARVB_RELEASE(VB_STRONGPUFF);
490  SETVB_PRESS(VB_STRONGPUFF);
491  //save fired state
492  fired[3] = 1;
493  }
494 
495  #ifdef DEVICE_FLIPMOUSE
496  }
497  #endif
498  } else {
499  if(fired[3] != 2)
500  {
501  //set/clear VBs
502  CLEARVB_PRESS(VB_STRONGPUFF);
503  SETVB_RELEASE(VB_STRONGPUFF);
504  //track fired state
505  fired[3] = 2;
506  }
507  }
508 }

525 void halAdcTaskMouse(void * pvParameters)
526 {
527  //analog values
528  adcData_t D;
529  D.strongmode = STRONG_NORMAL;
530  //int32_t x,y;
531  static uint16_t accelTimeX=0,accelTimeY=0;
532  int32_t tempX,tempY;
533  float moveVal, accumXpos = 0, accumYpos = 0;
534  //todo: use a define for the delay here... (currently used: value from vTaskDelay())
535  float accelFactor= 20 / 100000000.0f;
536  usb_command_t command;
537  command.len = 5;
538  command.data[0] = 'M'; //send a mouse report
539  command.data[1] = 0xFF; //for handling in hal_serial: having sticky buttons with 0xFF
540 
541  TickType_t xLastWakeTime;
542  //set adc data reference for timer
543  vTimerSetTimerID(adcStrongTimerHandle,&D);
544 
545  while(1)
546  {
547  //get mutex
548  if(xSemaphoreTake(adcSem, (TickType_t) 30) != pdTRUE)
549  {
550  ESP_LOGW(LOG_TAG,"Cannot obtain mutex for reading");
551  continue;
552  }
553 
554  //read out the analog voltages from all 5 channels (including deadzone)
555  halAdcReadData(&D);
556 
557  //if you want to slow down, uncomment following two lines
558  //ESP_LOGD(LOG_TAG,"X/Y square, X/Y ellipse: %d/%d, %d/%d",tempX,tempY,D.x,D.y);
559  //vTaskDelay(10);
560 
561  //report raw values.
562  halAdcReportRaw(D.up, D.down, D.left, D.right, D.pressure, D.x, D.y);
563 
564  //apply acceleration
565  if (D.x==0) accelTimeX=0;
566  else if (accelTimeX < ACCELTIME_MAX) accelTimeX+=adc_conf.acceleration;
567  if (D.y==0) accelTimeY=0;
568  else if (accelTimeY < ACCELTIME_MAX) accelTimeY+=adc_conf.acceleration;
569 
570  //calculate the current X movement by using acceleration, accel factor and sensitivity
571  moveVal = D.x * adc_conf.sensitivity_x * accelFactor * accelTimeX;
572  //limit value
573  if (moveVal>adc_conf.max_speed) moveVal=adc_conf.max_speed;
574  if (moveVal< -adc_conf.max_speed) moveVal=-adc_conf.max_speed;
575  //add to accumulated movement value
576  accumXpos+=moveVal;
577 
578  //do the same calculations for Y axis
579  moveVal = D.y * adc_conf.sensitivity_y * accelFactor * accelTimeY;
580  if (moveVal>adc_conf.max_speed) moveVal=adc_conf.max_speed;
581  if (moveVal< -adc_conf.max_speed) moveVal=-adc_conf.max_speed;
582  accumYpos+=moveVal;
583 
584  //cast to int again
585  tempX = accumXpos;
586  tempY = accumYpos;
587 
588  //limit to int8 values (to fit into mouse report)
589  if(tempX > 127) tempX = 127;
590  if(tempX < -127) tempX = -127;
591  if(tempY > 127) tempY = 127;
592  if(tempY < -127) tempY = -127;
593 
594  //TODO: wenn D.strongmode != noraml > eigene fkt. mit berechneten werten.
595  //if we are in a special strong mode, do NOT send accumulated data
596  //to USB/BLE. Instead, call halAdcProcessPressure
597  if(D.strongmode == STRONG_NORMAL)
598  {
599  //if at least one value is != 0, send to mouse.
600  if ((tempX != 0) || (tempY != 0))
601  {
602  command.data[2] = tempX;
603  command.data[3] = tempY;
604  accumXpos -= tempX;
605  accumYpos -= tempY;
606 
607  //post values to mouse queue (USB and/or BLE)
608  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_USB)
609  {
610  xQueueSend(hid_usb,&command,0);
611  }
612 
613  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_BLE)
614  {
615  xQueueSend(mouse_movement_ble,&command,0);
616  }
617  }
618  //pressure sensor is handled in another function
619  halAdcProcessPressure(&D);
620  } else {
621  //in special mode, process strong mdoe
622  halAdcProcessStrongMode(&D);
623  }
624 
625  //give mutex
626  xSemaphoreGive(adcSem);
627 
628  //delay the task.
629  vTaskDelayUntil( &xLastWakeTime, 20/portTICK_PERIOD_MS);
630  }
631 }

717 void halAdcCalibrate(void)
718 {
719  //check for initialized mutex
720  if(adcSem == NULL) return;
721 
722  //get mutex
723  if(xSemaphoreTake(adcSem, (TickType_t) 20))
724  {
725  ESP_LOGI(LOG_TAG,"Starting calibration, offsets: %d/%d",offsetx,offsety);
726  uint32_t up = 0;
727  uint32_t down = 0;
728  uint32_t left = 0;
729  uint32_t right = 0;
730  uint32_t pressure = 0;
731  int32_t temp;
732

734  for(uint8_t i = 0; i<8; i++)
735  {
736  #ifdef HAL_IO_ADC_CHANNEL_UP

739  up += temp;
740  #endif
741  #ifdef HAL_IO_ADC_CHANNEL_DOWN

743  if(temp == -1) { ESP_LOGE(LOG_TAG,"Cannot read channel down"); continue; }
744  down += temp;
745  #endif
746  #ifdef HAL_IO_ADC_CHANNEL_LEFT
747  temp = adc1_get_raw(HAL_IO_ADC_CHANNEL_LEFT);
748  if(temp == -1) { ESP_LOGE(LOG_TAG,"Cannot read channel left"); continue; }
749  left += temp;
750  #endif

799 void task_calibration(taskNoParameterConfig_t *param)
800 {
801  EventBits_t uxBits = 0;
802  if(param == NULL)
803  {
804  ESP_LOGE(LOG_TAG,"parameter = NULL, cannot proceed");
805  while(1) vTaskDelay(100000/portTICK_PERIOD_MS);
806  return;
807  }
808

810  uint8_t evGroupIndex = param->virtualButton / 4;
811  //calculate bitmask offset within the EventGroup
812  uint8_t evGroupShift = param->virtualButton % 4;
813  //final pointer to the EventGroup used by this task
814  EventGroupHandle_t *evGroup = NULL;
815  //ticks between task timeouts
816  const TickType_t xTicksToWait = 2000 / portTICK_PERIOD_MS;
817  //local VB
818  uint8_t vb = param->virtualButton;
819 
820  if(vb != VB_SINGLESHOT)
821  {
822  //check for correct offset
823  if(evGroupIndex >= NUMBER_VIRTUALBUTTONS)
824  {
825  ESP_LOGE(LOG_TAG,"virtual button group unsupported: %d ",evGroupIndex);
826  while(1) vTaskDelay(100000/portTICK_PERIOD_MS);
827  return;
828  }
829 
830  //test if event groups are already initialized, otherwise exit immediately
831  while(virtualButtonsOut[evGroupIndex] == 0)
832  {
833  ESP_LOGE("task_calib","uninitialized event group for virtual buttons, retry in 1s");
834  vTaskDelay(1000/portTICK_PERIOD_MS);
835  }
836  //save final event group for later
837  evGroup = virtualButtonsOut[evGroupIndex];
838  }
839 
840  while(1)
841  {
842  //in single shot mode, just trigger calibration & return
843  if(vb == VB_SINGLESHOT)
844  {
845  halAdcCalibrate();
846  return;
847  }
848  //wait for the flag
849  uxBits = xEventGroupWaitBits(evGroup,(1<<evGroupShift),pdTRUE,pdFALSE,xTicksToWait);
850  //test for a valid set flag (else branch would be timeout)
851  if(uxBits & (1<<evGroupShift))
852  {
853  halAdcCalibrate();
854  }
855  }
