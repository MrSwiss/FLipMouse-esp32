138 void halAdcTaskMouse(void * pvParameters)
139 {
140  //analog values
141  uint32_t up, down, left, right, pressure;
142  int32_t x,y;
143  static uint16_t accelTimeX=0,accelTimeY=0;
144  float tempX,tempY,moveVal, accumXpos, accumYpos;
145  //todo: use a define for the delay here... (currently used: value from vTaskDelay())
146  float accelFactor= 20 / 100000000.0f;
147  float max_speed= adc_conf.max_speed / 10.0f;
148  mouse_command_t command;
149 
150  while(1)
151  {
152  //read out the analog voltages from all 5 channels
153  up = adc1_to_voltage(HAL_IO_ADC_CHANNEL_UP, &characteristics);
154  down = adc1_to_voltage(HAL_IO_ADC_CHANNEL_DOWN, &characteristics);
155  left = adc1_to_voltage(HAL_IO_ADC_CHANNEL_LEFT, &characteristics);
156  right = adc1_to_voltage(HAL_IO_ADC_CHANNEL_RIGHT, &characteristics);
157  pressure = adc1_to_voltage(HAL_IO_ADC_CHANNEL_PRESSURE, &characteristics);
158 
159  //TODO: use gain?!?
160  //left = left * adc_conf.gainLeft / 50;
161  //right = right * adc_conf.gainRight / 50;
162  //up = up * adc_conf.gainUp / 50;
163  //down = down * adc_conf.gainDown / 50;
164 
165  //subtract offsets
166  tempX = (float)(left - right) - offsetx;
167  tempY = (float)(up - down) - offsety;
168 
169  //apply deadzone
170  if (tempX<-adc_conf.deadzone_x) tempX+=adc_conf.deadzone_x;
171  else if (tempX>adc_conf.deadzone_x) tempX-=adc_conf.deadzone_x;
172  else tempX=0.0;
173  if (tempY<-adc_conf.deadzone_y) tempY+=adc_conf.deadzone_y;
174  else if (tempY>adc_conf.deadzone_y) tempY-=adc_conf.deadzone_y;
175  else tempY=0.0;
176 
177  //apply acceleration
178  if (tempX==0) accelTimeX=0;
179  else if (accelTimeX < ACCELTIME_MAX) accelTimeX+=adc_conf.acceleration;
180  if (tempY==0) accelTimeY=0;
181  else if (accelTimeY < ACCELTIME_MAX) accelTimeY+=adc_conf.acceleration;
182 
183  //calculate the current X movement by using acceleration, accel factor and sensitivity
184  moveVal = tempX * adc_conf.sensitivity_x * accelFactor * accelTimeX;
185  //limit value
186  if (moveVal>adc_conf.max_speed) moveVal=adc_conf.max_speed;
187  if (moveVal< -adc_conf.max_speed) moveVal=-adc_conf.max_speed;
188  //add to accumulated movement value
189  accumXpos+=moveVal;
190 
191  //do the same calculations for Y axis
192  moveVal = tempY * adc_conf.sensitivity_y * accelFactor * accelTimeY;
193  if (moveVal>adc_conf.max_speed) moveVal=adc_conf.max_speed;
194  if (moveVal< -adc_conf.max_speed) moveVal=-adc_conf.max_speed;
195  accumYpos+=moveVal;
196 
197  //cast to int again
198  int xMove = (int)accumXpos;
199  int yMove = (int)accumYpos;
200  //limit to int8 values (to fit into mouse report)
201  if(xMove > 127) xMove = 127;
202  if(xMove < -127) xMove = -127;
203  if(yMove > 127) yMove = 127;
204  if(yMove < -127) yMove = -127;
205 
206  //if at least one value is != 0, send to mouse.
207  if ((xMove != 0) || (yMove != 0))
208  {
209  command.x = xMove;
210  command.y = yMove;
211  accumXpos -= xMove;
212  accumYpos -= yMove;
213 
214  //post values to mouse queue (USB and/or BLE)
215  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_USB)
216  {
217  ESP_LOGD(LOG_TAG,"Sending mouse X/Y (USB): %d/%d",xMove,yMove);
218  xQueueSend(mouse_movement_usb,&command,0);
219  }
220 
221  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_BLE)
222  {
223  ESP_LOGD(LOG_TAG,"Sending mouse X/Y (BLE): %d/%d",xMove,yMove);
224  xQueueSend(mouse_movement_ble,&command,0);
225  }
226  }
227 
228  //pressure sensor is handled in another function
229  halAdcProcessPressure(pressure);
230 
231  //delay the task.
232  vTaskDelay(20/portTICK_PERIOD_MS);
233  }
234 }

313 void task_calibration(taskNoParameterConfig_t *param)
314 {
315  EventBits_t uxBits;
316  if(param == NULL)
317  {
318  ESP_LOGE(LOG_TAG,"parameter = NULL, cannot proceed");
319  vTaskDelete(NULL);
320  return;
321  }
322 
323  //calculate array index of EventGroup array (each 4 VB have an own EventGroup)
324  uint8_t evGroupIndex = param->virtualButton / 4;
325  //calculate bitmask offset within the EventGroup
326  uint8_t evGroupShift = param->virtualButton % 4;
327  //final pointer to the EventGroup used by this task
328  EventGroupHandle_t *evGroup;
329  //ticks between task timeouts
330  const TickType_t xTicksToWait = 2000 / portTICK_PERIOD_MS;
331  //local VB
332  uint8_t vb = param->virtualButton;
333 
334  if(vb != VB_SINGLESHOT)
335  {
336  //check for correct offset
337  if(evGroupIndex >= NUMBER_VIRTUALBUTTONS)
338  {
339  ESP_LOGE(LOG_TAG,"virtual button group unsupported: %d ",evGroupIndex);
340  vTaskDelete(NULL);
341  return;
342  }
343 
344  //test if event groups are already initialized, otherwise exit immediately
345  while(virtualButtonsOut[evGroupIndex] == 0)
346  {
347  ESP_LOGE("task_calib","uninitialized event group for virtual buttons, retry in 1s");
348  vTaskDelay(1000/portTICK_PERIOD_MS);
349  }
350  //save final event group for later
351  evGroup = virtualButtonsOut[evGroupIndex];
352  }
353 
354  while(1)
355  {
356  //in single shot mode, just trigger calibration & return
357  if(vb == VB_SINGLESHOT)
358  {
359  halAdcCalibrate();
360  return;
361  }
362  //wait for the flag
363  uxBits = xEventGroupWaitBits(evGroup,(1<<evGroupShift),pdTRUE,pdFALSE,xTicksToWait);
364  //test for a valid set flag (else branch would be timeout)
365  if(uxBits & (1<<evGroupShift))
366  {
367  halAdcCalibrate();
368  }
369  }
370 }

382 void halAdcTaskThreshold(void * pvParameters)
383 {
384  //analog values
385  uint32_t up, down, left, right, pressure;
386  int32_t x,y;
387  uint8_t firedx = 0,firedy = 0;
388 
389  while(1)
390  {
391  //read out the analog voltages from all 5 channels
392  up = adc1_to_voltage(HAL_IO_ADC_CHANNEL_UP, &characteristics);
393  down = adc1_to_voltage(HAL_IO_ADC_CHANNEL_DOWN, &characteristics);
394  left = adc1_to_voltage(HAL_IO_ADC_CHANNEL_LEFT, &characteristics);
395  right = adc1_to_voltage(HAL_IO_ADC_CHANNEL_RIGHT, &characteristics);
396  pressure = adc1_to_voltage(HAL_IO_ADC_CHANNEL_PRESSURE, &characteristics);
397 
398  //if a value is outside threshold, activate debounce timer
399  x = (int32_t)(left - right) - offsetx;
400  y = (int32_t)(up - down) - offsety;
401 
402  //LEFT/RIGHT value exceeds threshold (deadzone) value?
403  if(abs(x) > adc_conf.deadzone_x)
404  {
405  //if yes, check if not set already (avoid lot of load)
406  if(firedx == 0)
407  {
408  ESP_LOGD("hal_adc","X-axis fired in alternative mode");
409  //set either left or right bit in debouncer input event group
410  if(x < 0) xEventGroupSetBits(virtualButtonsIn[VB_LEFT/4],(1<<(VB_LEFT%4)));
411  if(x > 0) xEventGroupSetBits(virtualButtonsIn[VB_RIGHT/4],(1<<(VB_RIGHT%4)));
412  //remember that we alread set the flag
413  firedx = 1;
414  }
415  } else {
416  //below threshold, clear the one-time flag setting variable
417  firedx = 0;
418  //also clear the debouncer event bits
419  xEventGroupClearBits(virtualButtonsIn[VB_LEFT/4],(1<<(VB_LEFT%4)));
420  xEventGroupClearBits(virtualButtonsIn[VB_RIGHT/4],(1<<(VB_RIGHT%4)));
421  }
422 
423  //UP/DOWN value exceeds threshold (deadzone) value?
424  if(abs(y) > adc_conf.deadzone_y)
425  {
426  //if yes, check if not set already (avoid lot of load)
427  if(firedy == 0)
428  {
429  ESP_LOGD("hal_adc","Y-axis fired in alternative mode");
430  //set either left or right bit in debouncer input event group
431  if(y < 0) xEventGroupSetBits(virtualButtonsIn[VB_UP/4],(1<<(VB_UP%4)));
432  if(y > 0) xEventGroupSetBits(virtualButtonsIn[VB_DOWN/4],(1<<(VB_DOWN%4)));
433  //remember that we alread set the flag
434  firedy = 1;
435  }
436  } else {
437  //below threshold, clear the one-time flag setting variable
438  firedy = 0;
439  //also clear the debouncer event bits
440  xEventGroupClearBits(virtualButtonsIn[VB_UP/4],(1<<(VB_UP%4)));
441  xEventGroupClearBits(virtualButtonsIn[VB_DOWN/4],(1<<(VB_DOWN%4)));
442  }
443 
444  //pressure sensor is handled in another function
445  halAdcProcessPressure(pressure);
446 
447  //delay the task.
448  vTaskDelay(20/portTICK_PERIOD_MS);
449  }

462 esp_err_t halAdcInit(adc_config_t* params)
463 {
464  esp_err_t ret;
465  //Init ADC and Characteristics
466  ret = adc1_config_width(ADC_WIDTH_BIT_12);
467  if(ret != ESP_OK) { ESP_LOGE("hal_adc","Error setting channel width"); return ret; }
468  ret = adc1_config_channel_atten(HAL_IO_ADC_CHANNEL_UP, ADC_ATTEN_DB_11);
469  if(ret != ESP_OK) { ESP_LOGE("hal_adc","Error setting atten on channel %d",HAL_IO_ADC_CHANNEL_UP); return ret; }
470  ret = adc1_config_channel_atten(HAL_IO_ADC_CHANNEL_DOWN, ADC_ATTEN_DB_11);
471  if(ret != ESP_OK) { ESP_LOGE("hal_adc","Error setting atten on channel %d",HAL_IO_ADC_CHANNEL_DOWN); return ret; }
472  ret = adc1_config_channel_atten(HAL_IO_ADC_CHANNEL_LEFT, ADC_ATTEN_DB_11);
473  if(ret != ESP_OK) { ESP_LOGE("hal_adc","Error setting atten on channel %d",HAL_IO_ADC_CHANNEL_LEFT); return ret; }
474  ret = adc1_config_channel_atten(HAL_IO_ADC_CHANNEL_RIGHT, ADC_ATTEN_DB_11);
475  if(ret != ESP_OK) { ESP_LOGE("hal_adc","Error setting atten on channel %d",HAL_IO_ADC_CHANNEL_RIGHT); return ret; }
476  ret = adc1_config_channel_atten(HAL_IO_ADC_CHANNEL_PRESSURE, ADC_ATTEN_DB_11);
477  if(ret != ESP_OK) { ESP_LOGE("hal_adc","Error setting atten on channel %d",HAL_IO_ADC_CHANNEL_PRESSURE); return ret; }
478  esp_adc_cal_get_characteristics(ADC_CAL_IDEAL_V_REF, ADC_ATTEN_DB_11, ADC_WIDTH_BIT_12, &characteristics);
479 
480  //check if every queue is already initialized
481  for(uint8_t i = 0; i<NUMBER_VIRTUALBUTTONS; i++)
482  {
483  if(virtualButtonsIn[i] == 0)
484  {
485  ESP_LOGE("hal_adc","virtualButtonsIn uninitialized, exiting");
486  return ESP_FAIL;
487  }
488  if(virtualButtonsOut[i] == 0)
489  {
490  ESP_LOGE("hal_adc","virtualButtonsOut uninitialized, exiting");
491  return ESP_FAIL;
492  }
493  }
494  if(joystick_movement_usb == NULL || joystick_movement_ble == NULL || \
495  mouse_movement_ble == NULL || mouse_movement_usb == NULL)
496  {
497  ESP_LOGE("hal_adc","queue uninitialized, exiting");
498  return ESP_FAIL;
499  }
500 
501  //not initializing full config, only ADC
502  if(params == NULL) return ESP_OK;
503 
504  //init remaining parts by updating config.
505  return halAdcUpdateConfig(params);
506 }
