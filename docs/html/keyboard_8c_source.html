1838  }
1839  if (cpoint <= 0xA0)
1840  {
1841  ESP_LOGE(TAG,"unkown cpoint");
1842  return 0;
1843  }
1844  if (cpoint < 0x100)

1909 uint8_t parse_for_keycode(uint8_t inputdata, uint8_t locale, uint8_t *keycode_modifier, uint8_t *deadkey_first_keycode)
1910 {
1911  static uint8_t utf8_state = 0;
1912  static uint16_t unicode_wchar = 0;
1913  static uint16_t keycode = 0;
1914  static uint8_t deadkey = 0;
1915 
1916  //avoid accessing arrays out of bound
1917  if(locale >= LAYOUT_MAX) return 0;
1918 
1919  if (inputdata < 0x80) {
1920  // single byte encoded, 0x00 to 0x7F
1921  utf8_state = 0;
1922  keycode = unicode_to_keycode(inputdata, locale);
1923  deadkey = deadkey_to_keycode(keycode,locale);
1924  *keycode_modifier = keycode_to_modifier(keycode,locale);
1925  if(deadkey) *deadkey_first_keycode = keycode_to_key(deadkey);
1926  else *deadkey_first_keycode = 0;
1927  return keycode_to_key(keycode & keycodes_masks[locale][3]);
1928  } else if (inputdata < 0xC0) {
1929  // 2nd, 3rd or 4th byte, 0x80 to 0xBF
1930  inputdata &= 0x3F;
1931  if (utf8_state == 1) {
1932  utf8_state = 0;
1933  keycode = unicode_to_keycode(unicode_wchar | inputdata, locale);
1934  deadkey = deadkey_to_keycode(keycode,locale);
1935  *keycode_modifier = keycode_to_modifier(keycode,locale);
1936  if(deadkey) *deadkey_first_keycode = keycode_to_key(deadkey);
1937  else *deadkey_first_keycode = 0;
1938  return keycode_to_key(keycode & keycodes_masks[locale][3]);
1939  } else if (utf8_state == 2) {
1940  unicode_wchar |= ((uint16_t)inputdata << 6);
1941  utf8_state = 1;
1942  }
1943  } else if (inputdata < 0xE0) {
1944  // begin 2 byte sequence, 0xC2 to 0xDF
1945  // or illegal 2 byte sequence, 0xC0 to 0xC1
1946  unicode_wchar = (uint16_t)(inputdata & 0x1F) << 6;
1947  utf8_state = 1;
1948  } else if (inputdata < 0xF0) {
1949  // begin 3 byte sequence, 0xE0 to 0xEF
1950  unicode_wchar = (uint16_t)(inputdata & 0x0F) << 12;
1951  utf8_state = 2;
1952  } else {
1953  // begin 4 byte sequence (not supported), 0xF0 to 0xF4
1954  // or illegal, 0xF5 to 0xFF
1955  utf8_state = 255;
1956  }
1957  return 0;
1958 }
