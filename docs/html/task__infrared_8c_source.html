51 void task_infrared(taskInfraredConfig_t *param)
52 {
53  //check for param struct
54  if(param == NULL)
55  {
56  ESP_LOGE(LOG_TAG,"param is NULL ");
57  while(1) vTaskDelay(100000/portTICK_PERIOD_MS);
58  return;
59  }
60  //event bits used for pending on debounced buttons
61  EventBits_t uxBits = 0;
62  //calculate array index of EventGroup array (each 4 VB have an own EventGroup)
63  uint8_t evGroupIndex = param->virtualButton / 4;
64  //calculate bitmask offset within the EventGroup
65  uint8_t evGroupShift = param->virtualButton % 4;
66  //save virtualbutton
67  uint vb = param->virtualButton;
68  //final pointer to the EventGroup used by this task
69  EventGroupHandle_t *evGroup = NULL;
70  //local copy of IR command name
71  char cmdName[SLOTNAME_LENGTH];
72  strcpy(cmdName,param->cmdName);
73 
74  //do all the eventgroup checking only if this is a persistent task
75  //-> virtualButton is NOT set to VB_SINGLESHOT
76  if(param->virtualButton != VB_SINGLESHOT)
77  {
78  //check for correct offset
79  if(evGroupIndex >= NUMBER_VIRTUALBUTTONS)
80  {
81  ESP_LOGE(LOG_TAG,"virtual button group unsupported: %d ",evGroupIndex);
82  while(1) vTaskDelay(100000/portTICK_PERIOD_MS);
83  return;
84  }
85 
86  //test if event groups are already initialized, otherwise exit immediately
87  while(virtualButtonsOut[evGroupIndex] == 0)
88  {
89  ESP_LOGE(LOG_TAG,"uninitialized event group, retry in 1s");
90  vTaskDelay(1000/portTICK_PERIOD_MS);
91  }
92  //event group intialized, store for later usage
93  evGroup = virtualButtonsOut[evGroupIndex];
94  }
95 
96  while(1)
97  {
98  //wait for the flag
99  if(vb != VB_SINGLESHOT)
100  {
101  uxBits = xEventGroupWaitBits(evGroup,(1<<evGroupShift)|(1<<(evGroupShift+4)),pdTRUE,pdFALSE,1000);
102  }
103  //test for a valid set flag or trigger if we are using singleshot
104  if((uxBits & (1<<evGroupShift)) || vb == VB_SINGLESHOT)
105  {
106  //local IR struct
107  halIOIR_t *cfg = NULL;
108  //transaction ID for IR data
109  uint32_t tid;
110  if(halStorageStartTransaction(&tid,20) == ESP_OK)
111  {
112  if(halStorageLoadIR(cmdName,cfg,tid) == ESP_OK)
113  {
114  //send pointer to IR send queue
115  if(cfg != NULL)
116  {
117  SENDIRSTRUCT(cfg);
118  } else {
119  ESP_LOGE(LOG_TAG,"IR cfg is NULL!");
120  }
121  //create tone
122  TONE(TONE_IR_SEND_FREQ,TONE_IR_SEND_DURATION);
123  } else {
124  ESP_LOGE(LOG_TAG,"Error loading IR cmd");
125  }
126  } else {
127  ESP_LOGE(LOG_TAG,"Error starting transaction for IR cmd");
128  }
129  }
130 
131  //function tasks in single shot mode MUST return to its caller
132  if(vb == VB_SINGLESHOT) return;
133  }
134 }
