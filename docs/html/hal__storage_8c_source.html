99 esp_err_t halStorageGetFree(uint32_t *total, uint32_t *free)
100 {

103 
104  //thanks to Ivan Grokhotkov, according to:
105  //https://github.com/espressif/esp-idf/issues/1660
106  /* Get volume information and free clusters of drive 0 */
107  int res = f_getfree("0:", &fre_clust, &fs);
108  if(res == 0)
109  {
110  /* Get total sectors and free sectors */
111  tot_sect = (fs->n_fatent - 2) * fs->csize;
112  fre_sect = fre_clust * fs->csize;
113  *free = fre_sect*512;
114  *total = tot_sect*512;
115  return ESP_OK;

117  return ESP_FAIL;
118  }
