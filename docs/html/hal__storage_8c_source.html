21  * This file contains the abstraction layer for storing FLipMouse/FABI
22  * slots to a storage system. In case of the ESP32 we use FAT
23  * with wear leveling.
24  * PLEASE ADJUST THE MAKEFILE TO USE 512Byte SECTORS AND MODE SAFETY!
25  * We don't do any safe copying, just building a checksum to detect
26  * faulty slots.
27  *
28  * This module can be used to store a config struct (generalConfig_t)
29  * and the corresponding virtual button config structs to the storage.
30  *
31  * Following commands are implemented here:
32  * -) list all slot names
33  * -) load default slot
34  * -) load next or previous slot
35  * -) load slot by name or number
36  * -) store a given slot
37  * -) delete one slot
38  * -) delete all slots
39  *
40  * This module starts one task on its own for maintaining the storage
41  * access.
42  * All methods called from this module will block via a semaphore until
43  * the requested operation is finished.
44  *
45  * Slots are stored in following naming convention (8.3 rule applies here):
46  * general slot config:
47  * xxx.fms (slot number, e.g., 001.fms for slot 1)
48  * virtual button config for slot xxx
49  * xxx_yyy.fms (slot & VB number, e.g., 001_000.fms for VB0 on slot 1)
50  *
51  * Maximum number of slots: 250! (e.g. 250.fms)
52  *
53  * @see generalConfig_t
54  */

344

347  //check if this file is available
348  //TODO: should we re-arrange slots if one is deleted or should we

358  fclose(f);
359  currentSlot++;
360  }
361  } while(1);
362 
363  return ESP_OK;

519 esp_err_t halStorageLoad(hal_storage_load_action navigate, generalConfig_t *cfg, uint32_t tid)
520 {
521  uint8_t slotCount = 0;
522 
523  //fetch current slot count
524  if(halStorageGetNumberOfSlots(tid, &slotCount) != ESP_OK)
525  {
526  ESP_LOGE(LOG_TAG,"cannot get number of available slots");
527  return ESP_FAIL;
528  }
529 
530  //check what we should load
531  switch(navigate)
532  {
533  case NEXT:
534  //check if we are at the last slot, if yes load first, increment otherwise
535  if(slotCount == storageCurrentSlotNumber) storageCurrentSlotNumber = 1;
536  else storageCurrentSlotNumber++;
537  return halStorageLoadNumber(storageCurrentSlotNumber,cfg,tid);
538  break;
539  case PREV:
540  //check if we are at the first slot, if yes load last, decrement otherwise
541  if(storageCurrentSlotNumber == 1) storageCurrentSlotNumber = slotCount;
542  else storageCurrentSlotNumber--;
543  return halStorageLoadNumber(storageCurrentSlotNumber,cfg,tid);
544  break;
545  case DEFAULT:
546  //load default slot via number 0
547  return halStorageLoadNumber(0,cfg,tid);
548  break;
549  case RESTOREFACTORYSETTINGS:
550  break;
551  default:
552  ESP_LOGE(LOG_TAG,"unknown navigate action in halStorageLoad!");
553  break;
554  }
555  return ESP_OK;
556 }

583 esp_err_t halStorageLoadNumber(uint8_t slotnumber, generalConfig_t *cfg, uint32_t tid)
584 {
585  char file[sizeof(base_path)+32];
586  uint32_t slotnamelen;
587  unsigned char md5sumfile[16], md5sumstruct[16];
588  char *slotname = (char *)malloc(SLOTNAME_LENGTH+1);
589 
590  if(halStorageChecks(tid) != ESP_OK) return ESP_FAIL;
591 
592  if(slotnumber > 250)
593  {
594  ESP_LOGE(LOG_TAG,"Slotnumber too high: %d, maximum 250",slotnumber);
595  return ESP_FAIL;
596  }
597 
598  //file naming convention for general config: xxx.fms
599  //each general config file contains:
600  //Name string (\0 terminated)
601  //16Bytes of MD5 checksum (calculated over general config, no slot name)
602  //XXXBytes of generalConfig_t struct
603 
604  //create filename from slotnumber
605  sprintf(file,"%s/%03d.fms",base_path,slotnumber);
606 
607  //open file for reading
608  ESP_LOGD(LOG_TAG,"Opening file %s",file);
609  FILE *f = fopen(file, "rb");
610  if(f == NULL)
611  {
612  //special case: requesting a default config which is not created on a fresh device
613  if(slotnumber == 0)
614  {
615  ESP_LOGW(LOG_TAG,"no default config. creating one & retry");
616  free(slotname);
617  halStorageCreateDefault(tid);
618  return ESP_FAIL;
619  } else {
620  ESP_LOGE(LOG_TAG,"cannot load requested slot number %u",slotnumber);
621  free(slotname);
622  return ESP_FAIL;
623  }
624  }
625 
626  fseek(f,0,SEEK_SET);
627 
628  //skip slot name
629  fread(&slotnamelen,sizeof(uint32_t),1,f);
630  if(slotnamelen > SLOTNAME_LENGTH + 1)
631  {
632  ESP_LOGE(LOG_TAG,"Slotname length too high: %u",slotnamelen);
633  free(slotname);
634  fclose(f);
635  return ESP_FAIL;
636  }
637  fread(slotname,sizeof(char),slotnamelen+1,f);
638  slotname[slotnamelen] = '\0';
639  ESP_LOGD(LOG_TAG,"Read slotname: %s, length %d",slotname,slotnamelen);
640 
641  //get MD5sum of saved slot
642  if(fread(md5sumfile,sizeof(md5sumfile),1,f) != 1)
643  {
644  //did not get a full checksum, maybe EOF.
645  ESP_LOGE(LOG_TAG,"cannot read MD5 sum");
646  fclose(f);
647  free(slotname);
648  return ESP_FAIL;
649  } else {
650  ESP_LOGD(LOG_TAG,"MD5 hash:");
651  ESP_LOG_BUFFER_HEXDUMP(LOG_TAG,md5sumfile,sizeof(md5sumfile),ESP_LOG_DEBUG);
652  }
653 
654  //read remaining general config
655  if(fread(cfg,sizeof(generalConfig_t),1,f) != 1)
656  {
657  //did not get a full config, maybe EOF.
658  ESP_LOGE(LOG_TAG,"cannot read general config, feof: %d",feof(f));
659  ESP_LOG_BUFFER_HEXDUMP(LOG_TAG,cfg,sizeof(generalConfig_t),ESP_LOG_DEBUG);
660  fclose(f);
661  free(slotname);
662  return ESP_FAIL;
663  }
664 
665  //TODO: check slot storage version, upgrade on any difference
666 
667  ESP_LOGW(LOG_TAG,"Loaded slot %s,nr: %d",slotname,slotnumber);
668  ESP_LOG_BUFFER_HEXDUMP(LOG_TAG,cfg,sizeof(generalConfig_t),ESP_LOG_DEBUG);
669 
670  //compare checksums
671  mbedtls_md5((unsigned char *)cfg, sizeof(generalConfig_t), md5sumstruct);
672  if(memcmp(md5sumfile, md5sumstruct, sizeof(md5sumfile)) != 0)
673  {
674  //error comparing md5 sum...
675  ESP_LOGE(LOG_TAG,"MD5 checksum mismatch, maybe slot is corrupted:");
676  ESP_LOG_BUFFER_HEXDUMP(LOG_TAG,md5sumfile,sizeof(md5sumfile),ESP_LOG_DEBUG);
677  ESP_LOG_BUFFER_HEXDUMP(LOG_TAG,md5sumstruct,sizeof(md5sumstruct),ESP_LOG_DEBUG);
678  fclose(f);
679  free(slotname);
680  return ESP_FAIL;
681  }
682 
683  ESP_LOGD(LOG_TAG,"Successfully read slotnumber %u with %u bytes payload", slotnumber, sizeof(generalConfig_t));
684 
685  //clean up
686  free(slotname);
687  fclose(f);
688 
689  //save current slot number to access the VB configs
690  storageCurrentSlotNumber = slotnumber;
691 
692  return ESP_OK;
693 }
