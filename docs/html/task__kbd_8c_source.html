21  * This file contains the task implementation for keyboard key triggering.
22  * This task is started by the configuration task and it is assigned
23  * to one virtual button. As soon as this button is triggered, the configured
24  * keys are sent to either USB, BLE or both.
25  * This task can be deleted all the time to change the configuration of
26  * one button.

35 void sendToQueue(QueueHandle_t queue, uint16_t *data)
36 {
37  size_t offset = 0;
38  //loop until either \0 termination or length of parameter is reached
39  while(data[offset] != 0 && offset < TASK_KEYBOARD_PARAMETERLENGTH)
40  {
41  xQueueSend(queue,&data[offset],TIMEOUT);
42  offset++;
43  }
44 }

46 void task_keyboard(taskKeyboardConfig_t *param)
47 {
48  //check for param struct
49  if(param == NULL)
50  {
51  ESP_LOGE(LOG_TAG,"param is NULL ");
52  vTaskDelete(NULL);
53  return;
54  }
55  //event bits used for pending on debounced buttons
56  EventBits_t uxBits = 0;
57  //store for later use
58  uint8_t keyboardType = param->type;
59  //calculate array index of EventGroup array (each 4 VB have an own EventGroup)
60  uint8_t evGroupIndex = param->virtualButton / 4;
61  //calculate bitmask offset within the EventGroup
62  uint8_t evGroupShift = param->virtualButton % 4;
63  //final pointer to the EventGroup used by this task
64  EventGroupHandle_t *evGroup = NULL;
65  //ticks between task timeouts
66  const TickType_t xTicksToWait = 2000 / portTICK_PERIOD_MS;
67  //local keystring
68  uint16_t keys[TASK_KEYBOARD_PARAMETERLENGTH];
69  memcpy(keys,param->keycodes_text,sizeof(keys[TASK_KEYBOARD_PARAMETERLENGTH]));
70  //local virtual button
71  uint8_t vb = param->virtualButton;
72 
73  if(vb != VB_SINGLESHOT)
74  {
75  //check for correct offset
76  if(evGroupIndex >= NUMBER_VIRTUALBUTTONS)
77  {
78  ESP_LOGE(LOG_TAG,"virtual button group unsupported: %d ",evGroupIndex);
79  vTaskDelete(NULL);
80  return;
81  }
82 
83  //test if event groups are already initialized, otherwise exit immediately
84  while(virtualButtonsOut[evGroupIndex] == 0)
85  {
86  ESP_LOGE(LOG_TAG,"uninitialized event group for virtual buttons, retry in 1s");
87  vTaskDelay(1000/portTICK_PERIOD_MS);
88  }
89  //event group initialized, store for later usage
90  evGroup = virtualButtonsOut[evGroupIndex];
91  }
92 
93  if(keyboardType != PRESS && keyboardType != RELEASE && keyboardType != PRESS_RELEASE &&
94  keyboardType != PRESS_RELEASE_BUTTON && keyboardType != WRITE)
95  {
96  ESP_LOGE(LOG_TAG,"unknown keyboard action type,quit...");
97  vTaskDelete(NULL);
98  return;
99  }
100 
101  while(1)
102  {
103  switch(keyboardType)
104  {
105  //these action are only triggered by a button press
106  case PRESS:
107  case RELEASE:
108  case PRESS_RELEASE:
109  case WRITE:
110  //wait for the flag (only if not in singleshot mode)
111  if(vb != VB_SINGLESHOT)
112  {
113  uxBits = xEventGroupWaitBits(evGroup,(1<<evGroupShift),pdTRUE,pdFALSE,xTicksToWait);
114  }
115  //test for a valid set flag (else branch would be timeout)
116  if((uxBits & (1<<evGroupShift)) || vb == VB_SINGLESHOT)
117  {
118  if(keyboardType == PRESS)
119  {
120  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_USB)
121  sendToQueue(keyboard_usb_press,keys);
122  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_BLE)
123  sendToQueue(keyboard_ble_press,keys);
124  }
125  if(keyboardType == RELEASE)
126  {
127  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_USB)
128  sendToQueue(keyboard_usb_release,keys);
129  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_BLE)
130  sendToQueue(keyboard_ble_release,keys);
131  }
132  if(keyboardType == PRESS_RELEASE || keyboardType == WRITE)
133  {
134  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_USB)
135  {
136  sendToQueue(keyboard_usb_press,keys);
137  vTaskDelay(2);
138  sendToQueue(keyboard_usb_release,keys);
139  }
140  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_BLE)
141  {
142  sendToQueue(keyboard_ble_press,keys);
143  vTaskDelay(2);
144  sendToQueue(keyboard_ble_release,keys);
145  }
146  }
147  }
148  break;
149 
150  //this action is triggered on a button press AND a release
151  case PRESS_RELEASE_BUTTON:
152  //wait for the flags (press & release)
153  if(vb != VB_SINGLESHOT)
154  {
155  uxBits = xEventGroupWaitBits(evGroup,(1<<evGroupShift)| \
156  (1<<(evGroupShift+4)),pdTRUE,pdFALSE,xTicksToWait);
157  }
158  //test for a valid set flag
159  if((uxBits & (1<<evGroupShift)) || vb == VB_SINGLESHOT)
160  {
161  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_USB)
162  sendToQueue(keyboard_usb_press,keys);
163  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_BLE)
164  sendToQueue(keyboard_ble_press,keys);
165  }
166  if((uxBits & (1<<(evGroupShift+4))) || vb == VB_SINGLESHOT)
167  {
168  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_USB)
169  sendToQueue(keyboard_usb_release,keys);
170  if(xEventGroupGetBits(connectionRoutingStatus) & DATATO_BLE)
171  sendToQueue(keyboard_ble_release,keys);
172  }
173 
174  break;
175 
176 
177  break;
178 
179  default:
180  ESP_LOGE(LOG_TAG,"unknown keyboard action type,quit...");
181  vTaskDelete(NULL);
182  return;
183  }
184 
185  //function tasks in singleshot must return
186  if(vb == VB_SINGLESHOT) return;
187  }
188 }
