77 void task_keyboard(taskKeyboardConfig_t *param)
78 {
79  //check for param struct
80  if(param == NULL)
81  {
82  ESP_LOGE(LOG_TAG,"param is NULL ");
83  vTaskDelete(NULL);
84  return;
85  }
86  //event bits used for pending on debounced buttons
87  EventBits_t uxBits = 0;
88  //store for later use
89  uint8_t keyboardType = param->type;
90  //calculate array index of EventGroup array (each 4 VB have an own EventGroup)
91  uint8_t evGroupIndex = param->virtualButton / 4;
92  //calculate bitmask offset within the EventGroup
93  uint8_t evGroupShift = param->virtualButton % 4;
94  //final pointer to the EventGroup used by this task
95  EventGroupHandle_t *evGroup = NULL;
96  //ticks between task timeouts
97  const TickType_t xTicksToWait = 2000 / portTICK_PERIOD_MS;
98  //local virtual button
99  uint8_t vb = param->virtualButton;
100  //local key array offset for each trigger
101  uint8_t keycodeoffset = 0;
102 
103  //if we are in singleshot mode, we do not need to test
104  //the virtual button groups (unused)
105  if(vb != VB_SINGLESHOT)
106  {
107  //check for correct offset
108  if(evGroupIndex >= NUMBER_VIRTUALBUTTONS)
109  {
110  ESP_LOGE(LOG_TAG,"virtual button group unsupported: %d ",evGroupIndex);
111  vTaskDelete(NULL);
112  return;
113  }
114 
115  //test if event groups are already initialized, otherwise exit immediately
116  while(virtualButtonsOut[evGroupIndex] == 0)
117  {
118  ESP_LOGE(LOG_TAG,"uninitialized event group for virtual buttons, retry in 1s");
119  vTaskDelay(1000/portTICK_PERIOD_MS);
120  }
121  //event group initialized, store for later usage
122  evGroup = virtualButtonsOut[evGroupIndex];
123  }
124 
125  //test if parameter is correct
126  if(keyboardType != PRESS && keyboardType != RELEASE && keyboardType != PRESS_RELEASE &&
127  keyboardType != PRESS_RELEASE_BUTTON && keyboardType != WRITE)
128  {
129  ESP_LOGE(LOG_TAG,"unknown keyboard action type,quit...");
130  vTaskDelete(NULL);
131  return;
132  }
133 
134  //local keystring (size is determined by input keystring)
135  uint8_t keylength = 0;
136  while(param->keycodes_text[keylength] != 0 && keylength < TASK_KEYBOARD_PARAMETERLENGTH) keylength++;
137 
138  //if count of keycode is 0, nothing to do here...
139  if(keylength == 0)
140  {
141  ESP_LOGI(LOG_TAG,"Empty kbd instance, quit");
142  if(vb == VB_SINGLESHOT) return; else vTaskDelete(NULL);
143  }
144 
145  //copy keys to local buffer
146  //after this copying, task parameters could be freed (not recommended although)
147  uint16_t *keys = malloc(sizeof(uint16_t)*keylength);
148  memcpy(keys,param->keycodes_text,sizeof(uint16_t)*keylength);
149  if(keys != NULL)
150  {
151  ESP_LOGI(LOG_TAG,"allocated %d keycodes",keylength);
152  } else {
153  ESP_LOGE(LOG_TAG,"cannot allocate %d bytes for keyarray",keylength);
154  if(vb == VB_SINGLESHOT) return; else vTaskDelete(NULL);
155  }
156 
157  while(1)
158  {
159  switch(keyboardType)
160  {
161  //these action are only triggered by a button press
162  case PRESS:
163  case RELEASE:
164  case PRESS_RELEASE:
165  case WRITE:
166  //wait for the flag (only if not in singleshot mode)
167  if(vb != VB_SINGLESHOT)
168  {
169  uxBits = xEventGroupWaitBits(evGroup,(1<<evGroupShift),pdTRUE,pdFALSE,xTicksToWait);
170  }
171  //reset offset
172  keycodeoffset = 0;
173 
174  while(keycodeoffset < keylength)
175  {
176  //test for a valid set flag (else branch would be timeout)
177  if((uxBits & (1<<evGroupShift)) || vb == VB_SINGLESHOT)
178  {
179  if(keyboardType == PRESS)
180  {
181  //send press action to USB/BLE
182  ESP_LOGD(LOG_TAG,"press: 0x%x",keys[keycodeoffset]);
183  sendKbd(keyboard_usb_press,keyboard_ble_press,&keys[keycodeoffset]);
184 
185  }
186  if(keyboardType == RELEASE)
187  {
188  //send release action to USB/BLE
189  ESP_LOGD(LOG_TAG,"release: 0x%x",keys[keycodeoffset]);
190  sendKbd(keyboard_usb_release,keyboard_ble_release,&keys[keycodeoffset]);
191  }
192  if(keyboardType == PRESS_RELEASE || keyboardType == WRITE)
193  {
194  ESP_LOGD(LOG_TAG,"press&release 1: 0x%x",keys[keycodeoffset]);
195  sendKbd(keyboard_usb_press,keyboard_ble_press,&keys[keycodeoffset]);
196  ESP_LOGD(LOG_TAG,"press&release 2: 0x%x",keys[keycodeoffset]);
197  sendKbd(keyboard_usb_release,keyboard_ble_release,&keys[keycodeoffset]);
198  }
199  }
200  keycodeoffset++;
201  }
202  break;
203 
204  //this action is triggered on a button press AND a release
205  case PRESS_RELEASE_BUTTON:
206  //wait for the flags (press & release)
207  if(vb != VB_SINGLESHOT)
208  {
209  uxBits = xEventGroupWaitBits(evGroup,(1<<evGroupShift)| \
210  (1<<(evGroupShift+4)),pdTRUE,pdFALSE,xTicksToWait);
211  }
212  //reset key array offset
213  keycodeoffset = 0;
214  while(keycodeoffset < keylength)
215  {
216  //test for a valid set flag
217  if((uxBits & (1<<evGroupShift)) || vb == VB_SINGLESHOT)
218  {
219  ESP_LOGD(LOG_TAG,"press&release button 1: 0x%x",keys[keycodeoffset]);
220  sendKbd(keyboard_usb_press,keyboard_ble_press,&keys[keycodeoffset]);
221  }
222  if((uxBits & (1<<(evGroupShift+4))) || vb == VB_SINGLESHOT)
223  {
224  ESP_LOGD(LOG_TAG,"press&release button 2: 0x%x",keys[keycodeoffset]);
225  sendKbd(keyboard_usb_release,keyboard_ble_release,&keys[keycodeoffset]);
226  }
227  keycodeoffset++;
228  }
229  break;
230 
231  default:
232  ESP_LOGE(LOG_TAG,"unknown keyboard action type,quit...");
233  vTaskDelete(NULL);
234  free(keys);
235  return;
236  }
237 
238  //function tasks in singleshot must return
239  if(vb == VB_SINGLESHOT)
240  {
241  free(keys);
242  return;
243  }
244  }
245 }
