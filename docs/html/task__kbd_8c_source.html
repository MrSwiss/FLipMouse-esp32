55 esp_err_t task_keyboard_getAT(char* output, void* cfg)
56 {
57  if(cfg == NULL) return ESP_FAIL;
58  taskKeyboardConfig_t *conf = (taskKeyboardConfig_t*)cfg;
59  uint8_t paramsoffset = 0;
60  char identifierarr[30];
61  sprintf(output,"AT ");
62 
63  //determine the AT command
64  switch(conf->type)
65  {
66  case PRESS:
67  strcat(output,"KH ");
68  break;
69  case RELEASE:
70  strcat(output,"KR ");
71  break;
72  case PRESS_RELEASE_BUTTON:
73  strcat(output,"KP ");
74  break;
75  case WRITE:
76  strcat(output,"KW ");
77  break;
78  default: return ESP_FAIL;
79  }
80 
81  if(conf->type != WRITE)
82  {
83  //add the parameter for KP/KR/KH
84  while(conf->keycodes_text[paramsoffset] != 0 && paramsoffset < TASK_KEYBOARD_PARAMETERLENGTH)
85  {
86  //0xE0, E2, E4 for modifiers, 0xF0 for keycodes
87  //append key identifiers
88 
89  //this is a normal keycode
90  if((conf->keycodes_text[paramsoffset] & 0x00FF) != 0)
91  {
92  if(parseKeycodeToIdentifier((conf->keycodes_text[paramsoffset] & 0x00FF) | 0xF000,identifierarr,30) == 1)
93  {
94  strcat(output, identifierarr);
95  strcat(output, " ");
96  } else {
97  ESP_LOGW(LOG_TAG,"Cannot find keycode identifier for 0x%04X @ %d",conf->keycodes_text[paramsoffset],paramsoffset);
98  }
99  }
100  //this is a modifier
101  if((conf->keycodes_text[paramsoffset] & 0xFF00) != 0)
102  {
103  if(parseKeycodeToIdentifier(((conf->keycodes_text[paramsoffset] & 0xFF00) >> 8) | 0xE000,identifierarr,30) == 1)
104  {
105  strcat(output, identifierarr);
106  strcat(output, " ");
107  } else if (parseKeycodeToIdentifier(((conf->keycodes_text[paramsoffset] & 0xFF00) >> 8) | 0xE200,identifierarr,30) == 1) {
108  strcat(output, identifierarr);
109  strcat(output, " ");
110  } else if (parseKeycodeToIdentifier(((conf->keycodes_text[paramsoffset] & 0xFF00) >> 8) | 0xE400,identifierarr,30) == 1) {
111  strcat(output, identifierarr);
112  strcat(output, " ");
113  } else {
114  ESP_LOGW(LOG_TAG,"Cannot find keycode identifier for 0x%04X @ %d",conf->keycodes_text[paramsoffset],paramsoffset);
115  }
116  }
117 
118  paramsoffset++;
119  }
120  } else {
121  //for write we get the original bytes from back to front.
122  for(uint8_t i = 1; i<=TASK_KEYBOARD_PARAMETERLENGTH;i++)
123  {
124  if(conf->keycodes_text[TASK_KEYBOARD_PARAMETERLENGTH-i] == 0 && i>1) break;
125  output[i+5] = (char)conf->keycodes_text[TASK_KEYBOARD_PARAMETERLENGTH-i];
126  }
127  }
128 
129  strcat(output,"\r\n");
130  return ESP_OK;
131 }

164 void task_keyboard(taskKeyboardConfig_t *param)
165 {
166  //check for param struct
167  if(param == NULL)
168  {
169  ESP_LOGE(LOG_TAG,"param is NULL ");
170  while(1) vTaskDelay(100000/portTICK_PERIOD_MS);
171  return;
172  }
173  //event bits used for pending on debounced buttons
174  EventBits_t uxBits = 0;
175  //store for later use
176  uint8_t keyboardType = param->type;
177  //calculate array index of EventGroup array (each 4 VB have an own EventGroup)
178  uint8_t evGroupIndex = param->virtualButton / 4;
179  //calculate bitmask offset within the EventGroup
180  uint8_t evGroupShift = param->virtualButton % 4;
181  //final pointer to the EventGroup used by this task
182  EventGroupHandle_t *evGroup = NULL;
183  //ticks between task timeouts
184  const TickType_t xTicksToWait = 2000 / portTICK_PERIOD_MS;
185  //local virtual button
186  uint8_t vb = param->virtualButton;
187  //local key array offset for each trigger
188  uint8_t keycodeoffset = 0;
189 
190  //if we are in singleshot mode, we do not need to test
191  //the virtual button groups (unused)
192  if(vb != VB_SINGLESHOT)
193  {
194  //check for correct offset
195  if(evGroupIndex >= NUMBER_VIRTUALBUTTONS)
196  {
197  ESP_LOGE(LOG_TAG,"virtual button group unsupported: %d ",evGroupIndex);
198  while(1) vTaskDelay(100000/portTICK_PERIOD_MS);
199  return;
200  }
201 
202  //test if event groups are already initialized, otherwise exit immediately
203  while(virtualButtonsOut[evGroupIndex] == 0)
204  {
205  ESP_LOGE(LOG_TAG,"uninitialized event group for virtual buttons, retry in 1s");
206  vTaskDelay(1000/portTICK_PERIOD_MS);
207  }
208  //event group initialized, store for later usage
209  evGroup = virtualButtonsOut[evGroupIndex];
210  }
211 
212  //test if parameter is correct
213  if(keyboardType != PRESS && keyboardType != RELEASE && \
214  keyboardType != PRESS_RELEASE_BUTTON && keyboardType != WRITE)
215  {
216  ESP_LOGE(LOG_TAG,"unknown keyboard action type,cannot continue...");
217  while(1) vTaskDelay(100000/portTICK_PERIOD_MS);
218  return;
219  }
220 
221  //local keystring (size is determined by input keystring)
222  uint8_t keylength = 0;
223  while(param->keycodes_text[keylength] != 0 && keylength < TASK_KEYBOARD_PARAMETERLENGTH) keylength++;
224 
225  //if count of keycode is 0, nothing to do here...
226  if(keylength == 0)
227  {
228  ESP_LOGI(LOG_TAG,"Empty kbd instance, quit");
229  if(vb == VB_SINGLESHOT) return; else { while(1) vTaskDelay(100000/portTICK_PERIOD_MS); }
230  }
231 
232  //copy keys to local buffer
233  //after this copying, task parameters could be freed (not recommended although)
234  uint16_t *keys = malloc(sizeof(uint16_t)*keylength);
235  memcpy(keys,param->keycodes_text,sizeof(uint16_t)*keylength);
236  if(keys != NULL)
237  {
238  ESP_LOGI(LOG_TAG,"allocated %d keycodes",keylength);
239  } else {
240  ESP_LOGE(LOG_TAG,"cannot allocate %d bytes for keyarray",keylength);
241  if(vb == VB_SINGLESHOT) return; else { while(1) vTaskDelay(100000/portTICK_PERIOD_MS); }
242  }
243 
244  while(1)
245  {
246  switch(keyboardType)
247  {
248  //these action are only triggered by a button press
249  case PRESS:
250  case RELEASE:
251  case WRITE:
252  //wait for the flag (only if not in singleshot mode)
253  if(vb != VB_SINGLESHOT)
254  {
255  uxBits = xEventGroupWaitBits(evGroup,(1<<evGroupShift),pdTRUE,pdFALSE,xTicksToWait);
256  }
257  //reset offset
258  keycodeoffset = 0;
259 
260  while(keycodeoffset < keylength)
261  {
262  //test for a valid set flag (else branch would be timeout)
263  if((uxBits & (1<<evGroupShift)) || vb == VB_SINGLESHOT)
264  {
265  if(keyboardType == PRESS)
266  {
267  //send press action to USB/BLE
268  ESP_LOGD(LOG_TAG,"press: 0x%x",keys[keycodeoffset]);
269  sendKbd(keyboard_usb_press,keyboard_ble_press,&keys[keycodeoffset]);
270 
271  }
272  if(keyboardType == RELEASE)
273  {
274  //send release action to USB/BLE
275  ESP_LOGD(LOG_TAG,"release: 0x%x",keys[keycodeoffset]);
276  sendKbd(keyboard_usb_release,keyboard_ble_release,&keys[keycodeoffset]);
277  }
278  if(keyboardType == WRITE)
279  {
280  ESP_LOGD(LOG_TAG,"press&release 1: 0x%x",keys[keycodeoffset]);
281  sendKbd(keyboard_usb_press,keyboard_ble_press,&keys[keycodeoffset]);
282  ESP_LOGD(LOG_TAG,"press&release 2: 0x%x",keys[keycodeoffset]);
283  sendKbd(keyboard_usb_release,keyboard_ble_release,&keys[keycodeoffset]);
284  }
285  }
286  keycodeoffset++;
287  }
288  break;
289 
290  //this action is triggered on a button press AND a release
291  case PRESS_RELEASE_BUTTON:
292  //wait for the flags (press & release)
293  if(vb != VB_SINGLESHOT)
294  {
295  uxBits = xEventGroupWaitBits(evGroup,(1<<evGroupShift)| \
296  (1<<(evGroupShift+4)),pdTRUE,pdFALSE,xTicksToWait);
297  }
298  //reset key array offset
299  keycodeoffset = 0;
300  while(keycodeoffset < keylength)
301  {
302  //test for a valid set flag
303  if((uxBits & (1<<evGroupShift)) || vb == VB_SINGLESHOT)
304  {
305  ESP_LOGD(LOG_TAG,"press&release button 1: 0x%x",keys[keycodeoffset]);
306  sendKbd(keyboard_usb_press,keyboard_ble_press,&keys[keycodeoffset]);
307  }
308  if((uxBits & (1<<(evGroupShift+4))) || vb == VB_SINGLESHOT)
309  {
310  ESP_LOGD(LOG_TAG,"press&release button 2: 0x%x",keys[keycodeoffset]);
311  sendKbd(keyboard_usb_release,keyboard_ble_release,&keys[keycodeoffset]);
312  }
313  keycodeoffset++;
314  }
315  break;
316 
317  default:
318  ESP_LOGE(LOG_TAG,"unknown keyboard action type,quit...");
319  free(keys);
320  while(1) vTaskDelay(100000/portTICK_PERIOD_MS);
321  return;
322  }
323 
324  //function tasks in singleshot must return
325  if(vb == VB_SINGLESHOT)
326  {
327  free(keys);
328  return;
329  }
330  }
331 }
