74 BLEService* BLEServer::createService(BLEUUID uuid, uint32_t numHandles) {
75  ESP_LOGD(LOG_TAG, ">> createService - %s", uuid.toString().c_str());
76  m_semaphoreCreateEvt.take("createService");
77 
78  // Check that a service with the supplied UUID does not already exist.
79  if (m_serviceMap.getByUUID(uuid) != nullptr) {
80  ESP_LOGE(LOG_TAG, "<< Attempt to create a new service with uuid %s but a service with that UUID already exists.",
81  uuid.toString().c_str());

83  return nullptr;
84  }
85 
86  BLEService* pService = new BLEService(uuid, numHandles);

88  pService->executeCreate(this); // Perform the API calls to actually create the service.
89 
90  m_semaphoreCreateEvt.wait("createService");
91 
92  ESP_LOGD(LOG_TAG, "<< createService");

169 void BLEServer::handleGATTServerEvent(
170  esp_gatts_cb_event_t event,
171  esp_gatt_if_t gatts_if,
172  esp_ble_gatts_cb_param_t* param) {
173 
174  ESP_LOGD(LOG_TAG, ">> handleGATTServerEvent: %s",
175  BLEUtils::gattServerEventTypeToString(event).c_str());
176 
177  // Invoke the handler for every Service we have.
178  m_serviceMap.handleGATTServerEvent(event, gatts_if, param);
179 
180  switch(event) {
181  // ESP_GATTS_ADD_CHAR_EVT - Indicate that a characteristic was added to the service.
182  // add_char:
183  // - esp_gatt_status_t status
184  // - uint16_t attr_handle
185  // - uint16_t service_handle
186  // - esp_bt_uuid_t char_uuid
187  //
188  case ESP_GATTS_ADD_CHAR_EVT: {
189  break;
190  } // ESP_GATTS_ADD_CHAR_EVT
191 
192 
193  // ESP_GATTS_CONNECT_EVT
194  // connect:
195  // - uint16_t conn_id
196  // - esp_bd_addr_t remote_bda
197  // - bool is_connected
198  //
199  case ESP_GATTS_CONNECT_EVT: {
200  m_connId = param->connect.conn_id; // Save the connection id.
201  if (m_pServerCallbacks != nullptr) {
202  m_pServerCallbacks->onConnect(this);
203  }
204  m_connectedCount++; // Increment the number of connected devices count.
205  break;
206  } // ESP_GATTS_CONNECT_EVT
207 
208 
209  // ESP_GATTS_CREATE_EVT
210  // Called when a new service is registered as having been created.
211  //
212  // create:
213  // * esp_gatt_status_t status
214  // * uint16_t service_handle
215  // * esp_gatt_srvc_id_t service_id
216  //
217  case ESP_GATTS_CREATE_EVT: {
218  BLEService* pService = m_serviceMap.getByUUID(param->create.service_id.id.uuid);
219  m_serviceMap.setByHandle(param->create.service_handle, pService);
220  m_semaphoreCreateEvt.give();
221  break;
222  } // ESP_GATTS_CREATE_EVT
223 
224 
225  // ESP_GATTS_DISCONNECT_EVT
226  //
227  // disconnect
228  // - uint16_t conn_id
229  // - esp_bd_addr_t remote_bda
230  // - bool is_connected
231  //
232  // If we receive a disconnect event then invoke the callback for disconnects (if one is present).
233  // we also want to start advertising again.
234  case ESP_GATTS_DISCONNECT_EVT: {
235  m_connectedCount--; // Decrement the number of connected devices count.
236  if (m_pServerCallbacks != nullptr) { // If we have callbacks, call now.
237  m_pServerCallbacks->onDisconnect(this);
238  }
239  startAdvertising(); //- do this with some delay from the loop()
240  break;
241  } // ESP_GATTS_DISCONNECT_EVT
242 
243 
244  // ESP_GATTS_READ_EVT - A request to read the value of a characteristic has arrived.
245  //
246  // read:
247  // - uint16_t conn_id
248  // - uint32_t trans_id
249  // - esp_bd_addr_t bda
250  // - uint16_t handle
251  // - uint16_t offset
252  // - bool is_long
253  // - bool need_rsp
254  //
255  case ESP_GATTS_READ_EVT: {

257  } // ESP_GATTS_READ_EVT
