97 {
98  uint8_t buffer[HID_CC_IN_RPT_LEN] = {0, 0};

121  buffer[i+2] = keyboard_cmd[i];

123 
124  LOG_DEBUG("the key vaule = %d,%d,%d, %d, %d, %d,%d, %d", buffer[0], buffer[1], buffer[2], buffer[3], buffer[4], buffer[5], buffer[6], buffer[7]);
125  hid_dev_send_report(hidd_le_env.gatt_if, conn_id,
