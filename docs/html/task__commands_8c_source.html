112 uint8_t doGeneralCmdParsing(uint8_t *cmdBuffer)
113 {
114  uint16_t param = 0;
115 
116  /*++++ AT ID ++++*/
117  if(memcmp(cmdBuffer,"AT ID",5) == 0) {
118  uint32_t sent = halSerialSendUSBSerial(HAL_SERIAL_TX_TO_CDC,IDSTRING,sizeof(IDSTRING),20);
119  if(sent != sizeof(IDSTRING))
120  {
121  ESP_LOGE(LOG_TAG,"Error sending response of AT ID");
122  } else return 1;
123  }
124  /*++++ AT BT ++++*/
125  if(memcmp(cmdBuffer,"AT BT",5) == 0) {
126  param = strtol((char*)&(cmdBuffer[6]),NULL,10);
127  switch(param)
128  {
129  case 1:
130  xEventGroupClearBits(connectionRoutingStatus,DATATO_BLE);
131  xEventGroupSetBits(connectionRoutingStatus,DATATO_USB);
132  ESP_LOGI(LOG_TAG,"Switched routing to USB only");
133  return 1;
134  break;
135  case 2:
136  xEventGroupClearBits(connectionRoutingStatus,DATATO_USB);
137  xEventGroupSetBits(connectionRoutingStatus,DATATO_BLE);
138  ESP_LOGI(LOG_TAG,"Switched routing to BLE only");
139  return 1;
140  break;
141  case 3:
142  xEventGroupSetBits(connectionRoutingStatus,DATATO_USB|DATATO_BLE);
143  ESP_LOGI(LOG_TAG,"Switched routing to USB & BLE");
144  return 1;
145  break;
146  default: sendErrorBack("AT BT param"); return 0;
147  }
148  }
149 
150  //not consumed, no general command found
151  return 0;
152 }

161 uint8_t doMouseParsing(uint8_t *cmdBuffer, taskMouseConfig_t *mouseinstance)
162 {
163  //we are doing AT command triggered action, so this mouse command
164  //is always a singleshot. Simply call the task function.
165  mouseinstance->virtualButton = VB_SINGLESHOT;
166 
167  /*++++ mouse clicks ++++*/
168  //AT CL, AT CR, AT CM, AT CD
169  if(memcmp(cmdBuffer,"AT C",4) == 0)
170  {
171  mouseinstance->actionparam = M_CLICK;
172  switch(cmdBuffer[4])
173  {
174  //do single clicks (left,right,middle)
175  case 'L': mouseinstance->type = LEFT; task_mouse(mouseinstance); return 1;
176  case 'R': mouseinstance->type = RIGHT; task_mouse(mouseinstance); return 1;
177  case 'M': mouseinstance->type = MIDDLE; task_mouse(mouseinstance); return 1;
178  //do left double click
179  case 'D':
180  mouseinstance->type = LEFT;
181  mouseinstance->actionparam = M_DOUBLE;
182  task_mouse(mouseinstance);
183  return 1;
184  //not an AT C? command for mouse, return 0 (not consumed)
185  default: return 0;
186  }
187  }
188 
189  /*++++ mouse wheel up/down; set stepsize ++++*/
190  if(memcmp(cmdBuffer,"AT W",4) == 0)
191  {
192  mouseinstance->type = WHEEL;
193  mouseinstance->actionparam = M_UNUSED;
194  switch(cmdBuffer[4])
195  {
196  //move mouse wheel up/down
197  case 'U': mouseinstance->actionvalue = mouse_get_wheel(); task_mouse(mouseinstance); return 1;
198  case 'D': mouseinstance->actionvalue = -mouse_get_wheel(); task_mouse(mouseinstance); return 1;
199  //set mouse wheel stepsize. If unsuccessful, default will return 0
200  case 'S':
201  int steps = strtol((char*)&(cmdBuffer[6]),NULL,10);
202  if(steps > 127 || steps < 127)
203  {
204  sendErrorBack("Wheel size out of range! (-127 to 127)");
205  return 0;
206  } else {
207  mouse_set_wheel(steps);
208  ESP_LOGI(LOG_TAG,"Setting mouse wheel steps: %d",steps);
209  return 1;
210  }
211  default: return 0;
212  }
213  }
214 
215  /*++++ mouse button press ++++*/
216  //AT PL, AT PR, AT PM
217  if(memcmp(cmdBuffer,"AT P",4) == 0)
218  {
219  mouseinstance->actionparam = M_HOLD;
220  switch(cmdBuffer[4])
221  {
222  case 'L': mouseinstance->type = LEFT; task_mouse(mouseinstance); return 1;
223  case 'R': mouseinstance->type = RIGHT; task_mouse(mouseinstance); return 1;
224  case 'M': mouseinstance->type = MIDDLE; task_mouse(mouseinstance); return 1;
225  default: return 0;
226  }
227  }
228 
229  /*++++ mouse button release ++++*/
230  //AT RL, AT RR, AT RM
231  if(memcmp(cmdBuffer,"AT R",4) == 0)
232  {
233  mouseinstance->actionparam = M_RELEASE;
234  switch(cmdBuffer[4])
235  {
236  case 'L': mouseinstance->type = LEFT; task_mouse(mouseinstance); return 1;
237  case 'R': mouseinstance->type = RIGHT; task_mouse(mouseinstance); return 1;
238  case 'M': mouseinstance->type = MIDDLE; task_mouse(mouseinstance); return 1;
239  default: return 0;
240  }
241  }
242 
243  /*++++ mouse move ++++*/
244  //AT RL, AT RR, AT RM
245  if(memcmp(cmdBuffer,"AT M",4) == 0)
246  {
247  mouseinstance->actionparam = M_UNUSED;
248  //mouseinstance->actionvalue = atoi((char*)&(cmdBuffer[6]));
249  //mouseinstance->actionvalue = strtoimax((char*)&(cmdBuffer[6]),&endBuf,10);
250  int param = strtol((char*)&(cmdBuffer[5]),NULL,10);
251  ESP_LOGI(LOG_TAG,"Mouse move %c, %d",cmdBuffer[4],mouseinstance->actionvalue);
252  if(param > 127 && param < -127)
253  {
254  ESP_LOGW(LOG_TAG,"AT MX parameter limit -127 - 127");
255  return 0;
256  } else {
257  mouseinstance->actionvalue = param;
258  }
259  switch(cmdBuffer[4])
260  {
261  case 'X': mouseinstance->type = X; task_mouse(mouseinstance); return 1;
262  case 'Y': mouseinstance->type = Y; task_mouse(mouseinstance); return 1;
263  default: return 0;
264  }
265  }
266 
267  return 0;
268 }

270 void task_commands(void *params)
271 {
272  uint8_t queuesready = checkqueues();
273  int received;
274  uint8_t commandBuffer[ATCMD_LENGTH];
275 
276  taskMouseConfig_t cmdMouse;
277  taskKeyboardConfig_t cmdKeyboard;
278  //taskJoystickConfig_t cmdJoystick;
279 
280  while(1)
281  {
282  if(queuesready)
283  {
284  //wait 30ms, to be nice to other tasks
285  vTaskDelay(30/portTICK_PERIOD_MS);
286 
287  //wait for incoming data
288  received = halSerialReceiveUSBSerial(commandBuffer,ATCMD_LENGTH);
289  //check received data at least for length
290  if(received < 5)
291  {
292  //special command "AT" without further command:
293  if(received >= 2 && memcmp(commandBuffer,"AT",2) == 0)
294  {
295  halSerialSendUSBSerial(HAL_SERIAL_TX_TO_CDC,"OK\r\n",5,100);
296  halSerialFlushRX();
297  continue;
298  }
299  if(received > 0)
300  {
301  ESP_LOGW(LOG_TAG,"Invalid AT commandlength %d",received);
302  ESP_LOG_BUFFER_HEXDUMP(LOG_TAG,commandBuffer,received,ESP_LOG_DEBUG);
303  }
304  halSerialFlushRX();
305  continue;
306  }
307 
308  //do parsing :-)
309  //simplified into smaller functions, to make reading easy.
310  if(doMouseParsing(commandBuffer,&cmdMouse)) continue;
311  if(doKeyboardParsing(commandBuffer,&cmdKeyboard)) continue;
312  //if(doJoystickParsing(commandBuffer,&cmdJoystick)) continue;
313  if(doStorageParsing(commandBuffer)) continue;
314  if(doGeneralCmdParsing(commandBuffer)) continue;
315  if(doInfraredParsing(commandBuffer)) continue;
316  if(doMouthpieceSettingsParsing(commandBuffer)) continue;
317 
318  //check reply by LPC USB chip:
319  if(memcmp(commandBuffer,"__OK__",6) == 0) continue;
320 
321  if(memcmp(commandBuffer,"_parameter error",16) == 0)
322  {
323  ESP_LOGE(LOG_TAG,"USB reply: parameter error");
324  continue;
325  }
326  if(memcmp(commandBuffer,"_unknown command",16) == 0)
327  {
328  ESP_LOGE(LOG_TAG,"USB reply: unknown cmd");
329  continue;
330  }
331  if(memcmp(commandBuffer,"_unknown error code",19) == 0)
332  {
333  ESP_LOGE(LOG_TAG,"USB reply: unknown error code");
334  continue;
335  }
336 
337  //if we are here, no parser was finding commands
338  ESP_LOGW(LOG_TAG,"Invalid AT cmd (%d characters), flushing:",received);
339  ESP_LOG_BUFFER_CHAR_LEVEL(LOG_TAG,commandBuffer,received, ESP_LOG_WARN);
340  halSerialFlushRX();
341  //ESP_LOG_BUFFER_HEXDUMP(LOG_TAG,commandBuffer,received,ESP_LOG_DEBUG);
342  halSerialSendUSBSerial(HAL_SERIAL_TX_TO_CDC,"?\r\n",2,100);
343 
344  } else {
345  //check again for initialized queues
346  ESP_LOGE(LOG_TAG,"Queues uninitialized, rechecking in 1s");
347  vTaskDelay(1000/portTICK_PERIOD_MS);
348  //re-check
349  queuesready = checkqueues();
350  }
351  }
352 }
