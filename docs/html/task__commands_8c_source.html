155 parserstate_t doMouthpieceSettingsParsing(uint8_t *cmdBuffer, taskNoParameterConfig_t *instance)
156 {
157  generalConfig_t *currentcfg = configGetCurrent();
158 
159  //clear any previous data
160  memset(instance,0,sizeof(taskNoParameterConfig_t));
161  instance->virtualButton = requestVBUpdate;
162  //set everything up for updates
163  requestVBTask = (void (*)(void *))&task_calibration;
164  requestVBParameterSize = sizeof(taskNoParameterConfig_t);
165  requestVBParameter = instance;
166  requestVBType = T_CALIBRATE;
167 
168  /*++++ calibrate mouthpiece ++++*/
169  if(CMD("AT CA"))
170  {
171  ESP_LOGI(LOG_TAG,"Calibrate");
172  return TRIGGERTASK;
173  }
174 
175  /*++++ stop report raw values ++++*/
176  if(CMD("AT ER"))
177  {
178  currentcfg->adc.reportraw = 0;
179  requestUpdate = 1;
180  ESP_LOGI(LOG_TAG,"Stop reporting raw values");
181  return NOACTION;
182  }
183  /*++++ start report raw values to serial ++++*/
184  if(CMD("AT SR"))
185  {
186  currentcfg->adc.reportraw = 1;
187  requestUpdate = 1;
188  ESP_LOGI(LOG_TAG,"Start reporting raw values");
189  return NOACTION;
190  }
191 
192  /*++++ mouthpiece mode ++++*/
193  if(CMD("AT MM"))
194  {
195  //assign to gain value
196  switch(cmdBuffer[6])
197  {
198  case '0': currentcfg->adc.mode = THRESHOLD; requestUpdate = 1; return NOACTION;
199  case '1': currentcfg->adc.mode = MOUSE; requestUpdate = 1; return NOACTION;
200  case '2': currentcfg->adc.mode = JOYSTICK; requestUpdate = 1; return NOACTION;
201  default: sendErrorBack("Mode is 0,1 or 2"); return UNKNOWNCMD;
202  }
203  }
204 
205  /*++++ mouthpiece mode - switch ++++*/
206  if(CMD("AT SW"))
207  {
208  switch(currentcfg->adc.mode)
209  {
210  case MOUSE: currentcfg->adc.mode = THRESHOLD; break;
211  case THRESHOLD: currentcfg->adc.mode = MOUSE; break;
212  case JOYSTICK: sendErrorBack("AT SW switches between mouse and threshold only"); break;
213  }
214  requestUpdate = 1;
215  return NOACTION;
216  }
217 
218  /*++++ mouthpiece gain ++++*/
219  //AT GU, GD, GL, GR
220  if(CMD4("AT G"))
221  {
222  unsigned int param = strtol((char*)&(cmdBuffer[5]),NULL,10);
223  ESP_LOGI(LOG_TAG,"Gain %c, %d",cmdBuffer[4],param);
224  if(param > 100)
225  {
226  sendErrorBack("Gain is 0-100");
227  return UNKNOWNCMD;
228  } else {
229  //assign to gain value
230  switch(cmdBuffer[4])
231  {
232  case 'U': currentcfg->adc.gain[0] = param; requestUpdate = 1; return NOACTION;
233  case 'D': currentcfg->adc.gain[1] = param; requestUpdate = 1; return NOACTION;
234  case 'L': currentcfg->adc.gain[2] = param; requestUpdate = 1; return NOACTION;
235  case 'R': currentcfg->adc.gain[3] = param; requestUpdate = 1; return NOACTION;
236  default: return UNKNOWNCMD;
237  }
238  }
239  }
240 
241  /*++++ mouthpiece sensitivity/acceleration ++++*/
242  //AT AX, AY, AC
243  if(CMD4("AT A"))
244  {
245  unsigned int param = strtol((char*)&(cmdBuffer[5]),NULL,10);
246  ESP_LOGI(LOG_TAG,"Sensitivity/accel %c, %d",cmdBuffer[4],param);
247  if(param > 100)
248  {
249  sendErrorBack("Sensitivity/accel is 0-100");
250  return UNKNOWNCMD;
251  } else {
252  //assign to sensitivity/acceleration value
253  switch(cmdBuffer[4])
254  {
255  case 'X': currentcfg->adc.sensitivity_x = param; requestUpdate = 1; return NOACTION;
256  case 'Y': currentcfg->adc.sensitivity_x = param; requestUpdate = 1; return NOACTION;
257  case 'C': currentcfg->adc.acceleration = param; requestUpdate = 1; return NOACTION;
258  default: return UNKNOWNCMD;
259  }
260  }
261  }
262 
263  /*++++ threshold sip/puff++++*/
264  //AT TS/TP
265  if(CMD4("AT T"))
266  {
267  unsigned int param = strtol((char*)&(cmdBuffer[5]),NULL,10);
268  ESP_LOGI(LOG_TAG,"Threshold %c, %d",cmdBuffer[4],param);
269  switch(cmdBuffer[4])
270  {
271  case 'S':
272  if(param > 512)
273  {
274  sendErrorBack("Threshold sip is 0-512");
275  return UNKNOWNCMD;
276  } else {
277  currentcfg->adc.threshold_sip = param;
278  requestUpdate = 1;
279  return NOACTION;
280  }
281  break;
282 
283  case 'T':
284  if(param < 512 || param > 1023)
285  {
286  sendErrorBack("Threshold puff is 512-1023");
287  return UNKNOWNCMD;
288  } else {
289  currentcfg->adc.threshold_puff = param;
290  requestUpdate = 1;
291  return NOACTION;
292  }
293  break;
294  default: return UNKNOWNCMD;
295  }
296  }
297 
298  /*++++ threshold strong sip/puff++++*/
299  //AT TS/TP
300  if(CMD4("AT S"))
301  {
302  unsigned int param = strtol((char*)&(cmdBuffer[5]),NULL,10);
303  ESP_LOGI(LOG_TAG,"Threshold strong %c, %d",cmdBuffer[4],param);
304  switch(cmdBuffer[4])
305  {
306  case 'S':
307  if(param > 512)
308  {
309  sendErrorBack("Threshold strong sip is 0-512");
310  return UNKNOWNCMD;
311  } else {
312  currentcfg->adc.threshold_strongsip = param;
313  requestUpdate = 1;
314  return NOACTION;
315  }
316  break;
317 
318  case 'T':
319  if(param < 512 || param > 1023)
320  {
321  sendErrorBack("Threshold strong puff is 512-1023");
322  return UNKNOWNCMD;
323  } else {
324  currentcfg->adc.threshold_strongpuff = param;
325  requestUpdate = 1;
326  return NOACTION;
327  }
328  break;
329  default: return UNKNOWNCMD;
330  }
331  }
332 
333  //not consumed, no command found for mouthpiece settings
334  return UNKNOWNCMD;

337 parserstate_t doStorageParsing(uint8_t *cmdBuffer, taskConfigSwitcherConfig_t *instance)
338 {
339  char slotname[SLOTNAME_LENGTH];
340  uint32_t tid = 0;
341  uint8_t slotnumber = 0;
342  generalConfig_t * currentcfg = configGetCurrent();
343  //clear any previous data
344  memset(instance,0,sizeof(taskConfigSwitcherConfig_t));
345 
346  //set everything up for updates
347  instance->virtualButton = requestVBUpdate;
348  requestVBTask = (void (*)(void *))&task_configswitcher;
349  requestVBParameterSize = sizeof(taskConfigSwitcherConfig_t);
350  requestVBParameter = instance;
351  requestVBType = T_CONFIGCHANGE;
352 
353  /*++++ save slot ++++*/
354  if(CMD("AT SA"))
355  {
356  if(halStorageStartTransaction(&tid,10) != ESP_OK)
357  {
358  ESP_LOGE(LOG_TAG,"Cannot start storage transaction");
359  return UNKNOWNCMD;
360  }
361  char *pch;
362  //find end of slotname
363  pch = strpbrk((char *)&cmdBuffer[6],"\r\n");
364  //set 0 terminator
365  *pch = '\0';
366  strcpy(slotname,(char*)&cmdBuffer[6]);
367  ESP_LOGI(LOG_TAG,"Save slot under name: %s",slotname);
368  halStorageGetNumberOfSlots(tid, &slotnumber);
369  //store general config
370  if(halStorageStore(tid,currentcfg,slotname,slotnumber+1) != ESP_OK)
371  {
372  ESP_LOGE(LOG_TAG,"Cannot store general cfg");
373  halStorageFinishTransaction(tid);
374  return UNKNOWNCMD;
375  }
376  //store each virtual button
377  for(uint8_t i = 0; i<(NUMBER_VIRTUALBUTTONS*4); i++)
378  {
379  size_t sizevb = 0;
380 
381  switch(currentcfg->virtualButtonCommand[i])
382  {
383  case T_MOUSE: sizevb = sizeof(taskMouseConfig_t); break;
384  case T_KEYBOARD: sizevb = sizeof(taskKeyboardConfig_t); break;
385  case T_CONFIGCHANGE: sizevb = sizeof(taskConfigSwitcherConfig_t); break;
386  case T_CALIBRATE: sizevb = sizeof(taskNoParameterConfig_t); break;
387  //case T_SENDIR: sizevb = sizeof(taskKeyboardConfig_t); break;
389  case T_NOFUNCTION: sizevb = sizeof(taskNoParameterConfig_t); break;
390  default: break;
391  }
392  if(halStorageStoreSetVBConfigs(slotnumber+1,i,currentcfg->virtualButtonConfig[i], \
393  sizevb,tid) != ESP_OK)
394  {
395  ESP_LOGE(LOG_TAG,"Cannot store VB %d",i);
396  halStorageFinishTransaction(tid);
397  return UNKNOWNCMD;
398  }
399  }
400  halStorageFinishTransaction(tid);
401  return NOACTION;
402  }
403 
404  /*++++ AT NE (load next) +++*/
405  if(CMD("AT NE"))
406  {
407  //save to config
408  instance->virtualButton = requestVBUpdate;
409  strcpy(instance->slotName,"__NEXT");
410  ESP_LOGI(LOG_TAG,"Load next, by parameter: %s",instance->slotName);
411  //we want to have the config sent to the task
412  return TRIGGERTASK;
413  }
414 
415  /*++++ AT LO (load) ++++*/
416  if(CMD("AT LO"))
417  {
418  //find end of slotname
419  char *pch = strpbrk((char *)&cmdBuffer[6],"\r\n");
420  //set 0 terminator
421  *pch = '\0';
422  //save to config
423  instance->virtualButton = requestVBUpdate;
424  strcpy(instance->slotName,(char*)&cmdBuffer[6]);
425  ESP_LOGI(LOG_TAG,"Load slot name: %s",instance->slotName);
426  //we want to have the config sent to the task
427  return TRIGGERTASK;
428  }
429  //not consumed, no command found for storage
430  return UNKNOWNCMD;
431 }

440 parserstate_t doGeneralCmdParsing(uint8_t *cmdBuffer)
441 {
442  uint16_t param = 0;
443  uint8_t param8 = 0;
444  generalConfig_t *currentcfg;
445 
446  currentcfg = configGetCurrent();
447  if(currentcfg == NULL)
448  {
449  ESP_LOGE(LOG_TAG,"Current config is null, cannot update general cmd");
450  return UNKNOWNCMD;
451  }
452 
453 
454  /*++++ AT RA ++++*/
455  if(CMD("AT RA")) {
456  //reset the reports (keyboard only, excepting all other parts)
457  halBLEReset(0xFE);
458  halSerialReset(0xFE);
459  return NOACTION;
460  }
461 
462  /*++++ AT LA + LI ++++*/
463  if(CMD4("AT L")) {
464  switch(cmdBuffer[4])
465  {
466  //print current slot configurations
467  case 'A': printAllSlots(1); return NOACTION;
468  //print slot names only
469  case 'I': printAllSlots(0); return NOACTION;
470  default: return UNKNOWNCMD;
471  }
472  }
473 
474  /*++++ AT KL ++++*/
475  if(CMD("AT KL")) {
476  param8 = strtol((char*)&(cmdBuffer[6]),NULL,10);
477  if(param8 < LAYOUT_MAX)
478  {
479  ESP_LOGI(LOG_TAG,"Changed locale from %d to %d",currentcfg->locale,param8);
480  currentcfg->locale = param8;
481  requestUpdate = 1;
482  return NOACTION;
483  } else {
484  sendErrorBack("Locale out of range");
485  return UNKNOWNCMD;
486  }
487  }
488 
489  /*++++ AT ID ++++*/
490  if(CMD("AT ID")) {
491  uint32_t sent = halSerialSendUSBSerial(HAL_SERIAL_TX_TO_CDC,IDSTRING,sizeof(IDSTRING),20);
492  if(sent != sizeof(IDSTRING))
493  {
494  ESP_LOGE(LOG_TAG,"Error sending response of AT ID");
495  return UNKNOWNCMD;
496  } else return NOACTION;
497  }
498 
499  /*++++ AT DE ++++*/
500  if(CMD("AT DE")) {
501  uint32_t tid;
502  if(halStorageStartTransaction(&tid,20) != ESP_OK)
503  {
504  return UNKNOWNCMD;
505  } else {
506  if(halStorageDeleteSlot(0,tid) != ESP_OK)
507  {
508  sendErrorBack("Error deleting all slots");
509  halStorageFinishTransaction(tid);
510  return UNKNOWNCMD;
511  } else {
512  halStorageFinishTransaction(tid);
513  return NOACTION;
514  }
515  }
516  }
517 
518  /*++++ AT BM ++++*/
519  if(CMD("AT BM")) {
520  param = strtol((char*)&(cmdBuffer[6]),NULL,10);
521  if(param >= VB_MAX)
522  {
523  sendErrorBack("VB nr too high");
524  return UNKNOWNCMD;
525  } else {
526  requestVBUpdate = param;
527  //return 3 to definetly trigger nothing else
528  return WAITFORNEWATCMD;
529  }
530  }
531 
532  /*++++ AT DL ++++*/
533  if(CMD4("AT DL")) {
534  uint32_t tid;
535  uint8_t slotnumber;
536  char *pch;
537 
538  if(halStorageStartTransaction(&tid,20) != ESP_OK)
539  {
540  return UNKNOWNCMD;
541  }
542 
543  switch(cmdBuffer[4])
544  {
545  //delete by given number
546  case 'L': slotnumber = strtol((char*)&(cmdBuffer[6]),NULL,10); break;
547  //delete by given name
548  case 'N':
549  //find end of slotname
550  pch = strpbrk((char *)&cmdBuffer[6],"\r\n");
551  //set 0 terminator
552  *pch = '\0';
553  halStorageGetNumberForName(tid,&slotnumber,(char *)&cmdBuffer[6]);
554  break;
555  default: return UNKNOWNCMD;
556  }
557 
558  if(halStorageDeleteSlot(slotnumber,tid) != ESP_OK)
559  {
560  sendErrorBack("Error deleting slot");
561  halStorageFinishTransaction(tid);
562  return UNKNOWNCMD;
563  } else {
564  halStorageFinishTransaction(tid);
565  return NOACTION;
566  }
567  }
568 
569  /*++++ AT BT ++++*/
570  if(CMD("AT BT")) {
571  param = strtol((char*)&(cmdBuffer[6]),NULL,10);
572  switch(param)
573  {
574  case 1:
575  currentcfg->ble_active = 0;
576  currentcfg->usb_active = 1;
577  requestUpdate = 1;
578  return NOACTION;
579  break;
580  case 2:
581  currentcfg->ble_active = 1;
582  currentcfg->usb_active = 0;
583  requestUpdate = 1;
584  return NOACTION;
585  break;
586  case 3:
587  currentcfg->ble_active = 1;
588  currentcfg->usb_active = 1;
589  requestUpdate = 1;
590  return NOACTION;
591  break;
592  default: sendErrorBack("AT BT param"); return UNKNOWNCMD;
593  }
594  }
595 
596  //not consumed, no general command found
597  return UNKNOWNCMD;
598 }

600 parserstate_t doKeyboardParsing(uint8_t *cmdBuffer, taskKeyboardConfig_t *kbdinstance, int length)
601 {
602  //clear any previous data
603  memset(kbdinstance,0,sizeof(taskKeyboardConfig_t));
604 
605  //set everything up for updates
606  requestVBTask = (void (*)(void *))&task_keyboard;
607  requestVBParameterSize = sizeof(taskKeyboardConfig_t);
608  requestVBParameter = kbdinstance;
609  requestVBType = T_KEYBOARD;
610 
611  //use global virtual button (normally VB_SINGLESHOT,
612  //can be changed by "AT BM"
613  kbdinstance->virtualButton = requestVBUpdate;
614  int offset = 0;
615  int offsetOut = 0;
616  int offsetEnd = TASK_KEYBOARD_PARAMETERLENGTH-1;
617  uint8_t deadkeyfirst = 0;
618  uint8_t modifier = 0;
619  uint8_t keycode = 0;
620 
621  //get general config (used to get the locale)
622  generalConfig_t *currentcfg;
623  currentcfg = configGetCurrent();
624 
625  if(currentcfg == NULL)
626  {
627  ESP_LOGE(LOG_TAG,"Major error, general config is NULL");
628  return UNKNOWNCMD;
629  }
630 
631  /*++++ AT KW ++++*/
632  if(CMD("AT KW")) {
633  //set keyboard to write
634  kbdinstance->type = WRITE;
635  //remove \r & \n
636  //cmdBuffer[length-2] = 0;
637  //cmdBuffer[length-1] = 0;
638 
639  //save each byte from KW string to keyboard instance
640  //offset must be less then buffer length - 6 bytes for "AT KW "
641  while(offset <= (length - 5))
642  {
643 
644 
645  //terminate...
646  if(cmdBuffer[offset+6] == '\r' || cmdBuffer[offset+6] == '\n' || \
647  cmdBuffer[offset+6] == 0 || offsetOut == offsetEnd) break;
648 
649  //save original byte
650  kbdinstance->keycodes_text[offsetEnd] = cmdBuffer[offset+6];
651  offsetEnd--;
652 
653  //parse ASCII/unicode to keycode sequence
654  keycode = unicode_to_keycode(cmdBuffer[offset+6], currentcfg->locale);
655  deadkeyfirst = deadkey_to_keycode(keycode,currentcfg->locale);
656  if(deadkeyfirst != 0) deadkeyfirst = keycode_to_key(deadkeyfirst);
657  modifier = keycode_to_modifier(keycode, currentcfg->locale);
658  keycode = keycode_to_key(keycode);
659 
660  //if a keycode is found
661  if(keycode != 0)
662  {
663  //is a deadkey necessary?
664  if(deadkeyfirst != 0)
665  {
666  kbdinstance->keycodes_text[offsetOut] = deadkeyfirst;
667  offsetOut++;
668  ESP_LOGD(LOG_TAG, "Deadkey 0x%X@%d",deadkeyfirst,offsetOut);
669  }
670 
671  //save keycode + modifier
672  kbdinstance->keycodes_text[offsetOut] = keycode | (modifier << 8);
673  ESP_LOGD(LOG_TAG, "Keycode 0x%X@%d, modifier: 0x%X",keycode,offsetOut,modifier);
674  offsetOut++;
675  } else {
676  ESP_LOGD(LOG_TAG, "Need another byte (unicode)");
677  }
678  offset++;
679 
680  if(offset == (TASK_KEYBOARD_PARAMETERLENGTH - 1))
681  {
682  sendErrorBack("AT KW parameter too long");
683  return UNKNOWNCMD;
684  }
685  }
686  //terminate keycode array with 0
687  ESP_LOGD(LOG_TAG,"Terminating @%d with 0",offsetOut);
688  kbdinstance->keycodes_text[offsetOut] = 0;
689  return TRIGGERTASK;
690  }
691 
692  /*++++ AT KP + KH + KR ++++*/
693  // (all commands with key identifiers)
694  if(CMD4("AT K")) {
695  //set keyboard instance accordingly
696  switch(cmdBuffer[4])
697  {
698  //KP: press with VB press flag, release with VB release flag
699  case 'P': kbdinstance->type = PRESS_RELEASE_BUTTON; break;
700  //KR: just release
701  case 'R': kbdinstance->type = RELEASE; break;
702  //KH: just press (hold)
703  case 'H': kbdinstance->type = PRESS; break;
704  default: return UNKNOWNCMD; break;
705  }
706 
707  //remove \r & \n
708  cmdBuffer[length-2] = ' ';
709  cmdBuffer[length-1] = ' ';
710 
711  char *pch;
712  uint8_t cnt = 0;
713  uint16_t keycode = 0;
714  pch = strpbrk((char *)&cmdBuffer[5]," ");
715  while (pch != NULL)
716  {
717  ESP_LOGD(LOG_TAG,"Token: %s",pch+1);
718 
719  //check if token seems to be a KEY_*
720  if(memcmp(pch+1,"KEY_",4) != 0)
721  {
722  ESP_LOGW(LOG_TAG,"Not a valid KEY_* identifier");
723  } else {
724  //parse identifier to keycode
725  keycode = parseIdentifierToKeycode(pch+1);
726 
727  if(keycode != 0)
728  {
729  //if found, save...
730  ESP_LOGD(LOG_TAG,"Keycode: %d",keycode);
731 
732  //get deadkeys
733  uint16_t deadk = 0;
734  deadk = deadkey_to_keycode(keycode, currentcfg->locale);
735  //found a deadkey
736  if(deadk != 0)
737  {
738  //save deadkey's keycode & modifier
739  kbdinstance->keycodes_text[cnt] = keycode_to_key(deadk) | \
740  (keycode_to_modifier(deadk, currentcfg->locale) << 8);
741  ESP_LOGI(LOG_TAG,"Deadkey (keycode + modifier): %4X @%d",kbdinstance->keycodes_text[cnt], cnt);
742  cnt++;
743  }
744 
745  //KEY_ * identifiers are either a key or a modifier
746  //determine which type and save accordingly:
747  if(keycode_is_modifier(keycode))
748  {
749  kbdinstance->keycodes_text[cnt] = keycode_to_key(keycode) << 8;
750  } else {
751  kbdinstance->keycodes_text[cnt] = keycode_to_key(keycode);
752  }
753  ESP_LOGI(LOG_TAG,"Keycode + modifier: %4X @ %d",kbdinstance->keycodes_text[cnt], cnt);
754  cnt++;
755  if(cnt == (TASK_KEYBOARD_PARAMETERLENGTH - 1))
756  {
757  sendErrorBack("AT KP/KH/KR parameter too long");
758  return UNKNOWNCMD;
759  }
760  } else {
761  ESP_LOGW(LOG_TAG,"No keycode found for this token.");
762  }
763  }
764  //split into tokens
765  pch = strpbrk(pch+1, " ");

903 void task_commands(void *params)
904 {
905  uint8_t queuesready = checkqueues();
906  int received;
907  //used for return values of doXXXParsing functions
908  parserstate_t parserstate;
909  uint8_t commandBuffer[ATCMD_LENGTH];
910 
911  //parameters for different tasks
912  taskMouseConfig_t *cmdMouse = malloc(sizeof(taskMouseConfig_t));
913  taskKeyboardConfig_t *cmdKeyboard = malloc(sizeof(taskKeyboardConfig_t));
914  taskNoParameterConfig_t *cmdNoConfig = malloc(sizeof(taskNoParameterConfig_t));
915  taskConfigSwitcherConfig_t *cmdCfgSwitcher = malloc(sizeof(taskConfigSwitcherConfig_t));
916  //taskJoystickConfig_t *cmdJoystick = malloc(sizeof(taskJoystickConfig_t));
917 
918  //check if we have all our pointers
919  if(cmdMouse == NULL || cmdKeyboard == NULL /*|| cmdJoystick == NULL */)
920  {
921  ESP_LOGE(LOG_TAG,"Cannot malloc memory for command parsing, EXIT!");
922  vTaskDelete(NULL);
923  return;
924  }
925 
926  while(1)
927  {
928  if(queuesready)
929  {
930  //wait 30ms, to be nice to other tasks
931  //vTaskDelay(30/portTICK_PERIOD_MS);
932 
933  //wait for incoming data
934  received = halSerialReceiveUSBSerial(commandBuffer,ATCMD_LENGTH);
935  //check received data at least for length
936  if(received < 5)
937  {
938  //special command "AT" without further command:
939  if(received >= 2 && memcmp(commandBuffer,"AT",2) == 0)
940  {
941  halSerialSendUSBSerial(HAL_SERIAL_TX_TO_CDC,"OK\r\n",5,100);
942  halSerialFlushRX();
943  continue;
944  }
945  if(received > 0)
946  {
947  ESP_LOGW(LOG_TAG,"Invalid AT commandlength %d",received);
948  ESP_LOG_BUFFER_HEXDUMP(LOG_TAG,commandBuffer,received,ESP_LOG_DEBUG);
949  }
950  halSerialFlushRX();
951  continue;
952  }
953 
954  //do parsing :-)
955  //simplified into smaller functions, to make reading easy.
956  //each parser returns either:
957  //0 if the command was not consumed (not a valid command for this part)
958  //1 if the command was consumed and needs to be forwarded to a task (singleshot/VB reload)
959  //2 if the command was consumed but no further action is needed
960 
961  //the variable requestVBUpdate which is used to determine if an
962  //AT command should be triggered as singleshot (== VB_SINGLESHOT)
963  //or should be assigned to a VB.
964 
965  //mouse parsing
966  parserstate = doMouseParsing(commandBuffer,cmdMouse);
967  //storage parsing
968  if(parserstate == UNKNOWNCMD) parserstate = doStorageParsing(commandBuffer,cmdCfgSwitcher);
969  if(parserstate == UNKNOWNCMD) parserstate = doKeyboardParsing(commandBuffer,cmdKeyboard,received);
970  if(parserstate == UNKNOWNCMD) parserstate = doGeneralCmdParsing(commandBuffer);
971  if(parserstate == UNKNOWNCMD) parserstate = doInfraredParsing(commandBuffer);
972  if(parserstate == UNKNOWNCMD) parserstate = doMouthpieceSettingsParsing(commandBuffer,cmdNoConfig);
973 
974  switch(parserstate)
975  {
976  case UNKNOWNCMD:
977  break;
978  case NOACTION:
979  //not in singleshot, but parsing parts didn't request anything?!?
980  //maybe wrong combination of AT BM and another AT command
981  if(requestVBUpdate != VB_SINGLESHOT)
982  {
983  ESP_LOGE(LOG_TAG,"Error assigning this AT command to a button!");
984  requestVBUpdate = VB_SINGLESHOT;
985  continue;
986  } else {
987  continue;
988  }
989  break;
990  //the AT command tells us to trigger a task or save an instance to a VB
991  case TRIGGERTASK:
992  if(requestVBUpdate == VB_SINGLESHOT)
993  {
994  //just trigger as singleshot
995  if(requestVBTask != NULL && requestVBParameter != NULL)
996  {
997  //call in singleshot mode
998  requestVBTask(requestVBParameter);
999  //reset the VB task callback
1000  requestVBTask = NULL;
1001  continue;
1002  } else {
1003  ESP_LOGE(LOG_TAG,"Should trigger a singleshot, but task or params are NULL");
1004  }
1005  } else {
1006  //attach to VB
1007  //allocate new memory (the parameter pointers might be differ each time)
1008  void *vbconfigparam = malloc(requestVBParameterSize);
1009  if(vbconfigparam != NULL)
1010  {
1011  //copy to new memory, which will be permanent after an config update
1012  memcpy(vbconfigparam,requestVBParameter,requestVBParameterSize);
1013  } else {
1014  ESP_LOGE(LOG_TAG,"Error allocating VB config!");
1015  }
1016  //trigger VB config update (which loads the config and triggers the update in config_switcher.c)
1017  if(configUpdateVB(vbconfigparam,requestVBType,requestVBUpdate) != ESP_OK)
1018  {
1019  ESP_LOGE(LOG_TAG,"Error updating VB config!");
1020  } else {
1021  ESP_LOGI(LOG_TAG,"VB %d updated in config!",requestVBUpdate);
1022  ESP_LOG_BUFFER_HEXDUMP(LOG_TAG,vbconfigparam,requestVBParameterSize,ESP_LOG_DEBUG);
1023  }
1024  //reset to singleshot mode
1025  requestVBUpdate = VB_SINGLESHOT;
1026  continue;
1027  }
1028  break;
1029  case WAITFORNEWATCMD:
1030  break;
1031  default: break;
1032  }
1033 
1034  //if a general config update is required
1035  if(requestUpdate != 0)
1036  {
1037  if(configUpdate() != ESP_OK)
1038  {
1039  ESP_LOGE(LOG_TAG,"Error updating general config!");
1040  }
1041  requestUpdate = 0;
1042  continue;
1043  }
1044 
1045  //after all that parsing & config updates
1046  //we either have a parserstate == 0 here -> no parser found
1047  //or we consumed this command.
1048  if(parserstate != UNKNOWNCMD) continue;
1049 
1050  //check reply by LPC USB chip:
1051  if(memcmp(commandBuffer,"__OK__",6) == 0) continue;
1052 
1053  if(memcmp(commandBuffer,"_parameter error",16) == 0)
1054  {
1055  ESP_LOGE(LOG_TAG,"USB reply: parameter error");
1056  continue;
1057  }
1058  if(memcmp(commandBuffer,"_unknown command",16) == 0)
1059  {
1060  ESP_LOGE(LOG_TAG,"USB reply: unknown cmd");
1061  continue;
1062  }
1063  if(memcmp(commandBuffer,"_unknown error code",19) == 0)
1064  {
1065  ESP_LOGE(LOG_TAG,"USB reply: unknown error code");
1066  continue;
1067  }
1068 
1069  //if we are here, no parser was finding commands
1070  ESP_LOGW(LOG_TAG,"Invalid AT cmd (%d characters), flushing:",received);
1071  ESP_LOG_BUFFER_CHAR_LEVEL(LOG_TAG,commandBuffer,received, ESP_LOG_WARN);
1072  halSerialFlushRX();
1073  //ESP_LOG_BUFFER_HEXDUMP(LOG_TAG,commandBuffer,received,ESP_LOG_DEBUG);
1074  halSerialSendUSBSerial(HAL_SERIAL_TX_TO_CDC,"?\r\n",2,100);
1075 
1076  } else {
1077  //check again for initialized queues
1078  ESP_LOGE(LOG_TAG,"Queues uninitialized, rechecking in 1s");
1079  vTaskDelay(1000/portTICK_PERIOD_MS);
1080  //re-check
1081  queuesready = checkqueues();
1082  }
1083  }
