122 uint8_t doMouthpieceSettingsParsing(uint8_t *cmdBuffer)
123 {
124  generalConfig_t *currentcfg = configGetCurrent();
125 
126  /*++++ calibrate mouthpiece ++++*/
127  if(CMD("AT CA"))
128  {
129  ESP_LOGI(LOG_TAG,"Calibrate");
130  return 1;
131  }
132 
133  /*++++ stop report raw values ++++*/
134  if(CMD("AT ER"))
135  {
136  currentcfg->adc.reportraw = 0;
137  requestUpdate = 1;
138  ESP_LOGI(LOG_TAG,"Stop reporting raw values");
139  return 2;
140  }
141  /*++++ start report raw values to serial ++++*/
142  if(CMD("AT SR"))
143  {
144  currentcfg->adc.reportraw = 1;
145  requestUpdate = 1;
146  ESP_LOGI(LOG_TAG,"Start reporting raw values");
147  return 2;
148  }
149 
150  /*++++ mouthpiece mode ++++*/
151  if(CMD("AT MM"))
152  {
153  //assign to gain value
154  switch(cmdBuffer[6])
155  {
156  case '0': currentcfg->adc.mode = THRESHOLD; requestUpdate = 1; return 2;
157  case '1': currentcfg->adc.mode = MOUSE; requestUpdate = 1; return 2;
158  case '2': currentcfg->adc.mode = JOYSTICK; requestUpdate = 1; return 2;
159  default: sendErrorBack("Mode is 0,1 or 2"); return 2;
160  }
161  }
162 
163  /*++++ mouthpiece mode - switch ++++*/
164  if(CMD("AT SW"))
165  {
166  switch(currentcfg->adc.mode)
167  {
168  case MOUSE: currentcfg->adc.mode = THRESHOLD; break;
169  case THRESHOLD: currentcfg->adc.mode = MOUSE; break;
170  case JOYSTICK: sendErrorBack("AT SW switches between mouse and threshold only"); break;
171  }
172  requestUpdate = 1;
173  return 2;
174  }
175 
176  /*++++ mouthpiece gain ++++*/
177  //AT GU, GD, GL, GR
178  if(CMD4("AT G"))
179  {
180  unsigned int param = strtol((char*)&(cmdBuffer[5]),NULL,10);
181  ESP_LOGI(LOG_TAG,"Gain %c, %d",cmdBuffer[4],param);
182  if(param > 100)
183  {
184  sendErrorBack("Gain is 0-100");
185  return 0;
186  } else {
187  //assign to gain value
188  switch(cmdBuffer[4])
189  {
190  case 'U': currentcfg->adc.gain[0] = param; requestUpdate = 1; return 2;
191  case 'D': currentcfg->adc.gain[1] = param; requestUpdate = 1; return 2;
192  case 'L': currentcfg->adc.gain[2] = param; requestUpdate = 1; return 2;
193  case 'R': currentcfg->adc.gain[3] = param; requestUpdate = 1; return 2;
194  default: return 0;
195  }
196  }
197  }
198 
199  /*++++ mouthpiece sensitivity/acceleration ++++*/
200  //AT AX, AY, AC
201  if(CMD4("AT A"))
202  {
203  unsigned int param = strtol((char*)&(cmdBuffer[5]),NULL,10);
204  ESP_LOGI(LOG_TAG,"Sensitivity/accel %c, %d",cmdBuffer[4],param);
205  if(param > 100)
206  {
207  sendErrorBack("Sensitivity/accel is 0-100");
208  return 0;
209  } else {
210  //assign to sensitivity/acceleration value
211  switch(cmdBuffer[4])
212  {
213  case 'X': currentcfg->adc.sensitivity_x = param; requestUpdate = 1; return 2;
214  case 'Y': currentcfg->adc.sensitivity_x = param; requestUpdate = 1; return 2;
215  case 'C': currentcfg->adc.acceleration = param; requestUpdate = 1; return 2;
216  default: return 0;
217  }
218  }
219  }
220 
221  /*++++ threshold sip/puff++++*/
222  //AT TS/TP
223  if(CMD4("AT T"))
224  {
225  unsigned int param = strtol((char*)&(cmdBuffer[5]),NULL,10);
226  ESP_LOGI(LOG_TAG,"Threshold %c, %d",cmdBuffer[4],param);
227  switch(cmdBuffer[4])
228  {
229  case 'S':
230  if(param > 512)
231  {
232  sendErrorBack("Threshold sip is 0-512");
233  return 0;
234  } else {
235  currentcfg->adc.threshold_sip = param;
236  requestUpdate = 1;
237  return 2;
238  }
239  break;
240 
241  case 'T':
242  if(param < 512 || param > 1023)
243  {
244  sendErrorBack("Threshold puff is 512-1023");
245  return 0;
246  } else {
247  currentcfg->adc.threshold_puff = param;
248  requestUpdate = 1;
249  return 2;
250  }
251  break;
252  default: return 0;
253  }
254  }
255 
256  /*++++ threshold strong sip/puff++++*/
257  //AT TS/TP
258  if(CMD4("AT S"))
259  {
260  unsigned int param = strtol((char*)&(cmdBuffer[5]),NULL,10);
261  ESP_LOGI(LOG_TAG,"Threshold strong %c, %d",cmdBuffer[4],param);
262  switch(cmdBuffer[4])
263  {
264  case 'S':
265  if(param > 512)
266  {
267  sendErrorBack("Threshold strong sip is 0-512");
268  return 0;
269  } else {
270  currentcfg->adc.threshold_strongsip = param;
271  requestUpdate = 1;
272  return 2;
273  }
274  break;
275 
276  case 'T':
277  if(param < 512 || param > 1023)
278  {
279  sendErrorBack("Threshold strong puff is 512-1023");
280  return 0;
281  } else {
282  currentcfg->adc.threshold_strongpuff = param;
283  requestUpdate = 1;
284  return 2;
285  }
286  break;
287  default: return 0;
288  }
289  }
290 
291  //not consumed, no command found for mouthpiece settings
292  return 0;

309 uint8_t doGeneralCmdParsing(uint8_t *cmdBuffer)
310 {
311  uint16_t param = 0;
312  uint8_t param8 = 0;
313  generalConfig_t *currentcfg;
314 
315  currentcfg = configGetCurrent();
316  if(currentcfg == NULL)
317  {
318  ESP_LOGE(LOG_TAG,"Current config is null, cannot update general cmd");
319  return 0;
320  }
321 
322 
323  /*++++ AT RA ++++*/
324  if(CMD("AT RA")) {
325  //reset the reports (keyboard only, excepting all other parts)
326  halBLEReset(0xFE);
327  halSerialReset(0xFE);
328  return 1;
329  }
330 
331  /*++++ AT KL ++++*/
332  if(CMD("AT KL")) {
333  param8 = strtol((char*)&(cmdBuffer[6]),NULL,10);
334  if(param8 < LAYOUT_MAX)
335  {
336  ESP_LOGI(LOG_TAG,"Changed locale from %d to %d",currentcfg->locale,param8);
337  currentcfg->locale = param8;
338  requestUpdate = 1;
339  return 1;
340  } else {
341  sendErrorBack("Locale out of range");
342  return 0;
343  }
344  }
345 
346  /*++++ AT ID ++++*/
347  if(CMD("AT ID")) {
348  uint32_t sent = halSerialSendUSBSerial(HAL_SERIAL_TX_TO_CDC,IDSTRING,sizeof(IDSTRING),20);
349  if(sent != sizeof(IDSTRING))
350  {
351  ESP_LOGE(LOG_TAG,"Error sending response of AT ID");
352  } else return 1;
353  }
354  /*++++ AT DE ++++*/
355  if(CMD("AT DE")) {
356  uint32_t tid;
357  if(halStorageStartTransaction(&tid,20) != ESP_OK)
358  {
359  return 0;
360  } else {
361  if(halStorageDeleteSlot(0,tid) != ESP_OK)
362  {
363  sendErrorBack("Error deleting all slots");
364  return 0;
365  } else {
366  return 1;
367  }
368  }
369  }
370  /*++++ AT BM ++++*/
371  if(CMD("AT BM")) {
372  param = strtol((char*)&(cmdBuffer[6]),NULL,10);
373  if(param >= VB_MAX)
374  {
375  sendErrorBack("VB nr too high");
376  return 0;
377  } else {
378  requestVBUpdate = param;
379  }
380  }
381 
382  /*++++ AT DL ++++*/
383  if(CMD("AT DL")) {
384  uint32_t tid;
385  uint8_t slotnumber;
386  slotnumber = strtol((char*)&(cmdBuffer[6]),NULL,10);
387  if(halStorageStartTransaction(&tid,20) != ESP_OK)
388  {
389  return 0;
390  } else {
391  if(halStorageDeleteSlot(slotnumber,tid) != ESP_OK)
392  {
393  sendErrorBack("Error deleting slot");
394  return 0;
395  } else {
396  return 1;
397  }
398  }
399  }
400  /*++++ AT BT ++++*/
401  if(CMD("AT BT")) {
402  param = strtol((char*)&(cmdBuffer[6]),NULL,10);
403  switch(param)
404  {
405  case 1:
406  currentcfg->ble_active = 0;
407  currentcfg->usb_active = 1;
408  requestUpdate = 1;
409  break;
410  case 2:
411  currentcfg->ble_active = 1;
412  currentcfg->usb_active = 0;
413  requestUpdate = 1;
414  break;
415  case 3:
416  currentcfg->ble_active = 1;
417  currentcfg->usb_active = 1;
418  requestUpdate = 1;
419  break;
420  default: sendErrorBack("AT BT param"); return 0;
421  }
422  }
423 
424  //not consumed, no general command found
425  return 0;
426 }

428 uint8_t doKeyboardParsing(uint8_t *cmdBuffer, taskKeyboardConfig_t *kbdinstance, int length)
429 {
430  //clear any previous data
431  memset(kbdinstance,0,sizeof(taskKeyboardConfig_t));
432 
433  //use global virtual button (normally VB_SINGLESHOT,
434  //can be changed by "AT BM"
435  kbdinstance->virtualButton = requestVBUpdate;
436  int offset = 0;
437  int offsetOut = 0;
438  uint8_t deadkeyfirst = 0;
439  uint8_t modifier = 0;
440  uint8_t keycode = 0;
441 
442  //get general config (used to get the locale)
443  generalConfig_t *currentcfg;
444  currentcfg = configGetCurrent();
445 
446  if(currentcfg == NULL)
447  {
448  ESP_LOGE(LOG_TAG,"Major error, general config is NULL");
449  return 0;
450  }
451 
452  /*++++ AT KW ++++*/
453  if(CMD("AT KW")) {
454  //set keyboard to write
455  kbdinstance->type = WRITE;
456  //remove \r & \n
457  //cmdBuffer[length-2] = 0;
458  //cmdBuffer[length-1] = 0;
459 
460  //save each byte from KW string to keyboard instance
461  //offset must be less then buffer length - 6 bytes for "AT KW "
462  while(offset <= (length - 5))
463  {
464 
465 
466  //terminate...
467  if(cmdBuffer[offset+6] == '\r' || cmdBuffer[offset+6] == '\n' || \
468  cmdBuffer[offset+6] == 0) break;
469 
470  //parse ASCII/unicode to keycode sequence
471  keycode = unicode_to_keycode(cmdBuffer[offset+6], currentcfg->locale);
472  deadkeyfirst = deadkey_to_keycode(keycode,currentcfg->locale);
473  if(deadkeyfirst != 0) deadkeyfirst = keycode_to_key(deadkeyfirst);
474  modifier = keycode_to_modifier(keycode, currentcfg->locale);
475  keycode = keycode_to_key(keycode);
476 
477  //if a keycode is found
478  if(keycode != 0)
479  {
480  //is a deadkey necessary?
481  if(deadkeyfirst != 0)
482  {
483  kbdinstance->keycodes_text[offsetOut] = deadkeyfirst;
484  offsetOut++;
485  ESP_LOGD(LOG_TAG, "Deadkey 0x%X@%d",deadkeyfirst,offsetOut);
486  }
487 
488  //save keycode + modifier
489  kbdinstance->keycodes_text[offsetOut] = keycode | (modifier << 8);
490  ESP_LOGD(LOG_TAG, "Keycode 0x%X@%d, modifier: 0x%X",keycode,offsetOut,modifier);
491  offsetOut++;
492  } else {
493  ESP_LOGD(LOG_TAG, "Need another byte (unicode)");
494  }
495  offset++;
496 
497  if(offset == (TASK_KEYBOARD_PARAMETERLENGTH - 1))
498  {
499  sendErrorBack("AT KW parameter too long");
500  return 0;
501  }
502  }
503  //terminate keycode array with 0
504  ESP_LOGD(LOG_TAG,"Terminating @%d with 0",offsetOut);
505  kbdinstance->keycodes_text[offsetOut] = 0;
506  return 1;
507  }
508 
509  /*++++ AT KP + KH + KR ++++*/
510  // (all commands with key identifiers)
511  if(CMD4("AT K")) {
512  //set keyboard instance accordingly
513  switch(cmdBuffer[4])
514  {
515  //KP: press with VB press flag, release with VB release flag
516  case 'P': kbdinstance->type = PRESS_RELEASE_BUTTON; break;
517  //KR: just release
518  case 'R': kbdinstance->type = RELEASE; break;
519  //KH: just press (hold)
520  case 'H': kbdinstance->type = PRESS; break;
521  default: return 0; break;
522  }
523 
524  //remove \r & \n
525  cmdBuffer[length-2] = ' ';
526  cmdBuffer[length-1] = ' ';
527 
528  char *pch;
529  uint8_t cnt = 0;
530  uint16_t keycode = 0;
531  pch = strpbrk((char *)&cmdBuffer[5]," ");
532  while (pch != NULL)
533  {
534  ESP_LOGD(LOG_TAG,"Token: %s",pch+1);
535 
536  //check if token seems to be a KEY_*
537  if(memcmp(pch+1,"KEY_",4) != 0)
538  {
539  ESP_LOGW(LOG_TAG,"Not a valid KEY_* identifier");
540  } else {
541  //parse identifier to keycode
542  keycode = parseIdentifierToKeycode(pch+1);
543 
544  if(keycode != 0)
545  {
546  //if found, save...
547  ESP_LOGD(LOG_TAG,"Keycode: %d",keycode);
548 
549  //get deadkeys
550  uint16_t deadk = 0;
551  deadk = deadkey_to_keycode(keycode, currentcfg->locale);
552  //found a deadkey
553  if(deadk != 0)
554  {
555  //save deadkey's keycode & modifier
556  kbdinstance->keycodes_text[cnt] = keycode_to_key(deadk) | \
557  (keycode_to_modifier(deadk, currentcfg->locale) << 8);
558  ESP_LOGI(LOG_TAG,"Deadkey (keycode + modifier): %4X @%d",kbdinstance->keycodes_text[cnt], cnt);
559  cnt++;
560  }
561 
562  //KEY_ * identifiers are either a key or a modifier
563  //determine which type and save accordingly:
564  if(keycode_is_modifier(keycode))
565  {
566  kbdinstance->keycodes_text[cnt] = keycode_to_key(keycode) << 8;
567  } else {
568  kbdinstance->keycodes_text[cnt] = keycode_to_key(keycode);
569  }
570  ESP_LOGI(LOG_TAG,"Keycode + modifier: %4X @ %d",kbdinstance->keycodes_text[cnt], cnt);
571  cnt++;
572  if(cnt == (TASK_KEYBOARD_PARAMETERLENGTH - 1))
573  {
574  sendErrorBack("AT KP/KH/KR parameter too long");
575  return 0;
576  }
577  } else {
578  ESP_LOGW(LOG_TAG,"No keycode found for this token.");
579  }
580  }
581  //split into tokens
582  pch = strpbrk(pch+1, " ");
583  }
584  //check if any key identifiers were found
585  if(cnt == 0)
586  {
587  sendErrorBack("No KEY_ identifiers found");
588  return 0;
589  } else {
590  //terminate keycode array with 0
591  ESP_LOGD(LOG_TAG,"Terminating @%d with 0",cnt);
592  kbdinstance->keycodes_text[cnt] = 0;
593  return 1;
594  }
595  }
596 
597  //not consumed, no command found for keyboard
598  return 0;
599 }

601 uint8_t doMouseParsing(uint8_t *cmdBuffer, taskMouseConfig_t *mouseinstance)
602 {
603  //clear any previous data
604  memset(mouseinstance,0,sizeof(taskMouseConfig_t));
605 
606  //use global virtual button (normally VB_SINGLESHOT,
607  //can be changed by "AT BM"
608  mouseinstance->virtualButton = requestVBUpdate;
609  int steps = 3;
610 
611  /*++++ mouse clicks ++++*/
612  //AT CL, AT CR, AT CM, AT CD
613  if(CMD4("AT C"))
614  {
615  mouseinstance->actionparam = M_CLICK;
616  switch(cmdBuffer[4])
617  {
618  //do single clicks (left,right,middle)
619  case 'L': mouseinstance->type = LEFT; return 1;
620  case 'R': mouseinstance->type = RIGHT; return 1;
621  case 'M': mouseinstance->type = MIDDLE; return 1;
622  //do left double click
623  case 'D':
624  mouseinstance->type = LEFT;
625  mouseinstance->actionparam = M_DOUBLE;
626  return 1;
627  //not an AT C? command for mouse, return 0 (not consumed)
628  default: return 0;
629  }
630  }
631 
632  /*++++ mouse wheel up/down; set stepsize ++++*/
633  if(CMD4("AT W"))
634  {
635  mouseinstance->type = WHEEL;
636  mouseinstance->actionparam = M_UNUSED;
637  switch(cmdBuffer[4])
638  {
639  //move mouse wheel up/down
640  case 'U': mouseinstance->actionvalue = mouse_get_wheel(); return 1;
641  case 'D': mouseinstance->actionvalue = -mouse_get_wheel(); return 1;
642  //set mouse wheel stepsize. If unsuccessful, default will return 0
643  case 'S':
644  steps = strtol((char*)&(cmdBuffer[6]),NULL,10);
645  if(steps > 127 || steps < 127)
646  {
647  sendErrorBack("Wheel size out of range! (-127 to 127)");
648  return 0;
649  } else {
650  mouse_set_wheel(steps);
651  ESP_LOGI(LOG_TAG,"Setting mouse wheel steps: %d",steps);
652  return 2;
653  }
654  default: return 0;
655  }
656  }
657 
658  /*++++ mouse button press ++++*/
659  //AT PL, AT PR, AT PM
660  if(CMD4("AT P"))
661  {
662  mouseinstance->actionparam = M_HOLD;
663  switch(cmdBuffer[4])
664  {
665  case 'L': mouseinstance->type = LEFT; return 1;
666  case 'R': mouseinstance->type = RIGHT; return 1;
667  case 'M': mouseinstance->type = MIDDLE; return 1;
668  default: return 0;
669  }
670  }
671 
672  /*++++ mouse button release ++++*/
673  //AT RL, AT RR, AT RM
674  if(CMD4("AT R"))
675  {
676  mouseinstance->actionparam = M_RELEASE;
677  switch(cmdBuffer[4])
678  {
679  case 'L': mouseinstance->type = LEFT; return 1;
680  case 'R': mouseinstance->type = RIGHT; return 1;
681  case 'M': mouseinstance->type = MIDDLE; return 1;
682  default: return 0;
683  }
684  }
685 
686  /*++++ mouse move ++++*/
687  //AT RL, AT RR, AT RM
688  if(CMD4("AT M"))
689  {
690  mouseinstance->actionparam = M_UNUSED;
691  //mouseinstance->actionvalue = atoi((char*)&(cmdBuffer[6]));
692  //mouseinstance->actionvalue = strtoimax((char*)&(cmdBuffer[6]),&endBuf,10);
693  int param = strtol((char*)&(cmdBuffer[5]),NULL,10);
694  ESP_LOGI(LOG_TAG,"Mouse move %c, %d",cmdBuffer[4],mouseinstance->actionvalue);
695  if(param > 127 && param < -127)
696  {
697  ESP_LOGW(LOG_TAG,"AT MX parameter limit -127 - 127");
698  return 0;
699  } else {
700  mouseinstance->actionvalue = param;
701  }
702  switch(cmdBuffer[4])
703  {
704  case 'X': mouseinstance->type = X; return 1;
705  case 'Y': mouseinstance->type = Y; return 1;
706  default: return 0;
707  }
708  }
709 
710  return 0;
711 }

713 void task_commands(void *params)
714 {
715  uint8_t queuesready = checkqueues();
716  int received;
717  //used for return values of doXXXParsing functions
718  uint8_t parserstate;
719  uint8_t commandBuffer[ATCMD_LENGTH];
720 
721  //function pointer to the task function which will eventuall be called
722  //in singleshot mode
723  void (*requestVBTask)(void *) = NULL;
724 
725  // parameter size which needs to be allocated for a VB config update
726  size_t requestVBParameterSize = 0;
727 
728  //pointer to VB parameter struct for VB config udpate
729  void *requestVBParameter = NULL;
730 
731  //reload parameter
732  command_type_t requestVBType = T_NOFUNCTION;
733 
734 
735  //parameters for different tasks
736  taskMouseConfig_t *cmdMouse = malloc(sizeof(taskMouseConfig_t));
737  taskKeyboardConfig_t *cmdKeyboard = malloc(sizeof(taskKeyboardConfig_t));
738  taskNoParameterConfig_t *cmdNoConfig = malloc(sizeof(taskNoParameterConfig_t));
739  //taskJoystickConfig_t *cmdJoystick = malloc(sizeof(taskJoystickConfig_t));
740 
741  //check if we have all our pointers
742  if(cmdMouse == NULL || cmdKeyboard == NULL /*|| cmdJoystick == NULL */)
743  {
744  ESP_LOGE(LOG_TAG,"Cannot malloc memory for command parsing, EXIT!");
745  vTaskDelete(NULL);
746  return;
747  }
748 
749  while(1)
750  {
751  if(queuesready)
